46  public abstract class AbstractConvertSubcommand
47      extends AbstractTerminalCommand {
48    private static final Logger LOGGER = LogManager.getLogger(AbstractConvertSubcommand.class);
49  
50    @NonNull
51    private static final String COMMAND = "convert";
52    @NonNull
53    private static final List<ExtraArgument> EXTRA_ARGUMENTS = ObjectUtils.notNull(List.of(
54        new DefaultExtraArgument("source-file-or-URL", true),
55        new DefaultExtraArgument("destination-file", false)));
56  
57    @NonNull
58    private static final Option OVERWRITE_OPTION = ObjectUtils.notNull(
59        Option.builder()
60            .longOpt("overwrite")
61            .desc("overwrite the destination if it exists")
62            .build());
63    @NonNull
64    private static final Option TO_OPTION = ObjectUtils.notNull(
65        Option.builder()
66            .longOpt("to")
67            .required()
68            .hasArg().argName("FORMAT")
69            .desc("convert to format: xml, json, or yaml")
70            .build());
71  
72    @Override
73    public String getName() {
74      return COMMAND;
75    }
76  
77    @Override
78    public Collection<? extends Option> gatherOptions() {
79      return ObjectUtils.notNull(List.of(
80          OVERWRITE_OPTION,
81          TO_OPTION));
82    }
83  
84    @Override
85    public List<ExtraArgument> getExtraArguments() {
86      return EXTRA_ARGUMENTS;
87    }
88  
89    @SuppressWarnings("PMD.PreserveStackTrace") // intended
90    @Override
91    public void validateOptions(CallingContext callingContext, CommandLine cmdLine) throws InvalidArgumentException {
92  
93      try {
94        String toFormatText = cmdLine.getOptionValue(TO_OPTION);
95        Format.valueOf(toFormatText.toUpperCase(Locale.ROOT));
96      } catch (IllegalArgumentException ex) {
97        InvalidArgumentException newEx = new InvalidArgumentException(
98            String.format("Invalid '%s' argument. The format must be one of: %s.",
99                OptionUtils.toArgument(TO_OPTION),
100               Format.names().stream()
101                   .collect(CustomCollectors.joiningWithOxfordComma("and"))));
102       newEx.setOption(TO_OPTION);
103       newEx.addSuppressed(ex);
104       throw newEx;
105     }
106 
107     List<String> extraArgs = cmdLine.getArgList();
108     if (extraArgs.isEmpty() || extraArgs.size() > 2) {
109       throw new InvalidArgumentException("Illegal number of arguments.");
110     }
111   }
112 
113   protected abstract static class AbstractConversionCommandExecutor
114       extends AbstractCommandExecutor {
115 
116     public AbstractConversionCommandExecutor(
117         @NonNull CallingContext callingContext,
118         @NonNull CommandLine commandLine) {
119       super(callingContext, commandLine);
120     }
121 
122     @NonNull
123     protected abstract IBindingContext getBindingContext();
124 
125     @SuppressWarnings({
126         "PMD.OnlyOneReturn", // readability
127         "PMD.CyclomaticComplexity", "PMD.CognitiveComplexity" // reasonable
128     })
129     @Override
130     public ExitStatus execute() {
131       CommandLine cmdLine = getCommandLine();
132 
133       List<String> extraArgs = cmdLine.getArgList();
134 
135       Path destination = null;
136       if (extraArgs.size() > 1) {
137         destination = Paths.get(extraArgs.get(1)).toAbsolutePath();
138       }
139 
140       if (destination != null) {
141         if (Files.exists(destination)) {
142           if (!cmdLine.hasOption(OVERWRITE_OPTION)) {
143             return ExitCode.INVALID_ARGUMENTS.exitMessage(
144                 String.format("The provided destination '%s' already exists and the '%s' option was not provided.",
145                     destination,
146                     OptionUtils.toArgument(OVERWRITE_OPTION)));
147           }
148           if (!Files.isWritable(destination)) {
149             return ExitCode.IO_ERROR.exitMessage(
150                 "The provided destination '" + destination + "' is not writable.");
151           }
152         } else {
153           Path parent = destination.getParent();
154           if (parent != null) {
155             try {
156               Files.createDirectories(parent);
157             } catch (IOException ex) {
158               return ExitCode.INVALID_TARGET.exit().withThrowable(ex); // NOPMD readability
159             }
160           }
161         }
162       }
163 
164       String sourceName = extraArgs.get(0);
165       URI source;
166       URI cwd = Paths.get("").toAbsolutePath().toUri();
167       try {
168         source = UriUtils.toUri(sourceName, cwd);
169       } catch (URISyntaxException ex) {
170         return ExitCode.IO_ERROR.exitMessage("Cannot load source '%s' as it is not a valid file or URI.")
171             .withThrowable(ex);
172       }
173       assert source != null;
174 
175       String toFormatText = cmdLine.getOptionValue(TO_OPTION);
176       Format toFormat = Format.valueOf(toFormatText.toUpperCase(Locale.ROOT));
177 
178       IBindingContext bindingContext = getBindingContext();
179       try {
180         IBoundLoader loader = bindingContext.newBoundLoader();
181         if (LOGGER.isInfoEnabled()) {
182           LOGGER.info("Converting '{}'.", source);
183         }
184 
185         if (destination == null) {
186           // write to STDOUT
187           OutputStreamWriter writer = new OutputStreamWriter(System.out, StandardCharsets.UTF_8);
188           handleConversion(source, toFormat, writer, loader);
189         } else {
190           try (Writer writer = Files.newBufferedWriter(
191               destination,
192               StandardCharsets.UTF_8,
193               StandardOpenOption.CREATE,
194               StandardOpenOption.WRITE,
195               StandardOpenOption.TRUNCATE_EXISTING)) {
196             assert writer != null;
197             handleConversion(source, toFormat, writer, loader);
198           }
199         }
200       } catch (IOException | IllegalArgumentException ex) {
201         return ExitCode.PROCESSING_ERROR.exit().withThrowable(ex); // NOPMD readability
202       }
203       if (destination != null && LOGGER.isInfoEnabled()) {
204         LOGGER.info("Generated {} file: {}", toFormat.toString(), destination);
205       }
206       return ExitCode.OK.exit();
207     }
208 
209     protected abstract void handleConversion(
210         @NonNull URI source,
211         @NonNull Format toFormat,
212         @NonNull Writer writer,
213         @NonNull IBoundLoader loader) throws FileNotFoundException, IOException;
214   }
