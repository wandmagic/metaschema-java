48  public class ValidateModuleCommand
49      extends AbstractValidateContentCommand {
50    private static final Logger LOGGER = LogManager.getLogger(ValidateModuleCommand.class);
51    @NonNull
52    private static final String COMMAND = "validate";
53  
54    @Override
55    public String getName() {
56      return COMMAND;
57    }
58  
59    @Override
60    public String getDescription() {
61      return "Validate that the specified Module is well-formed and valid to the Module model";
62    }
63  
64    @Override
65    public ICommandExecutor newExecutor(CallingContext callingContext, CommandLine commandLine) {
66      return new ValidateModuleCommandExecutor(callingContext, commandLine);
67    }
68  
69    private final class ValidateModuleCommandExecutor
70        extends AbstractValidationCommandExecutor {
71      private Path tempDir;
72  
73      private ValidateModuleCommandExecutor(
74          @NonNull CallingContext callingContext,
75          @NonNull CommandLine commandLine) {
76        super(callingContext, commandLine);
77      }
78  
79      private Path getTempDir() throws IOException {
80        if (tempDir == null) {
81          tempDir = Files.createTempDirectory("validation-");
82          tempDir.toFile().deleteOnExit();
83        }
84        return tempDir;
85      }
86  
87      @Override
88      protected IBindingContext getBindingContext(Set<IConstraintSet> constraintSets)
89          throws MetaschemaException, IOException {
90        IBindingContext retval;
91        if (constraintSets.isEmpty()) {
92          retval = IBindingContext.instance();
93        } else {
94          ExternalConstraintsModulePostProcessor postProcessor
95              = new ExternalConstraintsModulePostProcessor(constraintSets);
96          retval = new DefaultBindingContext(CollectionUtil.singletonList(postProcessor));
97        }
98        retval.getBoundDefinitionForClass(METASCHEMA.class);
99        return retval;
100     }
101 
102     @Override
103     public List<Source> getXmlSchemas(@NonNull URL targetResource) throws IOException {
104       List<Source> retval = new LinkedList<>();
105       retval.add(XmlUtil.getStreamSource(
106           ObjectUtils.requireNonNull(
107               ModuleLoader.class.getResource("/schema/xml/metaschema.xsd"),
108               "Unable to load '/schema/xml/metaschema.xsd' on the classpath")));
109       return CollectionUtil.unmodifiableList(retval);
110     }
111 
112     @Override
113     public JSONObject getJsonSchema(@NonNull JSONObject json) throws IOException {
114       IModule module = IBindingContext.instance().registerModule(MetaschemaModelModule.class);
115 
116       Path schemaFile = Files.createTempFile(getTempDir(), "schema-", ".json");
117       assert schemaFile != null;
118       IMutableConfiguration<SchemaGenerationFeature<?>> configuration = new DefaultConfiguration<>();
119       ISchemaGenerator.generateSchema(module, schemaFile, SchemaFormat.JSON, configuration);
120       try (BufferedReader reader = ObjectUtils.notNull(Files.newBufferedReader(schemaFile, StandardCharsets.UTF_8))) {
121         return JsonUtil.toJsonObject(reader);
122       }
123     }
124   }
