6   package gov.nist.secauto.metaschema.core.model.constraint.impl;
7   
8   import gov.nist.secauto.metaschema.core.datatype.markup.MarkupLine;
9   import gov.nist.secauto.metaschema.core.datatype.markup.MarkupMultiline;
10  import gov.nist.secauto.metaschema.core.model.IAttributable;
11  import gov.nist.secauto.metaschema.core.model.constraint.IKeyConstraint;
12  import gov.nist.secauto.metaschema.core.model.constraint.IKeyField;
13  import gov.nist.secauto.metaschema.core.model.constraint.ISource;
14  
15  import java.util.List;
16  import java.util.Map;
17  import java.util.Set;
18  
19  import edu.umd.cs.findbugs.annotations.NonNull;
20  import edu.umd.cs.findbugs.annotations.Nullable;
21  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
22  
23  abstract class AbstractKeyConstraint
24      extends AbstractConstraint
25      implements IKeyConstraint {
26    @NonNull
27    private final List<IKeyField> keyFields;
28  
29    /**
30     * Create a new key-based constraint, which uses a set of key fields to build a
31     * key.
32     *
33     * @param id
34     *          the optional identifier for the constraint
35     * @param formalName
36     *          the constraint's formal name or {@code null} if not provided
37     * @param description
38     *          the constraint's semantic description or {@code null} if not
39     *          provided
40     * @param source
41     *          information about the constraint source
42     * @param level
43     *          the significance of a violation of this constraint
44     * @param target
45     *          the Metapath expression identifying the nodes the constraint targets
46     * @param properties
47     *          a collection of associated properties
48     * @param keyFields
49     *          a list of key fields associated with the constraint
50     * @param remarks
51     *          optional remarks describing the intent of the constraint
52     */
53    @SuppressFBWarnings("CT_CONSTRUCTOR_THROW")
54    protected AbstractKeyConstraint(
55        @Nullable String id,
56        @Nullable String formalName,
57        @Nullable MarkupLine description,
58        @NonNull ISource source,
59        @NonNull Level level,
60        @NonNull String target,
61        @NonNull Map<IAttributable.Key, Set<String>> properties,
62        @NonNull List<IKeyField> keyFields,
63        @Nullable MarkupMultiline remarks) {
64      super(id, formalName, description, source, level, target, properties, remarks);
65      if (keyFields.isEmpty()) {
66        throw new IllegalArgumentException("an empty list of key fields is not allowed");
67      }
68      this.keyFields = keyFields;
69    }
70  
71    @Override
72    public List<IKeyField> getKeyFields() {
73      return keyFields;
74    }
75  }
