6   package gov.nist.secauto.metaschema.core.model;
7   
8   import gov.nist.secauto.metaschema.core.util.ObjectUtils;
9   
10  import edu.umd.cs.findbugs.annotations.NonNull;
11  
12  /**
13   * A base class for an assembly that is a member of a containing model.
14   *
15   * @param <PARENT>
16   *          the Java type of the parent model container for this instance
17   * @param <DEFINITION>
18   *          the Java type of the related assembly definition
19   * @param <INSTANCE>
20   *          the Java type of the implementing instance type
21   * @param <PARENT_DEFINITION>
22   *          the Java type of the containing assembly definition
23   */
24  public abstract class AbstractAssemblyInstance<
25      PARENT extends IContainerModel,
26      DEFINITION extends IAssemblyDefinition,
27      INSTANCE extends IAssemblyInstance,
28      PARENT_DEFINITION extends IAssemblyDefinition>
29      extends AbstractNamedModelInstance<PARENT, PARENT_DEFINITION>
30      implements IAssemblyInstance, IFeatureDefinitionReferenceInstance<DEFINITION, INSTANCE> {
31  
32    /**
33     * Construct a new assembly instance that is contained with the provided parent
34     * container.
35     *
36     * @param parent
37     *          the parent container for this instance
38     */
39    protected AbstractAssemblyInstance(@NonNull PARENT parent) {
40      super(parent);
41    }
42  
43    @Override
44    public DEFINITION getDefinition() {
45      // this should always be not null
46      return ObjectUtils.asType(ObjectUtils.requireNonNull(
47          getContainingModule()
48              .getScopedAssemblyDefinitionByName(getReferencedDefinitionQName())));
49    }
50  
51    /**
52     * Generates a "coordinate" string for the assembly instance.
53     *
54     * @return the coordinate
55     */
56    @SuppressWarnings("null")
57    @Override
58    public String toCoordinates() {
59      IDefinition definition = getDefinition();
60      return String.format("assembly instance %s -> %s in module %s (@%d(%d)",
61          getXmlQName(),
62          definition.getDefinitionQName(),
63          getContainingDefinition().getContainingModule().getShortName(),
64          hashCode(),
65          definition.hashCode());
66    }
67  }
