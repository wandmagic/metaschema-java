6   package gov.nist.secauto.metaschema.core.model;
7   
8   import edu.umd.cs.findbugs.annotations.NonNull;
9   
10  /**
11   * An enumeration that identifies the type of a Metaschema construct.
12   */
13  public enum ModelType {
14    ASSEMBLY("assembly"),
15    FIELD("field"),
16    FLAG("flag"),
17    CHOICE("choice"),
18    CHOICE_GROUP("choice-group");
19  
20    private final String name;
21  
22    ModelType(@NonNull String name) {
23      this.name = name;
24    }
25  
26    @Override
27    public String toString() {
28      return name;
29    }
30  }
