10  import org.apache.commons.lang3.tuple.Pair;

12  import org.apache.xmlbeans.XmlCursor.XmlBookmark;
13  import org.apache.xmlbeans.XmlLineNumber;
14  import org.apache.xmlbeans.XmlObject;

17  import java.util.Collection;
18  import java.util.LinkedHashMap;
19  import java.util.Map;
20  import java.util.concurrent.atomic.AtomicInteger;
21  import java.util.stream.Collectors;
22  
23  import javax.xml.namespace.QName;
24  
25  import edu.umd.cs.findbugs.annotations.NonNull;

28  /**
29   * Supports parsing Metaschema assembly and field XMLBeans objects that contain
30   * other Metaschema objects.
31   *
32   * @param <T>
33   *          the Java type of the state that is passed to the element parsing
34   *          handlers
35   */
36  public class XmlObjectParser<T> {
37    private static final XmlOptions XML_OPTIONS = new XmlOptions().setXPathUseSaxon(false).setXPathUseXmlBeans(true);
38    private final Map<QName, Handler<T>> elementNameToHandlerMap;
39    private final String xpath;
40  
41    private static String generatePath(@NonNull Collection<QName> nodes) {
42      // build a mapping of namespace prefix to namespace
43      AtomicInteger count = new AtomicInteger();
44      Map<String, String> namespaceToPrefixMap = nodes.stream()
45          .map(QName::getNamespaceURI)
46          .distinct()
47          .map(ns -> Pair.of(ns, "m" + count.getAndIncrement()))
48          .collect(Collectors.toMap(
49              Pair::getKey,
50              Pair::getValue,
51              (k1, k2) -> k1,
52              LinkedHashMap::new));
53  
54      // generate namespace declarations using prefix and namespace
55      StringBuilder builder = new StringBuilder(24);
56      namespaceToPrefixMap.entrySet().forEach((entry) -> {
57        builder.append("declare namespace ")
58            .append(entry.getValue())
59            .append("='")
60            .append(entry.getKey())
61            .append("';");
62      });
63  
64      // generate child path
65      builder.append(nodes.stream()
66          .map(qname -> {
67            return new StringBuilder()
68                .append("$this/")
69                .append(namespaceToPrefixMap.get(qname.getNamespaceURI()))
70                .append(':')
71                .append(qname.getLocalPart())
72                .toString();
73          }).collect(Collectors.joining("|")));
74  
75      return builder.toString();
76    }
77  
78    /**
79     * Construct a new XmlObject parser.
80     *
81     * @param elementNameToHandlerMap
82     *          the mapping of element names to associated handlers
83     */
84    public XmlObjectParser(@NonNull Map<QName, Handler<T>> elementNameToHandlerMap) {
85      this.elementNameToHandlerMap = elementNameToHandlerMap;
86      this.xpath = generatePath(ObjectUtils.notNull(elementNameToHandlerMap.keySet()));
87    }
88  
89    private Map<QName, Handler<T>> getElementNameToHandlerMap() {
90      return elementNameToHandlerMap;
91    }
92  
93    private String getXpath() {
94      return xpath;
95    }
96  
97    /**
98     * Get the resource location of the provided object.
99     *
100    * @param obj
101    *          the XMLBeans object to get the location for
102    * @return the resource location or {@code null} if the location is not known
103    */
104   @SuppressWarnings({ "resource", "null" })
105   @Nullable
106   public static String toLocation(@NonNull XmlObject obj) {
107     return toLocation(obj.newCursor());
108   }
109 
110   /**
111    * Get the resource location of the provided cursor.
112    *
113    * @param cursor
114    *          the XMLBeans cursor to get the location for
115    * @return the resource location or {@code null} if the location is not known
116    */
117   @Nullable
118   public static String toLocation(@NonNull XmlCursor cursor) {
119     String retval = null;
120     XmlBookmark bookmark = cursor.getBookmark(XmlLineNumber.class);
121     if (bookmark != null) {
122       StringBuilder locationBuilder = new StringBuilder();
123       XmlLineNumber lineNumber = (XmlLineNumber) bookmark;
124 
125       String source = cursor.documentProperties().getSourceName();
126       if (source != null) {
127         locationBuilder.append(source)
128             .append(':');
129       }
130 
131       locationBuilder.append(lineNumber.getLine())
132           .append(':')
133           .append(lineNumber.getColumn());
134 
135       retval = locationBuilder.toString();
136     }
137     return retval;
138   }
139 
140   /**
141    * Used to determine which parser {@link Handler} implementation to use to parse
142    * the object.
143    * <p>
144    * Subclasses can override this method to implement a more efficient or advanced
145    * detection method.
146    *
147    * @param cursor
148    *          the current XmlCursor location
149    * @param obj
150    *          the strongly typed XmlObject at the current location
151    * @return the identified handler
152    * @throws IllegalStateException
153    *           if a suitable handler cannot be identified
154    */
155   @NonNull
156   protected Handler<T> identifyHandler(@NonNull XmlCursor cursor, @NonNull XmlObject obj) {
157     QName qname = cursor.getName();
158     Handler<T> retval = getElementNameToHandlerMap().get(qname);
159     if (retval == null) {
160       String location = toLocation(cursor);
161       if (location == null) {
162         location = "";
163       } else {
164         location = new StringBuilder()
165             .append(" at location '")
166             .append(location)
167             .append('\'')
168             .toString();
169       }
170       throw new IllegalStateException(String.format("Unhandled node '%s'%s.", qname, location));
171     }
172     return retval;
173   }
174 
175   /**
176    * Parse an XmlObject element tree using the configured child element handlers.
177    *
178    * @param xmlObject
179    *          the XmlObject container to parse
180    * @param state
181    *          parsing state to pass to the handlers
182    * @return the state
183    */
184   public T parse(@NonNull XmlObject xmlObject, T state) {
185     try (XmlCursor cursor = xmlObject.newCursor()) {
186       assert cursor != null;
187       cursor.selectPath(getXpath(), XML_OPTIONS);
188       while (cursor.toNextSelection()) {
189         XmlObject obj = cursor.getObject();
190         assert obj != null;
191         Handler<T> handler = identifyHandler(cursor, obj);
192         handler.handle(obj, state);
193       }
194     }
195     return state;
196   }
197 
198   /**
199    * Provides a common interface for element parsing handlers.
200    *
201    * @param <T>
202    *          the Java type of the state that is passed to the element parsing
203    *          handlers
204    */
205   @FunctionalInterface
206   public interface Handler<T> {
207     /**
208      * Parse the provided {@code obj} using the provided {@code state}.
209      *
210      * @param obj
211      *          the object to parse
212      * @param state
213      *          the state to use for parsing
214      */
215     void handle(@NonNull XmlObject obj, T state);
216   }
