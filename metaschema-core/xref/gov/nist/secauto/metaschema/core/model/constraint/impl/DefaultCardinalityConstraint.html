6   package gov.nist.secauto.metaschema.core.model.constraint.impl;
7   
8   import gov.nist.secauto.metaschema.core.datatype.markup.MarkupLine;
9   import gov.nist.secauto.metaschema.core.datatype.markup.MarkupMultiline;
10  import gov.nist.secauto.metaschema.core.model.IAttributable;
11  import gov.nist.secauto.metaschema.core.model.constraint.ICardinalityConstraint;
12  import gov.nist.secauto.metaschema.core.model.constraint.ISource;
13  
14  import java.util.Map;
15  import java.util.Set;
16  
17  import edu.umd.cs.findbugs.annotations.NonNull;
18  import edu.umd.cs.findbugs.annotations.Nullable;
19  
20  public final class DefaultCardinalityConstraint
21      extends AbstractConstraint
22      implements ICardinalityConstraint {
23    @Nullable
24    private final Integer minOccurs;
25    @Nullable
26    private final Integer maxOccurs;
27  
28    /**
29     * Construct a new cardinality constraint which enforces that the number of
30     * items matching the target fall within the inclusive range described by the
31     * {@code minOccurs} or {@code maxOccurs} values.
32     *
33     * @param id
34     *          the optional identifier for the constraint
35     * @param formalName
36     *          the constraint's formal name or {@code null} if not provided
37     * @param description
38     *          the constraint's semantic description or {@code null} if not
39     *          provided
40     * @param source
41     *          information about the constraint source
42     * @param level
43     *          the significance of a violation of this constraint
44     * @param target
45     *          the Metapath expression identifying the nodes the constraint targets
46     * @param properties
47     *          a collection of associated properties
48     * @param minOccurs
49     *          if provided, the constraint ensures that the count of targets is at
50     *          least this value
51     * @param maxOccurs
52     *          if provided, the constraint ensures that the count of targets is at
53     *          most this value
54     * @param remarks
55     *          optional remarks describing the intent of the constraint
56     */
57    @SuppressWarnings("PMD.ExcessiveParameterList")
58    public DefaultCardinalityConstraint(
59        @Nullable String id,
60        @Nullable String formalName,
61        @Nullable MarkupLine description,
62        @NonNull ISource source,
63        @NonNull Level level,
64        @NonNull String target,
65        @NonNull Map<IAttributable.Key, Set<String>> properties,
66        @Nullable Integer minOccurs,
67        @Nullable Integer maxOccurs,
68        MarkupMultiline remarks) {
69      super(id, formalName, description, source, level, target, properties, remarks);
70      if (minOccurs == null && maxOccurs == null) {
71        throw new IllegalArgumentException("at least one of minOccurs or maxOccurs must be provided");
72      }
73      this.minOccurs = minOccurs;
74      this.maxOccurs = maxOccurs;
75    }
76  
77    @Override
78    public Integer getMinOccurs() {
79      return minOccurs;
80    }
81  
82    @Override
83    public Integer getMaxOccurs() {
84      return maxOccurs;
85    }
86  
87  }
