6   package gov.nist.secauto.metaschema.core.model.impl;
7   
8   import gov.nist.secauto.metaschema.core.model.IContainerFlagSupport;
9   import gov.nist.secauto.metaschema.core.model.IFlagInstance;
10  
11  import java.util.Map;
12  
13  import javax.xml.namespace.QName;
14  
15  import edu.umd.cs.findbugs.annotations.NonNull;
16  import edu.umd.cs.findbugs.annotations.Nullable;
17  
18  public class DefaultContainerFlagSupport<FI extends IFlagInstance> implements IContainerFlagSupport<FI> {
19    @NonNull
20    private final Map<QName, FI> instances;
21    @Nullable
22    private final FI jsonKey;
23  
24    /**
25     * Construct a new flag container using the provided flag instances.
26     *
27     * @param instances
28     *          a collection of flag instances
29     * @param jsonKey
30     *          the JSON key flag instance or {@code null} if no JSON key is
31     *          configured
32     */
33    public DefaultContainerFlagSupport(
34        @NonNull Map<QName, FI> instances,
35        @Nullable FI jsonKey) {
36      this.instances = instances;
37      this.jsonKey = jsonKey;
38    }
39  
40    @Override
41    public Map<QName, FI> getFlagInstanceMap() {
42      return instances;
43    }
44  
45    /**
46     * Retrieves the flag instance to use as as the property name for the containing
47     * object in JSON who's value will be the object containing the flag.
48     *
49     * @return the flag instance if a JSON key is configured, or {@code null}
50     *         otherwise
51     */
52    @Override
53    public FI getJsonKeyFlagInstance() {
54      return jsonKey;
55    }
56  }
