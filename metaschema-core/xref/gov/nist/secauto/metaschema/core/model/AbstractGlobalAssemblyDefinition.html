6   package gov.nist.secauto.metaschema.core.model;
7   
8   import edu.umd.cs.findbugs.annotations.NonNull;
9   
10  /**
11   * A base class for an assembly definition defined globally within a Metaschema
12   * module.
13   *
14   * @param <MODULE>
15   *          the Java type of the containing module
16   * @param <INSTANCE>
17   *          the expected Java type of an instance of this definition
18   * @param <FLAG>
19   *          the expected Java type of flag children
20   * @param <MODEL>
21   *          the expected Java type of model children
22   * @param <NAMED_MODEL>
23   *          the expected Java type of named model children
24   * @param <FIELD>
25   *          the expected Java type of field children
26   * @param <ASSEMBLY>
27   *          the expected Java type of assembly children
28   * @param <CHOICE>
29   *          the expected Java type of choice children
30   * @param <CHOICE_GROUP>
31   *          the expected Java type of choice group children
32   */
33  public abstract class AbstractGlobalAssemblyDefinition<
34      MODULE extends IModule,
35      INSTANCE extends IAssemblyInstance,
36      FLAG extends IFlagInstance,
37      MODEL extends IModelInstanceAbsolute,
38      NAMED_MODEL extends INamedModelInstanceAbsolute,
39      FIELD extends IFieldInstanceAbsolute,
40      ASSEMBLY extends IAssemblyInstanceAbsolute,
41      CHOICE extends IChoiceInstance,
42      CHOICE_GROUP extends IChoiceGroupInstance>
43      extends AbstractGlobalDefinition<MODULE, INSTANCE>
44      implements IAssemblyDefinition, IFeatureContainerFlag<FLAG>, IFeatureContainerModelAssembly<
45          MODEL,
46          NAMED_MODEL,
47          FIELD,
48          ASSEMBLY,
49          CHOICE,
50          CHOICE_GROUP> {
51  
52    /**
53     * Construct a new global assembly definition.
54     *
55     * @param module
56     *          the parent module containing this definition
57     */
58    protected AbstractGlobalAssemblyDefinition(@NonNull MODULE module) {
59      super(module, module::toModelQName);
60    }
61  }
