6   package gov.nist.secauto.metaschema.core.model.constraint.impl;
7   
8   import gov.nist.secauto.metaschema.core.datatype.IDataTypeAdapter;
9   import gov.nist.secauto.metaschema.core.datatype.markup.MarkupLine;
10  import gov.nist.secauto.metaschema.core.datatype.markup.MarkupMultiline;
11  import gov.nist.secauto.metaschema.core.model.IAttributable;
12  import gov.nist.secauto.metaschema.core.model.constraint.IMatchesConstraint;
13  import gov.nist.secauto.metaschema.core.model.constraint.ISource;
14  
15  import java.util.Map;
16  import java.util.Set;
17  import java.util.regex.Pattern;
18  
19  import edu.umd.cs.findbugs.annotations.NonNull;
20  import edu.umd.cs.findbugs.annotations.Nullable;
21  
22  public final class DefaultMatchesConstraint
23      extends AbstractConstraint
24      implements IMatchesConstraint {
25    private final Pattern pattern;
26    private final IDataTypeAdapter<?> dataType;
27  
28    /**
29     * Create a new matches constraint, which enforces a value pattern and/or data
30     * type.
31     *
32     * @param id
33     *          the optional identifier for the constraint
34     * @param formalName
35     *          the constraint's formal name or {@code null} if not provided
36     * @param description
37     *          the constraint's semantic description or {@code null} if not
38     *          provided
39     * @param source
40     *          information about the constraint source
41     * @param level
42     *          the significance of a violation of this constraint
43     * @param target
44     *          the Metapath expression identifying the nodes the constraint targets
45     * @param properties
46     *          a collection of associated properties
47     * @param pattern
48     *          the value pattern to match or {@code null} if there is no match
49     *          pattern
50     * @param dataType
51     *          the value data type to match or {@code null} if there is no match
52     *          data type
53     * @param remarks
54     *          optional remarks describing the intent of the constraint
55     */
56    @SuppressWarnings("PMD.ExcessiveParameterList")
57    public DefaultMatchesConstraint(
58        @Nullable String id,
59        @Nullable String formalName,
60        @Nullable MarkupLine description,
61        @NonNull ISource source,
62        @NonNull Level level,
63        @NonNull String target,
64        @NonNull Map<IAttributable.Key, Set<String>> properties,
65        @Nullable Pattern pattern,
66        @Nullable IDataTypeAdapter<?> dataType,
67        @Nullable MarkupMultiline remarks) {
68      super(id, formalName, description, source, level, target, properties, remarks);
69      if (pattern == null && dataType == null) {
70        throw new IllegalArgumentException("a pattern or data type must be provided");
71      }
72      this.pattern = pattern;
73      this.dataType = dataType;
74    }
75  
76    @Override
77    public Pattern getPattern() {
78      return pattern;
79    }
80  
81    @Override
82    public IDataTypeAdapter<?> getDataType() {
83      return dataType;
84    }
85  
86  }
