6   package gov.nist.secauto.metaschema.core.model;
7   
8   import gov.nist.secauto.metaschema.core.util.ObjectUtils;
9   
10  import edu.umd.cs.findbugs.annotations.NonNull;
11  
12  /**
13   * A base class for an assembly instance defined inline.
14   *
15   * @param <PARENT>
16   *          the Java type of the parent model container for this instance
17   * @param <DEFINITION>
18   *          the Java type of the related assembly definition
19   * @param <INSTANCE>
20   *          the expected Java type of an instance of this definition
21   * @param <PARENT_DEFINITION>
22   *          the Java type of the containing assembly definition
23   * @param <FLAG>
24   *          the expected Java type of flag children
25   */
26  public abstract class AbstractInlineFieldDefinition<
27      PARENT extends IContainerModel,
28      DEFINITION extends IFieldDefinition,
29      INSTANCE extends IFieldInstance,
30      PARENT_DEFINITION extends IAssemblyDefinition,
31      FLAG extends IFlagInstance>
32      extends AbstractNamedModelInstance<PARENT, PARENT_DEFINITION>
33      implements IFieldInstance, IFieldDefinition,
34      IFeatureContainerFlag<FLAG>,
35      IFeatureDefinitionInstanceInlined<DEFINITION, INSTANCE> {
36  
37    /**
38     * Construct a new inline assembly definition.
39     *
40     * @param parent
41     *          the parent model containing this instance
42     */
43    protected AbstractInlineFieldDefinition(@NonNull PARENT parent) {
44      super(parent);
45    }
46  
47    @Override
48    public final DEFINITION getDefinition() {
49      return ObjectUtils.asType(this);
50    }
51  
52    @Override
53    @NonNull
54    public final INSTANCE getInlineInstance() {
55      return ObjectUtils.asType(this);
56    }
57  
58    @Override
59    public final FLAG getJsonKey() {
60      return IFeatureContainerFlag.super.getJsonKey();
61    }
62  }
