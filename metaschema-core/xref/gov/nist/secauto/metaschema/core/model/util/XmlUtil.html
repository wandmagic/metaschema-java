6   package gov.nist.secauto.metaschema.core.model.util;
7   
8   import java.io.IOException;
9   import java.net.URL;
10  
11  import javax.xml.transform.Source;
12  import javax.xml.transform.stream.StreamSource;
13  
14  import edu.umd.cs.findbugs.annotations.NonNull;
15  
16  public final class XmlUtil {
17  
18    private XmlUtil() {
19      // disable construction
20    }
21  
22    /**
23     * Create a {@link Source} based on the provided {@code url}.
24     * <p>
25     * The caller of this method must ensure that the stream associated with this
26     * source is closed.
27     *
28     * @param url
29     *          the URL to use for the source
30     * @return a new source
31     * @throws IOException
32     *           if an error occurred while creating the underlying stream
33     */
34    @SuppressWarnings("resource") // user of source is expected to close
35    @NonNull
36    public static StreamSource getStreamSource(@NonNull URL url) throws IOException {
37      return new StreamSource(url.openStream(), url.toString());
38    }
39  }
