6   package gov.nist.secauto.metaschema.core.model.constraint.impl;
7   
8   import gov.nist.secauto.metaschema.core.datatype.markup.MarkupMultiline;
9   import gov.nist.secauto.metaschema.core.metapath.MetapathExpression;
10  import gov.nist.secauto.metaschema.core.model.constraint.IKeyField;
11  import gov.nist.secauto.metaschema.core.model.constraint.ISource;
12  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
13  
14  import java.util.regex.Pattern;
15  
16  import edu.umd.cs.findbugs.annotations.NonNull;
17  import edu.umd.cs.findbugs.annotations.Nullable;
18  import nl.talsmasoftware.lazy4j.Lazy;
19  import nl.talsmasoftware.lazy4j.LazyEvaluationException;
20  
21  public class DefaultKeyField implements IKeyField {
22  
23    @Nullable
24    private final Pattern pattern;
25    @NonNull
26    private final Lazy<MetapathExpression> target;
27    @Nullable
28    private final MarkupMultiline remarks;
29  
30    /**
31     * Construct a new key field based on the provided target. An optional pattern
32     * can be used to extract a portion of the resulting key value.
33     *
34     * @param target
35     *          a Metapath expression identifying the target of the key field
36     * @param pattern
37     *          an optional used to extract a portion of the resulting key value
38     * @param remarks
39     *          optional remarks describing the intent of the constraint
40     * @param source
41     *          the descriptor for the resource containing the constraint
42     */
43    public DefaultKeyField(
44        @NonNull String target,
45        @Nullable Pattern pattern,
46        @Nullable MarkupMultiline remarks,
47        @NonNull ISource source) {
48      this.pattern = pattern;
49      this.target = ObjectUtils.notNull(Lazy.lazy(() -> MetapathExpression.compile(target, source.getStaticContext())));
50      this.remarks = remarks;
51    }
52  
53    @Override
54    public Pattern getPattern() {
55      return pattern;
56    }
57  
58    @Override
59    public String getTarget() {
60      return getTargetMetapath().getPath();
61    }
62  
63    @Override
64    public MetapathExpression getTargetMetapath() {
65      try {
66        return ObjectUtils.notNull(target.get());
67      } catch (LazyEvaluationException ex) {
68        Throwable cause = ex.getCause();
69        cause.addSuppressed(ex);
70        if (cause instanceof RuntimeException) {
71          throw (RuntimeException) cause;
72        }
73        throw ex;
74      }
75    }
76  
77    @Override
78    public MarkupMultiline getRemarks() {
79      return remarks;
80    }
81  }
