6   package gov.nist.secauto.metaschema.core.model.constraint;
7   
8   import gov.nist.secauto.metaschema.core.model.IAssemblyDefinition;
9   import gov.nist.secauto.metaschema.core.model.constraint.impl.AbstractDefinitionTargetedConstraints;
10  
11  import edu.umd.cs.findbugs.annotations.NonNull;
12  
13  /**
14   * A set of constraints targeting a {@link IAssemblyDefinition} based on a
15   * target Metapath expression.
16   *
17   * @see #getTargetExpression()
18   */
19  public class AssemblyTargetedConstraints
20      extends AbstractDefinitionTargetedConstraints<IAssemblyDefinition, IModelConstrained>
21      implements IFeatureModelConstrained {
22  
23    /**
24     * Construct a new set of targeted constraints.
25     *
26     * @param target
27     *          the Metapath expression that can be used to find matching targets
28     * @param constraints
29     *          the constraints to apply to matching targets
30     */
31    public AssemblyTargetedConstraints(
32        @NonNull String target,
33        @NonNull IModelConstrained constraints) {
34      super(target, constraints);
35    }
36  
37    @Override
38    public void target(@NonNull IAssemblyDefinition definition) {
39      applyTo(definition);
40    }
41  
42    @Override
43    protected void applyTo(@NonNull IAssemblyDefinition definition) {
44      super.applyTo(definition);
45      getIndexConstraints().forEach(definition::addConstraint);
46      getUniqueConstraints().forEach(definition::addConstraint);
47      getHasCardinalityConstraints().forEach(definition::addConstraint);
48    }
49  }
