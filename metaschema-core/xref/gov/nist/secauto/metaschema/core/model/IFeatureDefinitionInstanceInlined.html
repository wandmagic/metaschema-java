6   package gov.nist.secauto.metaschema.core.model;
7   
8   import gov.nist.secauto.metaschema.core.datatype.markup.MarkupLine;
9   import gov.nist.secauto.metaschema.core.util.ObjectUtils;
10  
11  import java.util.Locale;
12  
13  import javax.xml.namespace.QName;
14  
15  import edu.umd.cs.findbugs.annotations.NonNull;
16  import edu.umd.cs.findbugs.annotations.Nullable;
17  
18  /**
19   * A trait indicating that the implementation is a localized definition that is
20   * declared in-line as an instance.
21   *
22   * @param <DEFINITION>
23   *          the associated definition Java type
24   * @param <INSTANCE>
25   *          the associated instance Java type
26   */
27  public interface IFeatureDefinitionInstanceInlined<
28      DEFINITION extends IDefinition,
29      INSTANCE extends INamedInstance>
30      extends IDefinition, INamedInstance {
31    @Override
32    default boolean isInline() {
33      // has to be inline
34      return true;
35    }
36  
37    @Override
38    default QName getDefinitionQName() {
39      return getReferencedDefinitionQName();
40    }
41  
42    @Override
43    default DEFINITION getDefinition() {
44      return ObjectUtils.asType(this);
45    }
46  
47    @Override
48    @NonNull
49    default INSTANCE getInlineInstance() {
50      return ObjectUtils.asType(this);
51    }
52  
53    @Override
54    default String getEffectiveFormalName() {
55      return getFormalName();
56    }
57  
58    @Override
59    default MarkupLine getEffectiveDescription() {
60      return getDescription();
61    }
62  
63    @Override
64    default String getEffectiveName() {
65      // don't use use-name
66      return getName();
67    }
68  
69    @Override
70    default Integer getEffectiveIndex() {
71      return getIndex();
72    }
73  
74    @Override
75    @Nullable
76    default Object getEffectiveDefaultValue() {
77      // This is an inline instance that is both a definition and an instance. Don't
78      // delegate to the definition, since this would be redundant.
79      return getDefaultValue();
80    }
81  
82    /**
83     * Generates a "coordinate" string for the provided inline definition instance.
84     *
85     * A coordinate consists of the element's:
86     * <ul>
87     * <li>containing Metaschema module's short name</li>
88     * <li>model type</li>
89     * <li>definition name</li>
90     * <li>hash code</li>
91     * </ul>
92     *
93     * @return the coordinate
94     */
95    @SuppressWarnings("null")
96    @Override
97    default String toCoordinates() {
98      IModule module = getContainingModule();
99      return String.format("%s-inline-definition:%s:%s/%s@%d",
100         getModelType().toString().toLowerCase(Locale.ROOT),
101         module.getShortName(),
102         getContainingDefinition().getName(),
103         getName(),
104         hashCode());
105   }
106 }
