6   package gov.nist.secauto.metaschema.core.model.constraint;
7   
8   import gov.nist.secauto.metaschema.core.model.IFieldDefinition;
9   import gov.nist.secauto.metaschema.core.model.constraint.impl.AbstractDefinitionTargetedConstraints;
10  
11  import edu.umd.cs.findbugs.annotations.NonNull;
12  
13  /**
14   * A set of constraints targeting a {@link IFieldDefinition} based on a target
15   * Metapath expression.
16   *
17   * @see #getTargetExpression()
18   */
19  public class FieldTargetedConstraints
20      extends AbstractDefinitionTargetedConstraints<IFieldDefinition, IValueConstrained> {
21  
22    /**
23     * Construct a new set of targeted constraints.
24     *
25     * @param target
26     *          the Metapath expression that can be used to find matching targets
27     * @param constraints
28     *          the constraints to apply to matching targets
29     */
30    public FieldTargetedConstraints(
31        @NonNull String target,
32        @NonNull IValueConstrained constraints) {
33      super(target, constraints);
34    }
35  
36    @Override
37    public void target(@NonNull IFieldDefinition definition) {
38      applyTo(definition);
39    }
40  }
