6   package gov.nist.secauto.metaschema.core.model.constraint.impl;
7   
8   import gov.nist.secauto.metaschema.core.model.IAssemblyDefinition;
9   import gov.nist.secauto.metaschema.core.model.IDefinition;
10  import gov.nist.secauto.metaschema.core.model.IFieldDefinition;
11  import gov.nist.secauto.metaschema.core.model.IFlagDefinition;
12  import gov.nist.secauto.metaschema.core.model.constraint.AbstractTargetedConstraints;
13  import gov.nist.secauto.metaschema.core.model.constraint.IValueConstrained;
14  
15  import java.util.Locale;
16  
17  import edu.umd.cs.findbugs.annotations.NonNull;
18  
19  /**
20   * Provides an base implementation for a set of constraints that target a given
21   * definition using a target Metapath expression.
22   *
23   * @param <T>
24   *          the Java type of the definition target
25   *
26   * @param <S>
27   *          the Java type of the constraint container
28   */
29  public abstract class AbstractDefinitionTargetedConstraints<
30      T extends IDefinition,
31      S extends IValueConstrained>
32      extends AbstractTargetedConstraints<S> {
33  
34    /**
35     * Construct a new set of targeted constraints.
36     *
37     * @param target
38     *          the Metapath expression that can be used to find matching targets
39     * @param constraints
40     *          the constraints to apply to matching targets
41     */
42    protected AbstractDefinitionTargetedConstraints(
43        @NonNull String target,
44        @NonNull S constraints) {
45      super(target, constraints);
46    }
47  
48    /**
49     * Apply the constraints to the provided {@code definition}.
50     * <p>
51     * This will be called when a definition is found that matches the target
52     * expression.
53     *
54     * @param definition
55     *          the definition to apply the constraints to.
56     */
57    protected void applyTo(@NonNull T definition) {
58      getAllowedValuesConstraints().forEach(definition::addConstraint);
59      getMatchesConstraints().forEach(definition::addConstraint);
60      getIndexHasKeyConstraints().forEach(definition::addConstraint);
61      getExpectConstraints().forEach(definition::addConstraint);
62    }
63  
64    @Override
65    public void target(@NonNull IFlagDefinition definition) {
66      throw new IllegalStateException(
67          String.format("The targeted definition '%s' from metaschema '%s' is not a %s definition.",
68              definition.getEffectiveName(),
69              definition.getContainingModule().getQName().toString(),
70              definition.getModelType().name().toLowerCase(Locale.ROOT)));
71    }
72  
73    @Override
74    public void target(@NonNull IFieldDefinition definition) {
75      throw new IllegalStateException(
76          String.format("The targeted definition '%s' from metaschema '%s' is not a %s definition.",
77              definition.getEffectiveName(),
78              definition.getContainingModule().getQName().toString(),
79              definition.getModelType().name().toLowerCase(Locale.ROOT)));
80    }
81  
82    @Override
83    public void target(@NonNull IAssemblyDefinition definition) {
84      throw new IllegalStateException(
85          String.format("The targeted definition '%s' from metaschema '%s' is not a %s definition.",
86              definition.getEffectiveName(),
87              definition.getContainingModule().getQName().toString(),
88              definition.getModelType().name().toLowerCase(Locale.ROOT)));
89    }
90  }
