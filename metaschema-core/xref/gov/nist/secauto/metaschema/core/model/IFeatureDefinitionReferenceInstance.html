6   package gov.nist.secauto.metaschema.core.model;
7   
8   import gov.nist.secauto.metaschema.core.datatype.markup.MarkupLine;
9   
10  import edu.umd.cs.findbugs.annotations.NonNull;
11  import edu.umd.cs.findbugs.annotations.Nullable;
12  
13  public interface IFeatureDefinitionReferenceInstance<
14      DEFINITION extends IDefinition,
15      INSTANCE extends INamedInstance>
16      extends INamedInstance {
17  
18    @Override
19    DEFINITION getDefinition();
20  
21    /**
22     * Get the instance this definition is combined with.
23     *
24     * @return the instance or {@code null} if the definition is not inline
25     */
26    @Nullable
27    default INSTANCE getInlineInstance() {
28      return null;
29    }
30  
31    @Override
32    default String getEffectiveFormalName() {
33      String result = getFormalName();
34      return result == null ? getDefinition().getEffectiveFormalName() : result;
35    }
36  
37    @Override
38    default MarkupLine getEffectiveDescription() {
39      MarkupLine result = getDescription();
40      return result == null ? getDefinition().getEffectiveDescription() : result;
41    }
42  
43    @Override
44    @NonNull
45    default String getEffectiveName() {
46      String result = getUseName();
47      if (result == null) {
48        // fall back to the definition
49        IDefinition def = getDefinition();
50        result = def.getEffectiveName();
51      }
52      return result;
53    }
54  
55    @Override
56    @Nullable
57    default Integer getEffectiveIndex() {
58      Integer result = getUseIndex();
59      if (result == null) {
60        // fall back to the definition
61        IDefinition def = getDefinition();
62        result = def.getEffectiveIndex();
63      }
64      return result;
65    }
66  
67    /**
68     * The resolved default value, which allows an instance to override a
69     * definition's default value.
70     *
71     * @return the default value or {@code null} if not defined on either the
72     *         instance or definition
73     */
74    @Override
75    @Nullable
76    default Object getEffectiveDefaultValue() {
77      Object retval = getDefaultValue();
78      if (retval == null) {
79        retval = getDefinition().getDefaultValue();
80      }
81      return retval;
82    }
83  }
