79    private enum Duplicates {
80      REJECT("reject", (key, v1, v2) -> {
81        throw new JsonFunctionException(
82            JsonFunctionException.DUPLICATE_KEYS,
83            String.format("Duplicate key '%s' not allowed.", key.getKey().asString()));
84      }),
85  
86      USE_FIRST("use-first", (key, v1, v2) -> v1),
87      USE_LAST("use-last", (key, v1, v2) -> v2),
88      USE_ANY("use-any", (key, v1, v2) -> RANDOM.nextBoolean() ? v1 : v2),
89      @SuppressWarnings("null")
90      COMBINE("combine", (key, v1, v2) -> {
91        return Stream.concat(
92            v1.asSequence().stream(),
93            v2.asSequence().stream()).collect(ISequence.toSequence());
94      });
95  
96      private static final Map<String, Duplicates> BY_NAME;
97  
98      @NonNull
99      private final String name;
100     @NonNull
101     private final CustomCollectors.DuplicateHandler<IMapKey, ICollectionValue> duplicateHander;
102 
103     static {
104       @SuppressWarnings("PMD.UseConcurrentHashMap") Map<String, Duplicates> map = new HashMap<>();
105       for (Duplicates value : values()) {
106         map.put(value.getName(), value);
107       }
108       BY_NAME = Collections.unmodifiableMap(map);
109     }
110 
111     @Nullable
112     public static Duplicates lookup(@NonNull String name) {
113       return BY_NAME.get(name);
114     }
115 
116     Duplicates(@NonNull String name,
117         @NonNull CustomCollectors.DuplicateHandler<IMapKey, ICollectionValue> duplicateHander) {
118       this.name = name;
119       this.duplicateHander = duplicateHander;
120     }
121 
122     public String getName() {
123       return name;
124     }
125 
126     public CustomCollectors.DuplicateHandler<IMapKey, ICollectionValue> getDuplicateHander() {
127       return duplicateHander;
128     }
129   }

131   private MapMerge() {
132     // disable construction
133   }

135   @SuppressWarnings("unused")
136   @NonNull
137   private static ISequence<?> executeOneArg(@NonNull IFunction function,
138       @NonNull List<ISequence<?>> arguments,
139       @NonNull DynamicContext dynamicContext,
140       IItem focus) {
141     ISequence<IMapItem<?>> maps = FunctionUtils.asType(ObjectUtils.requireNonNull(arguments.get(0)));
142 
143     return ISequence.of(merge(maps, CollectionUtil.emptyMap()));
144   }
145 
146   @SuppressWarnings("unused")
147   @NonNull
148   private static ISequence<?> executeTwoArg(@NonNull IFunction function,
149       @NonNull List<ISequence<?>> arguments,
150       @NonNull DynamicContext dynamicContext,
151       IItem focus) {
152     ISequence<IMapItem<?>> maps = FunctionUtils.asType(ObjectUtils.requireNonNull(arguments.get(0)));
153     IMapItem<?> options = FunctionUtils.asType(ObjectUtils.requireNonNull(arguments.get(1).getFirstItem(true)));
154 
155     return ISequence.of(merge(maps, options));
156   }

158   /**
159    * An implementation of XPath 3.1 <a href=
160    * "https://www.w3.org/TR/xpath-functions-31/#func-map-merge">map:merge</a>.
161    *
162    * @param maps
163    *          a collection of maps to merge
164    * @param options
165    *          settings that affect the merge behavior
166    * @return a map containing the merged entries
167    */
168   @SuppressWarnings({ "null", "PMD.OnlyOneReturn" })
169   @NonNull
170   public static IMapItem<?> merge(@NonNull Collection<? extends Map<IMapKey, ? extends ICollectionValue>> maps,
171       @NonNull Map<IMapKey, ? extends ICollectionValue> options) {
172     if (maps.isEmpty()) {
173       return IMapItem.empty();
174     }
175 
176     // handle the "duplicates" option
177     ICollectionValue duplicatesOption = options.get(DUPLICATES_OPTION);
178 
179     Duplicates duplicates;
180     if (duplicatesOption == null) {
181       // default option
182       duplicates = Duplicates.USE_FIRST;
183     } else {
184       // resolve the provided option
185       IAnyAtomicItem atomicValue = FnData.fnData(duplicatesOption.asSequence()).getFirstItem(true);
186       if (atomicValue == null) {
187         throw new JsonFunctionException(
188             JsonFunctionException.INVALID_OPTION,
189             String.format("Missing '%s' option value.", DUPLICATES_OPTION.getKey().asString()));
190       }
191       String duplicatesValue = IStringItem.cast(atomicValue).asString();
192       duplicates = Duplicates.lookup(duplicatesValue);
193       if (duplicates == null) {
194         throw new JsonFunctionException(
195             JsonFunctionException.INVALID_OPTION,
196             String.format("Invalid '%s' option value '%s'.", DUPLICATES_OPTION.getKey().asString(), duplicatesValue));
197       }
198     }
199 
200     // merge the maps
201     return IMapItem.ofCollection(maps.stream()
202         .flatMap(map -> map.entrySet().stream())
203         // collect the entries into a new map
204         .collect(CustomCollectors.toMap(Map.Entry::getKey, Map.Entry::getValue, duplicates.getDuplicateHander(),
205             HashMap::new)));
206   }
