6   package gov.nist.secauto.metaschema.core.metapath.item.atomic;
7   
8   import gov.nist.secauto.metaschema.core.datatype.adapter.MetaschemaDataTypeProvider;
9   import gov.nist.secauto.metaschema.core.metapath.function.InvalidValueForCastFunctionException;
10  
11  import edu.umd.cs.findbugs.annotations.NonNull;
12  import inet.ipaddr.ipv4.IPv4Address;
13  
14  public interface IIPv4AddressItem extends IIPAddressItem {
15  
16    /**
17     * Construct a new IPv4 item using the provided {@code value}.
18     *
19     * @param value
20     *          an IPv4 value
21     * @return the new item
22     */
23    @NonNull
24    static IIPv4AddressItem valueOf(@NonNull IPv4Address value) {
25      return new IPv4AddressItemImpl(value);
26    }
27  
28    /**
29     * Cast the provided type to this item type.
30     *
31     * @param item
32     *          the item to cast
33     * @return the original item if it is already this type, otherwise a new item
34     *         cast to this type
35     * @throws InvalidValueForCastFunctionException
36     *           if the provided {@code item} cannot be cast to this type
37     */
38    @NonNull
39    static IIPv4AddressItem cast(@NonNull IAnyAtomicItem item) {
40      return MetaschemaDataTypeProvider.IP_V4_ADDRESS.cast(item);
41    }
42  
43    @Override
44    IPv4Address asIpAddress();
45  
46    @Override
47    default IIPv4AddressItem castAsType(IAnyAtomicItem item) {
48      return cast(item);
49    }
50  
51    @Override
52    default int compareTo(IAnyAtomicItem item) {
53      return compareTo(cast(item));
54    }
55  }
