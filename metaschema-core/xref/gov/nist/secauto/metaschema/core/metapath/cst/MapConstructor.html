6   package gov.nist.secauto.metaschema.core.metapath.cst;
7   
8   import gov.nist.secauto.metaschema.core.metapath.DynamicContext;
9   import gov.nist.secauto.metaschema.core.metapath.ICollectionValue;
10  import gov.nist.secauto.metaschema.core.metapath.ISequence;
11  import gov.nist.secauto.metaschema.core.metapath.InvalidTypeMetapathException;
12  import gov.nist.secauto.metaschema.core.metapath.function.library.FnData;
13  import gov.nist.secauto.metaschema.core.metapath.item.IItem;
14  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IAnyAtomicItem;
15  import gov.nist.secauto.metaschema.core.metapath.item.function.IMapItem;
16  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
17  
18  import java.util.List;
19  import java.util.Map;
20  import java.util.stream.Collectors;
21  
22  import edu.umd.cs.findbugs.annotations.NonNull;
23  
24  /**
25   * An implementation of the
26   * <a href="https://www.w3.org/TR/xpath-31/#id-map-constructors">Map
27   * Constructor</a> supporting the creation of a Metapath {@link IMapItem}.
28   */
29  public class MapConstructor implements IExpression {
30    @NonNull
31    private final List<MapConstructor.Entry> entries;
32  
33    /**
34     * Construct a new map constructor expression that uses the provided entry
35     * expressions to initialize the map entries.
36     *
37     * @param entries
38     *          the expressions used to produce the map entries
39     */
40    public MapConstructor(@NonNull List<MapConstructor.Entry> entries) {
41      this.entries = entries;
42    }
43  
44    @Override
45    public List<MapConstructor.Entry> getChildren() {
46      return entries;
47    }
48  
49    @Override
50    public ISequence<? extends IItem> accept(DynamicContext dynamicContext, ISequence<?> focus) {
51      return IMapItem.ofCollection(
52          ObjectUtils.notNull(getChildren().stream()
53              .map(item -> {
54                IExpression keyExpression = item.getKeyExpression();
55                IAnyAtomicItem key = FnData.fnData(keyExpression.accept(dynamicContext, focus))
56                    .getFirstItem(true);
57                if (key == null) {
58                  throw new InvalidTypeMetapathException(null, String.format(
59                      "The expression '%s' did not result in a single key atomic value.", keyExpression.toASTString()));
60                }
61                ICollectionValue value = item.getValueExpression().accept(dynamicContext, focus).toCollectionValue();
62  
63                return IMapItem.entry(key, value);
64              }).collect(Collectors.toMap(Map.Entry::getKey, Map.Entry::getValue))))
65          .asSequence();
66    }
67  
68    @Override
69    public <RESULT, CONTEXT> RESULT accept(IExpressionVisitor<RESULT, CONTEXT> visitor, CONTEXT context) {
70      return visitor.visitMapConstructor(this, context);
71    }
72  
73    /**
74     * A map entry expression used to produce an entry in a {@link IMapItem}.
75     */
76    public static class Entry implements IExpression {
77      @NonNull
78      private final IExpression keyExpression;
79      @NonNull
80      private final IExpression valueExpression;
81  
82      /**
83       * Construct a new map entry expression using the provided ket and value
84       * expressions.
85       *
86       * @param keyExpression
87       *          the expression used to get the map entry key
88       * @param valueExpression
89       *          the expression used to get the map entry value
90       */
91      public Entry(@NonNull IExpression keyExpression, @NonNull IExpression valueExpression) {
92        this.keyExpression = keyExpression;
93        this.valueExpression = valueExpression;
94      }
95  
96      /**
97       * Get the map entry key expression.
98       *
99       * @return the key expression
100      */
101     @NonNull
102     public IExpression getKeyExpression() {
103       return keyExpression;
104     }
105 
106     /**
107      * Get the map entry value expression.
108      *
109      * @return the value expression
110      */
111     @NonNull
112     public IExpression getValueExpression() {
113       return valueExpression;
114     }
115 
116     @SuppressWarnings("null")
117     @Override
118     public List<? extends IExpression> getChildren() {
119       return List.of(keyExpression, valueExpression);
120     }
121 
122     @Override
123     public ISequence<? extends IItem> accept(DynamicContext dynamicContext, ISequence<?> focus) {
124       throw new UnsupportedOperationException("handled by the map constructor");
125     }
126 
127     @Override
128     public <RESULT, CONTEXT> RESULT accept(IExpressionVisitor<RESULT, CONTEXT> visitor, CONTEXT context) {
129       return visitor.visitMapConstructorEntry(this, context);
130     }
131 
132   }
