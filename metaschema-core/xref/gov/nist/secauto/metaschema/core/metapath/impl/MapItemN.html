6   package gov.nist.secauto.metaschema.core.metapath.impl;
7   
8   import gov.nist.secauto.metaschema.core.metapath.ICollectionValue;
9   import gov.nist.secauto.metaschema.core.metapath.item.function.IMapKey;
10  import gov.nist.secauto.metaschema.core.util.CollectionUtil;
11  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
12  
13  import java.util.Map;
14  
15  import edu.umd.cs.findbugs.annotations.NonNull;
16  
17  /**
18   * An map item that supports an unbounded number of entries.
19   *
20   * @param <VALUE>
21   *          the Java type of the entry values
22   */
23  public class MapItemN<VALUE extends ICollectionValue>
24      extends AbstractMapItem<VALUE> {
25    @NonNull
26    private final Map<IMapKey, VALUE> entries;
27  
28    /**
29     * Construct a new map item with the provided entries.
30     *
31     * @param entries
32     *          the entries to add to the map
33     */
34    @SafeVarargs
35    public MapItemN(@NonNull Map.Entry<IMapKey, ? extends VALUE>... entries) {
36      this(ObjectUtils.notNull(Map.ofEntries(entries)));
37    }
38  
39    /**
40     * Construct a new map item using the entries from the provided map.
41     *
42     * @param entries
43     *          a map containing the entries to add to the map
44     */
45    public MapItemN(@NonNull Map<IMapKey, VALUE> entries) {
46      this.entries = CollectionUtil.unmodifiableMap(entries);
47    }
48  
49    @Override
50    public Map<IMapKey, VALUE> getValue() {
51      return entries;
52    }
53  }
