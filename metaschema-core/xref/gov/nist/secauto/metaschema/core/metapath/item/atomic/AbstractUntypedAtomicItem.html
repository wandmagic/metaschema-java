6   package gov.nist.secauto.metaschema.core.metapath.item.atomic;
7   
8   import gov.nist.secauto.metaschema.core.metapath.impl.AbstractStringMapKey;
9   import gov.nist.secauto.metaschema.core.metapath.item.function.IMapKey;
10  
11  import edu.umd.cs.findbugs.annotations.NonNull;
12  
13  /**
14   * A base class for untyped atomic items.
15   *
16   * @param <TYPE>
17   *          the Java type of the wrapped value
18   */
19  public abstract class AbstractUntypedAtomicItem<TYPE>
20      extends AbstractAnyAtomicItem<TYPE>
21      implements IUntypedAtomicItem {
22  
23    /**
24     * Construct a new untyped atomic valued item.
25     *
26     * @param value
27     *          the value
28     */
29    protected AbstractUntypedAtomicItem(@NonNull TYPE value) {
30      super(value);
31    }
32  
33    @Override
34    public IMapKey asMapKey() {
35      return new MapKey();
36    }
37  
38    private final class MapKey
39        extends AbstractStringMapKey {
40  
41      @Override
42      public IUntypedAtomicItem getKey() {
43        return AbstractUntypedAtomicItem.this;
44      }
45    }
46  }
