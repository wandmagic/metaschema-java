6   package gov.nist.secauto.metaschema.core.metapath.cst;
7   
8   import gov.nist.secauto.metaschema.core.metapath.ISequence;
9   import gov.nist.secauto.metaschema.core.metapath.item.IItem;
10  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
11  
12  import java.util.List;
13  
14  import edu.umd.cs.findbugs.annotations.NonNull;
15  
16  /**
17   * The CST node for a Metapath
18   * <a href="https://www.w3.org/TR/xpath-31/#combining_seq">intersect
19   * expression</a>.
20   */
21  public class Intersect
22      extends AbstractFilterExpression {
23  
24    /**
25     * Construct a new Metapath except expression CST node.
26     *
27     * @param left
28     *          an expression indicating the items to filter
29     * @param right
30     *          an expression indicating the items to keep
31     */
32  
33    public Intersect(@NonNull IExpression left, @NonNull IExpression right) {
34      super(left, right);
35    }
36  
37    @Override
38    protected ISequence<?> applyFilterTo(@NonNull ISequence<?> result, @NonNull List<? extends IItem> items) {
39      return ISequence.of(ObjectUtils.notNull(result.stream()
40          .distinct()
41          .filter(items::contains)));
42    }
43  
44    @Override
45    public <RESULT, CONTEXT> RESULT accept(@NonNull IExpressionVisitor<RESULT, CONTEXT> visitor, CONTEXT context) {
46      return visitor.visitIntersect(this, context);
47    }
48  }
