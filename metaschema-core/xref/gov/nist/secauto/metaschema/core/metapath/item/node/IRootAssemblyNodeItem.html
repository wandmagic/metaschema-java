2   package gov.nist.secauto.metaschema.core.metapath.item.node;
3   
4   import gov.nist.secauto.metaschema.core.metapath.format.IPathFormatter;
5   import gov.nist.secauto.metaschema.core.model.IAssemblyInstance;
6   import gov.nist.secauto.metaschema.core.util.ObjectUtils;
7   
8   import javax.xml.namespace.QName;
9   
10  import edu.umd.cs.findbugs.annotations.NonNull;
11  
12  /**
13   * A marker interface used to expose root node functionality for an assembly
14   * node that has root information.
15   */
16  public interface IRootAssemblyNodeItem extends IAssemblyNodeItem, IFeatureChildNodeItem {
17  
18    /**
19     * Get the name of this node.
20     * <p>
21     * This overrides the default behavior using the root name for the assembly.
22     */
23    @Override
24    default QName getQName() {
25      return ObjectUtils.requireNonNull(getDefinition().getRootXmlQName());
26    }
27  
28    /**
29     * Get the parent document node item for this root.
30     *
31     * @return the parent document item
32     */
33    @NonNull
34    IDocumentNodeItem getDocumentNodeItem();
35  
36    @Override
37    @NonNull
38    default IDocumentNodeItem getParentNodeItem() {
39      return getDocumentNodeItem();
40    }
41  
42    @Override
43    default IAssemblyNodeItem getParentContentNodeItem() {
44      // there is no assembly parent
45      return null;
46    }
47  
48    @Override
49    default IAssemblyInstance getInstance() {
50      // there is no instance
51      return null;
52    }
53  
54    @Override
55    default IRootAssemblyNodeItem getNodeItem() {
56      return this;
57    }
58  
59    @Override
60    default int getPosition() {
61      // a root is always in the first position
62      return 1;
63    }
64  
65    @Override
66    default String format(@NonNull IPathFormatter formatter) {
67      return formatter.formatRootAssembly(this);
68    }
69  }
