8   import gov.nist.secauto.metaschema.core.metapath.DynamicContext;
9   import gov.nist.secauto.metaschema.core.metapath.ISequence;
10  import gov.nist.secauto.metaschema.core.metapath.function.FunctionUtils;
11  import gov.nist.secauto.metaschema.core.metapath.function.IArgument;
12  import gov.nist.secauto.metaschema.core.metapath.function.IFunction;
13  import gov.nist.secauto.metaschema.core.metapath.function.IFunctionExecutor;
14  import gov.nist.secauto.metaschema.core.metapath.item.IItem;
15  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IAnyAtomicItem;
16  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
17  
18  import java.net.URI;
19  import java.util.List;
20  
21  import edu.umd.cs.findbugs.annotations.NonNull;
22  
23  /**
24   * Implements the XPath 3.1
25   * <a href= "https://www.w3.org/TR/xpath-functions-31/#casting">casting
26   * functions</a>.
27   *
28   * @param <ITEM>
29   *          the Metapath atomic item's Java type
30   */
31  public final class CastFunction<ITEM extends IAnyAtomicItem> implements IFunctionExecutor {
32    @NonNull
33    private final ICastExecutor<ITEM> castExecutor;
34  
35    @NonNull
36    static <ITEM extends IAnyAtomicItem> IFunction signature(
37        @NonNull URI namespace,
38        @NonNull String name,
39        @NonNull Class<ITEM> resulingAtomicType,
40        @NonNull ICastExecutor<ITEM> executor) {
41      return signature(
42          ObjectUtils.notNull(namespace.toASCIIString()),
43          name,
44          resulingAtomicType,
45          executor);
46    }
47  
48    @NonNull
49    static <ITEM extends IAnyAtomicItem> IFunction signature(
50        @NonNull String namespace,
51        @NonNull String name,
52        @NonNull Class<ITEM> resulingAtomicType,
53        @NonNull ICastExecutor<ITEM> executor) {
54      return IFunction.builder()
55          .name(name)
56          .namespace(namespace)
57          .deterministic()
58          .contextIndependent()
59          .focusIndependent()
60          .argument(IArgument.builder()
61              .name("arg1")
62              .type(IAnyAtomicItem.class)
63              .zeroOrOne()
64              .build())
65          .returnType(resulingAtomicType)
66          .returnZeroOrOne()
67          .functionHandler(newCastExecutor(executor))
68          .build();
69    }
70  
71    @NonNull
72    private static <ITEM extends IAnyAtomicItem> CastFunction<ITEM>
73        newCastExecutor(@NonNull ICastExecutor<ITEM> executor) {
74      return new CastFunction<>(executor);
75    }
76  
77    private CastFunction(@NonNull ICastExecutor<ITEM> castExecutor) {
78      this.castExecutor = castExecutor;
79    }
80  
81    @Override
82    public ISequence<ITEM> execute(@NonNull IFunction function,
83        @NonNull List<ISequence<?>> arguments,
84        @NonNull DynamicContext dynamicContext,
85        IItem focus) {
86  
87      ISequence<? extends IAnyAtomicItem> arg = FunctionUtils.asType(
88          ObjectUtils.notNull(arguments.get(0)));
89  
90      IAnyAtomicItem item = arg.getFirstItem(true);
91      if (item == null) {
92        return ISequence.empty(); // NOPMD - readability
93      }
94  
95      ITEM castItem = castExecutor.cast(item);
96      return ISequence.of(castItem);
97    }
98  
99    @FunctionalInterface
100   public interface ICastExecutor<ITEM extends IAnyAtomicItem> {
101     /**
102      * Cast the provided {@code item}.
103      *
104      * @param item
105      *          the item to cast
106      * @return the item cast to the appropriate type
107      */
108     @NonNull
109     ITEM cast(@NonNull IAnyAtomicItem item);
110   }
111 }
