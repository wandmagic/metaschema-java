80    /**
81     * Given the provided parent node item, generate a mapping of flag name to flag
82     * node item for each flag on the parent assembly.
83     *
84     * @param parent
85     *          the parent assembly containing flags
86     * @return a mapping of flag name to flag item
87     */
88    @SuppressWarnings("PMD.UseConcurrentHashMap") // need an ordered Map
89    @NonNull
90    protected Map<QName, IFlagNodeItem> generateFlags(@NonNull IModelNodeItem<?, ?> parent) {
91      Map<QName, IFlagNodeItem> retval = new LinkedHashMap<>();
92  
93      Object parentValue = parent.getValue();
94      assert parentValue != null;
95      for (IFlagInstance instance : parent.getDefinition().getFlagInstances()) {
96        Object flagValue = instance.getValue(parentValue);
97        if (flagValue != null) {
98          IFlagNodeItem item = newFlagNodeItem(instance, parent, flagValue);
99          retval.put(instance.getXmlQName(), item);
100       }
101     }
102     return retval.isEmpty() ? CollectionUtil.emptyMap() : CollectionUtil.unmodifiableMap(retval);
103   }

105   /**
106    * Given the provided parent node item, generate a mapping of model instance
107    * name to model node item(s) for each model instance on the parent assembly.
108    *
109    * @param parent
110    *          the parent assembly containing model instances
111    * @return a mapping of model instance name to model node item(s)
112    */
113   @SuppressWarnings({ "PMD.UseConcurrentHashMap", "PMD.CognitiveComplexity" }) // need an ordered map
114   @NonNull
115   protected Map<QName, List<? extends IModelNodeItem<?, ?>>> generateModelItems(
116       @NonNull IAssemblyNodeItem parent) {
117     Map<QName, List<? extends IModelNodeItem<?, ?>>> retval = new LinkedHashMap<>();
118 
119     Object parentValue = parent.getValue();
120     assert parentValue != null;
121     for (IModelInstance instance : CollectionUtil.toIterable(getValuedModelInstances(parent.getDefinition()))) {
122       if (instance instanceof INamedModelInstanceAbsolute) {
123         INamedModelInstanceAbsolute namedInstance = (INamedModelInstanceAbsolute) instance;
124 
125         Object instanceValue = namedInstance.getValue(parentValue);
126         if (instanceValue != null) {
127           List<IModelNodeItem<?, ?>> items = generateModelInstanceItems(
128               parent,
129               namedInstance,
130               ObjectUtils.notNull(namedInstance.getItemValues(instanceValue).stream()));
131           retval.put(namedInstance.getXmlQName(), items);
132         }
133       } else if (instance instanceof IChoiceGroupInstance) {
134         IChoiceGroupInstance choiceInstance = (IChoiceGroupInstance) instance;
135 
136         Object instanceValue = choiceInstance.getValue(parentValue);
137         if (instanceValue != null) {
138           Map<INamedModelInstanceGrouped, List<Object>> instanceMap
139               = choiceInstance.getItemValues(instanceValue).stream()
140                   .map(item -> {
141                     assert item != null;
142                     INamedModelInstanceGrouped itemInstance = choiceInstance.getItemInstance(item);
143                     return Map.entry(itemInstance, item);
144                   })
145                   .collect(Collectors.groupingBy(
146                       entry -> entry.getKey(),
147                       LinkedHashMap::new,
148                       Collectors.mapping(entry -> entry.getValue(), Collectors.toUnmodifiableList())));
149 
150           for (Map.Entry<INamedModelInstanceGrouped, List<Object>> entry : instanceMap.entrySet()) {
151             INamedModelInstanceGrouped namedInstance = entry.getKey();
152             assert namedInstance != null;
153 
154             List<IModelNodeItem<?, ?>> items = generateModelInstanceItems(
155                 parent,
156                 namedInstance,
157                 ObjectUtils.notNull(entry.getValue().stream()));
158             retval.put(namedInstance.getXmlQName(), items);
159           }
160         }
161       }
162 
163     }
164     return retval.isEmpty() ? CollectionUtil.emptyMap() : CollectionUtil.unmodifiableMap(retval);
165   }

167   private List<IModelNodeItem<?, ?>> generateModelInstanceItems(
168       @NonNull IAssemblyNodeItem parent,
169       @NonNull INamedModelInstance namedInstance,
170       @NonNull Stream<?> itemValues) {
171     AtomicInteger index = new AtomicInteger(); // NOPMD - intentional
172 
173     // the item values will be all non-null items
174     return itemValues.map(itemValue -> {
175       assert itemValue != null;
176       return newModelItem(namedInstance, parent, index.incrementAndGet(), itemValue);
177     }).collect(Collectors.toUnmodifiableList());
178   }

180   @Override
181   public Supplier<ModelContainer> newMetaschemaModelSupplier(@NonNull IModuleNodeItem item) {
182     return () -> {
183       IModule module = item.getModule();
184 
185       // build flags from Metaschema definitions
186       Map<QName, IFlagNodeItem> flags = ObjectUtils.notNull(
187           Collections.unmodifiableMap(module.getExportedFlagDefinitions().stream()
188               .map(def -> newFlagNodeItem(ObjectUtils.notNull(def), item))
189               .collect(
190                   Collectors.toMap(
191                       IFlagNodeItem::getQName,
192                       Function.identity(),
193                       (v1, v2) -> v2,
194                       LinkedHashMap::new))));
195 
196       // build model items from Metaschema definitions
197       Stream<IFieldNodeItem> fieldStream = module.getExportedFieldDefinitions().stream()
198           .map(def -> newFieldNodeItem(ObjectUtils.notNull(def), item));
199       Stream<IAssemblyNodeItem> assemblyStream = module.getExportedAssemblyDefinitions().stream()
200           .map(def -> newAssemblyNodeItem(ObjectUtils.notNull(def), item));
201 
202       Map<QName, List<? extends IModelNodeItem<?, ?>>> modelItems
203           = ObjectUtils.notNull(Stream.concat(fieldStream, assemblyStream)
204               .collect(
205                   Collectors.collectingAndThen(
206                       Collectors.groupingBy(IModelNodeItem::getQName),
207                       Collections::unmodifiableMap)));
208       return new ModelContainer(flags, modelItems);
209     };
210   }

230   @NonNull
231   protected Map<QName, IFlagNodeItem> generateMetaschemaFlags(
232       @NonNull IModelNodeItem<?, ?> parent) {
233     Map<QName, IFlagNodeItem> retval = new LinkedHashMap<>(); // NOPMD - intentional
234 
235     for (IFlagInstance instance : parent.getDefinition().getFlagInstances()) {
236       assert instance != null;
237       IFlagNodeItem item = newFlagNodeItem(instance, parent);
238       retval.put(instance.getXmlQName(), item);
239     }
240     return retval.isEmpty() ? CollectionUtil.emptyMap() : CollectionUtil.unmodifiableMap(retval);
241   }
242 
243   @NonNull
244   protected Map<QName, List<? extends IModelNodeItem<?, ?>>> generateMetaschemaModelItems(
245       @NonNull IAssemblyNodeItem parent) {
246     Map<QName, List<? extends IModelNodeItem<?, ?>>> retval = new LinkedHashMap<>(); // NOPMD - intentional
247 
248     for (INamedModelInstance instance : CollectionUtil.toIterable(getNamedModelInstances(parent.getDefinition()))) {
249       assert instance != null;
250       IModelNodeItem<?, ?> item = newModelItem(instance, parent);
251       retval.put(instance.getXmlQName(), Collections.singletonList(item));
252     }
253     return retval.isEmpty() ? CollectionUtil.emptyMap() : CollectionUtil.unmodifiableMap(retval);
254   }
