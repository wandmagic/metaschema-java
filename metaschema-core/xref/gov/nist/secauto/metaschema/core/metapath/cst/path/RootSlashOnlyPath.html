6   package gov.nist.secauto.metaschema.core.metapath.cst.path;
7   
8   import gov.nist.secauto.metaschema.core.metapath.DynamicContext;
9   import gov.nist.secauto.metaschema.core.metapath.ISequence;
10  import gov.nist.secauto.metaschema.core.metapath.cst.IExpression;
11  import gov.nist.secauto.metaschema.core.metapath.cst.IExpressionVisitor;
12  import gov.nist.secauto.metaschema.core.metapath.item.ItemUtils;
13  import gov.nist.secauto.metaschema.core.metapath.item.node.INodeItem;
14  import gov.nist.secauto.metaschema.core.util.CollectionUtil;
15  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
16  
17  import java.util.List;
18  
19  public class RootSlashOnlyPath
20      extends AbstractPathExpression<INodeItem> {
21  
22    @Override
23    public List<? extends IExpression> getChildren() {
24      return CollectionUtil.emptyList();
25    }
26  
27    @Override
28    public Class<INodeItem> getBaseResultType() {
29      return INodeItem.class;
30    }
31  
32    @Override
33    public <RESULT, CONTEXT> RESULT accept(IExpressionVisitor<RESULT, CONTEXT> visitor, CONTEXT context) {
34      return visitor.visitRootSlashOnlyPath(this, context);
35    }
36  
37    @Override
38    public ISequence<? extends INodeItem> accept(
39        DynamicContext dynamicContext,
40        ISequence<?> focus) {
41  
42      return ObjectUtils.notNull(focus.stream()
43          .map(ItemUtils::checkItemIsNodeItemForStep)
44          .map(item -> Axis.ANCESTOR_OR_SELF.execute(ObjectUtils.notNull(item)).findFirst().get())
45          .collect(ISequence.toSequence()));
46    }
47  }
