2   package gov.nist.secauto.metaschema.core.metapath.item.node;
3   
4   import java.util.List;
5   
6   import edu.umd.cs.findbugs.annotations.NonNull;
7   
8   /**
9    * Used by implementations of this class to visit a sequence of node items in a
10   * directed graph, using depth-first ordering.
11   *
12   * @param <CONTEXT>
13   *          the type of data to pass to each visited node
14   * @param <RESULT>
15   *          the type of result produced by visitation
16   */
17  public abstract class AbstractNodeItemVisitor<CONTEXT, RESULT> implements INodeItemVisitor<CONTEXT, RESULT> {
18    /**
19     * Visit the provided {@code item}.
20     *
21     * @param item
22     *          the item to visit
23     * @param context
24     *          provides contextual information for use by the visitor
25     * @return the result produced by visiting the item
26     */
27    public final RESULT visit(@NonNull INodeItemVisitable item, CONTEXT context) {
28      return item.accept(this, context);
29    }
30  
31    /**
32     * Visit any child flags associated with the provided {@code item}.
33     *
34     * @param item
35     *          the item to visit
36     * @param context
37     *          provides contextual information for use by the visitor
38     * @return the result produced by visiting the item's flags
39     */
40    protected RESULT visitFlags(@NonNull INodeItem item, CONTEXT context) {
41      RESULT result = defaultResult();
42      for (IFlagNodeItem flag : item.getFlags()) {
43        assert flag != null;
44        if (!shouldVisitNextChild(item, flag, result, context)) {
45          break;
46        }
47  
48        RESULT childResult = flag.accept(this, context);
49        result = aggregateResult(result, childResult, context);
50      }
51      return result;
52    }
53  
54    /**
55     * Visit any child model items associated with the provided {@code item}.
56     *
57     * @param item
58     *          the item to visit
59     * @param context
60     *          provides contextual information for use by the visitor
61     * @return the result produced by visiting the item's child model items
62     */
63    protected RESULT visitModelChildren(@NonNull INodeItem item, CONTEXT context) {
64      RESULT result = defaultResult();
65  
66      for (List<? extends IModelNodeItem<?, ?>> childItems : item.getModelItems()) {
67        for (IModelNodeItem<?, ?> childItem : childItems) {
68          assert childItem != null;
69          if (!shouldVisitNextChild(item, childItem, result, context)) {
70            break;
71          }
72  
73          RESULT childResult = childItem.accept(this, context);
74          result = aggregateResult(result, childResult, context);
75        }
76      }
77      return result;
78    }
79  
80    /**
81     * Determine if the child should be visited next, or skipped.
82     *
83     * @param parent
84     *          the parent of the child to visit next
85     * @param child
86     *          the next child to visit
87     * @param result
88     *          the current visitation result
89     * @param context
90     *          provides contextual information for use by the visitor
91     * @return {@code true} if the child should be visited, or {@code false} if the
92     *         child should be skipped
93     */
94    protected boolean shouldVisitNextChild(
95        @NonNull INodeItem parent,
96        @NonNull INodeItem child,
97        RESULT result,
98        CONTEXT context) {
99      // this is the default behavior, which can be overridden
100     return true;
101   }
102 
103   /**
104    * Determine if the child should be visited next, or skipped.
105    *
106    * @param parent
107    *          the parent of the child to visit next
108    * @param child
109    *          the next child to visit
110    * @param result
111    *          the current visitation result
112    * @param context
113    *          provides contextual information for use by the visitor
114    * @return {@code true} if the child should be visited, or {@code false} if the
115    *         child should be skipped
116    */
117   protected boolean shouldVisitNextChild(
118       @NonNull INodeItem parent,
119       @NonNull IModelNodeItem<?, ?> child,
120       RESULT result,
121       CONTEXT context) {
122     // this is the default behavior, which can be overridden
123     return true;
124   }
125 
126   /**
127    * The initial, default visitation result, which will be used as the basis for
128    * aggregating results produced when visiting.
129    *
130    * @return the default result
131    * @see #aggregateResult(Object, Object, Object)
132    */
133   protected abstract RESULT defaultResult();
134 
135   /**
136    * Combine two results into a single, aggregate result.
137    *
138    * @param first
139    *          the original result
140    * @param second
141    *          the new result to combine with the original result
142    * @param context
143    *          provides contextual information for use by the visitor
144    * @return the combined result
145    */
146   protected RESULT aggregateResult(RESULT first, RESULT second, CONTEXT context) {
147     // this is the default behavior, which can be overridden
148     return second;
149   }
150 
151   @Override
152   public RESULT visitDocument(IDocumentNodeItem item, CONTEXT context) {
153     // this is the default behavior, which can be overridden
154     return visitModelChildren(item, context);
155     // return visitAssembly(item.getRootAssemblyNodeItem(), context);
156   }
157 
158   @Override
159   public RESULT visitFlag(IFlagNodeItem item, CONTEXT context) {
160     // this is the default behavior, which can be overridden
161     return defaultResult();
162   }
163 
164   @Override
165   public RESULT visitField(IFieldNodeItem item, CONTEXT context) {
166     // this is the default behavior, which can be overridden
167     return visitFlags(item, context);
168   }
169 
170   @Override
171   public RESULT visitAssembly(IAssemblyNodeItem item, CONTEXT context) {
172     // this is the default behavior, which can be overridden
173     return aggregateResult(visitFlags(item, context), visitModelChildren(item, context), context);
174   }
175 
176   @Override
177   public RESULT visitAssembly(IAssemblyInstanceGroupedNodeItem item, CONTEXT context) {
178     // this is the default behavior, which can be overridden
179     return aggregateResult(visitFlags(item, context), visitModelChildren(item, context), context);
180   }
181 
182   @Override
183   public RESULT visitMetaschema(IModuleNodeItem item, CONTEXT context) {
184     // this is the default behavior, which can be overridden
185     return aggregateResult(visitFlags(item, context), visitModelChildren(item, context), context);
186   }
