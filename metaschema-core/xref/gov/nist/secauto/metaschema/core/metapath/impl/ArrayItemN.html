6   package gov.nist.secauto.metaschema.core.metapath.impl;
7   
8   import gov.nist.secauto.metaschema.core.metapath.ICollectionValue;
9   import gov.nist.secauto.metaschema.core.util.CollectionUtil;
10  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
11  
12  import java.util.List;
13  
14  import edu.umd.cs.findbugs.annotations.NonNull;
15  
16  /**
17   * An array item that supports an unbounded number of items.
18   *
19   * @param <ITEM>
20   *          the Java type of the items
21   */
22  public class ArrayItemN<ITEM extends ICollectionValue>
23      extends AbstractArrayItem<ITEM> {
24    @NonNull
25    private final List<ITEM> items;
26  
27    /**
28     * Construct a new array item with the provided items.
29     *
30     * @param items
31     *          the items to add to the array
32     */
33    @SafeVarargs
34    public ArrayItemN(@NonNull ITEM... items) {
35      this(ObjectUtils.notNull(List.of(items)));
36    }
37  
38    /**
39     * Construct a new array item using the items from the provided list.
40     *
41     * @param items
42     *          a list containing the items to add to the array
43     */
44    public ArrayItemN(@NonNull List<ITEM> items) {
45      this.items = CollectionUtil.unmodifiableList(items);
46    }
47  
48    @Override
49    public List<ITEM> getValue() {
50      return items;
51    }
52  
53  }
