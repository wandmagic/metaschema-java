6   package gov.nist.secauto.metaschema.core.metapath.item.atomic;
7   
8   import gov.nist.secauto.metaschema.core.datatype.adapter.MetaschemaDataTypeProvider;
9   import gov.nist.secauto.metaschema.core.metapath.function.InvalidValueForCastFunctionException;
10  
11  import edu.umd.cs.findbugs.annotations.NonNull;
12  
13  public interface IHostnameItem extends IStringItem {
14    /**
15     * Construct a new host name item using the provided string {@code value}.
16     *
17     * @param value
18     *          a string representing an host name value
19     * @return the new item
20     */
21    @NonNull
22    static IHostnameItem valueOf(@NonNull String value) {
23      try {
24        return new HostnameItemImpl(MetaschemaDataTypeProvider.HOSTNAME.parse(value));
25      } catch (IllegalArgumentException ex) {
26        throw new InvalidValueForCastFunctionException(String.format("Unable to parse string value '%s'", value),
27            ex);
28      }
29    }
30  
31    /**
32     * Cast the provided type to this item type.
33     *
34     * @param item
35     *          the item to cast
36     * @return the original item if it is already this type, otherwise a new item
37     *         cast to this type
38     * @throws InvalidValueForCastFunctionException
39     *           if the provided {@code item} cannot be cast to this type
40     */
41    @NonNull
42    static IHostnameItem cast(@NonNull IAnyAtomicItem item) {
43      return MetaschemaDataTypeProvider.HOSTNAME.cast(item);
44    }
45  
46    @Override
47    default IHostnameItem castAsType(IAnyAtomicItem item) {
48      return cast(item);
49    }
50  }
