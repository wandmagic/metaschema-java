6   package gov.nist.secauto.metaschema.core.metapath.item.atomic;
7   
8   import gov.nist.secauto.metaschema.core.datatype.IDataTypeAdapter;
9   import gov.nist.secauto.metaschema.core.metapath.IPrintable;
10  import gov.nist.secauto.metaschema.core.metapath.item.function.IMapItem;
11  import gov.nist.secauto.metaschema.core.metapath.item.function.IMapKey;
12  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
13  
14  import java.util.Set;
15  
16  import edu.umd.cs.findbugs.annotations.NonNull;
17  
18  public interface IAnyAtomicItem extends IAtomicValuedItem, IPrintable {
19    @NonNull
20    Set<Class<? extends IAnyAtomicItem>> PRIMITIVE_ITEM_TYPES = ObjectUtils.notNull(Set.of(
21        IStringItem.class,
22        IBooleanItem.class,
23        IDecimalItem.class,
24        IDurationItem.class,
25        IDateTimeItem.class,
26        IDateItem.class,
27        IBase64BinaryItem.class,
28        IAnyUriItem.class));
29  
30    @Override
31    @NonNull
32    default IAnyAtomicItem toAtomicItem() {
33      return this;
34    }
35  
36    /**
37     * Get the "wrapped" value represented by this item.
38     *
39     * @return the value
40     */
41    @Override
42    @NonNull
43    Object getValue();
44  
45    @Override
46    @NonNull
47    String toString();
48  
49    /**
50     * Get the item's string value.
51     *
52     * @return the string value value of the item
53     */
54    @Override
55    @NonNull
56    String asString();
57  
58    /**
59     * Get the atomic item value as a map key for use with an {@link IMapItem}.
60     *
61     * @return the map key
62     */
63    @NonNull
64    IMapKey asMapKey();
65  
66    /**
67     * Get a new {@link IStringItem} based on the the textual value of the item's
68     * "wrapped" value.
69     *
70     * @return a new string item
71     */
72    @NonNull
73    default IStringItem asStringItem() {
74      return IStringItem.valueOf(asString());
75    }
76  
77    /**
78     * Get the item's type adapter.
79     *
80     * @return the type adapter for the item
81     */
82    @NonNull
83    IDataTypeAdapter<?> getJavaTypeAdapter();
84    //
85    // <T extends IValuedItem> T cast(IValuedItem item);
86  
87    /**
88     * Cast the provided {@code item} to be the same type as this item.
89     *
90     * @param item
91     *          the item to cast
92     * @return the result from casting
93     */
94    @NonNull
95    IAnyAtomicItem castAsType(@NonNull IAnyAtomicItem item);
96  
97    /**
98     * Compares this value with the argument. Ordering is item type dependent.
99     *
100    * @param other
101    *          the item to compare with this value
102    * @return a negative integer, zero, or a positive integer if this value is less
103    *         than, equal to, or greater than the {@code item}.
104    */
105   int compareTo(@NonNull IAnyAtomicItem other);
106 }
