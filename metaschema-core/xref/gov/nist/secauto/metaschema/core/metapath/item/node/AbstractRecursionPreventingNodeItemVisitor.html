2   package gov.nist.secauto.metaschema.core.metapath.item.node;
3   
4   import gov.nist.secauto.metaschema.core.model.IAssemblyDefinition;
5   
6   public abstract class AbstractRecursionPreventingNodeItemVisitor<CONTEXT, RESULT>
7       extends AbstractNodeItemVisitor<CONTEXT, RESULT> {
8   
9     @Override
10    public RESULT visitAssembly(IAssemblyNodeItem item, CONTEXT context) {
11      // only walk new records to avoid looping
12      // check if this item's definition is the same as an ancestor's
13      return isDecendant(item, item.getDefinition())
14          ? defaultResult()
15          : super.visitAssembly(item, context);
16    }
17  
18    /**
19     * Determines if the provided node is a descendant of the assembly definition.
20     *
21     * @param node
22     *          the node item to test
23     * @param assemblyDefinition
24     *          the assembly definition to determine as an ancestor of the node
25     * @return {@code true} if the assembly definition is the node's ancestor, or
26     *         {@code false} otherwise
27     */
28    protected boolean isDecendant(IAssemblyNodeItem node, IAssemblyDefinition assemblyDefinition) {
29      return node.ancestor()
30          .map(ancestor -> {
31            boolean retval = false;
32            if (ancestor instanceof IAssemblyNodeItem) {
33              IAssemblyDefinition ancestorDef = ((IAssemblyNodeItem) ancestor).getDefinition();
34              retval = ancestorDef.equals(assemblyDefinition);
35            }
36            return retval;
37          }).anyMatch(value -> value);
38    }
39  }
