2   package gov.nist.secauto.metaschema.core.metapath.item.node;
3   
4   import gov.nist.secauto.metaschema.core.model.IAssemblyDefinition;
5   import gov.nist.secauto.metaschema.core.util.ObjectUtils;
6   
7   import edu.umd.cs.findbugs.annotations.NonNull;
8   import nl.talsmasoftware.lazy4j.Lazy;
9   
10  class RootAssemblyValuedNodeItemImpl
11      implements IRootAssemblyNodeItem, IFeatureModelContainerItem {
12    @NonNull
13    private final IAssemblyDefinition definition;
14    @NonNull
15    private final IDocumentNodeItem parent;
16  
17    @NonNull
18    private final Lazy<ModelContainer> model;
19    @NonNull
20    private final Object value;
21  
22    public RootAssemblyValuedNodeItemImpl(
23        @NonNull IAssemblyDefinition definition,
24        @NonNull IDocumentNodeItem parent,
25        @NonNull Object value,
26        @NonNull INodeItemGenerator generator) {
27      this.definition = definition;
28      this.parent = parent;
29      this.model = ObjectUtils.notNull(Lazy.lazy(generator.newDataModelSupplier((IAssemblyNodeItem) this)));
30      this.value = value;
31    }
32  
33    @Override
34    public IAssemblyDefinition getDefinition() {
35      return definition;
36    }
37  
38    @Override
39    public IDocumentNodeItem getDocumentNodeItem() {
40      return parent;
41    }
42  
43    @Override
44    public IAssemblyNodeItem getParentContentNodeItem() {
45      // there is no parent assembly
46      return null;
47    }
48  
49    @Override
50    @NonNull
51    public Object getValue() {
52      return value;
53    }
54  
55    @SuppressWarnings("null")
56    @Override
57    public ModelContainer getModel() {
58      return model.get();
59    }
60  }
