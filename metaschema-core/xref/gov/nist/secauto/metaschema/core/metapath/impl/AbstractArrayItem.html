7   
8   import gov.nist.secauto.metaschema.core.metapath.DynamicContext;
9   import gov.nist.secauto.metaschema.core.metapath.ICollectionValue;
10  import gov.nist.secauto.metaschema.core.metapath.ISequence;
11  import gov.nist.secauto.metaschema.core.metapath.function.FunctionUtils;
12  import gov.nist.secauto.metaschema.core.metapath.function.IArgument;
13  import gov.nist.secauto.metaschema.core.metapath.function.ISequenceType;
14  import gov.nist.secauto.metaschema.core.metapath.function.Occurrence;
15  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IAnyAtomicItem;
16  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IIntegerItem;
17  import gov.nist.secauto.metaschema.core.metapath.item.function.IArrayItem;
18  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
19  
20  import java.util.EnumSet;
21  import java.util.List;
22  import java.util.Objects;
23  import java.util.Set;
24  
25  import javax.xml.namespace.QName;
26  
27  import edu.umd.cs.findbugs.annotations.NonNull;
28  
29  public abstract class AbstractArrayItem<ITEM extends ICollectionValue>
30      extends ImmutableCollections.AbstractImmutableDelegatedList<ITEM>
31      implements IArrayItem<ITEM> {
32    @NonNull
33    public static final QName QNAME = new QName("array");
34    @NonNull
35    public static final Set<FunctionProperty> PROPERTIES = ObjectUtils.notNull(
36        EnumSet.of(FunctionProperty.DETERMINISTIC));
37    @NonNull
38    public static final List<IArgument> ARGUMENTS = ObjectUtils.notNull(List.of(
39        IArgument.builder().name("position").type(IIntegerItem.class).one().build()));
40    @NonNull
41    public static final ISequenceType RESULT = ISequenceType.of(IAnyAtomicItem.class, Occurrence.ZERO_OR_ONE);
42  
43    @SuppressWarnings({ "rawtypes", "unchecked" })
44    @NonNull
45    private static final IArrayItem<?> EMPTY = new ArrayItemN<>();
46  
47    /**
48     * Get an immutable array item that is empty.
49     *
50     * @param <T>
51     *          the item Java type
52     * @return the empty array item
53     */
54    @SuppressWarnings("unchecked")
55    @NonNull
56    public static <T extends ICollectionValue> IArrayItem<T> empty() {
57      return (IArrayItem<T>) EMPTY;
58    }
59  
60    @Override
61    public ISequence<?> execute(List<ISequence<?>> arguments, DynamicContext dynamicContext,
62        ISequence<?> focus) {
63      ISequence<? extends IIntegerItem> arg = FunctionUtils.asType(
64          ObjectUtils.notNull(arguments.get(0)));
65  
66      IIntegerItem position = arg.getFirstItem(true);
67      if (position == null) {
68        return ISequence.empty(); // NOPMD - readability
69      }
70  
71      int index = position.asInteger().intValueExact() - 1;
72      ICollectionValue result = getValue().get(index);
73      return result.asSequence();
74    }
75  
76    @Override
77    public int hashCode() {
78      return Objects.hash(getValue());
79    }
80  
81    @Override
82    public boolean equals(Object other) {
83      return other == this
84          || other instanceof IArrayItem && getValue().equals(((IArrayItem<?>) other).getValue());
85    }
86  
87    @Override
88    public String asString() {
89      return ObjectUtils.notNull(toString());
90    }
91  }
