6   package gov.nist.secauto.metaschema.core.metapath;
7   
8   import gov.nist.secauto.metaschema.core.configuration.AbstractConfigurationFeature;
9   
10  import edu.umd.cs.findbugs.annotations.NonNull;
11  
12  /**
13   * Provides a mechanism to configure Metapath evaluation settings.
14   *
15   * @param <V>
16   *          the feature value Java type
17   */
18  public final class MetapathEvaluationFeature<V>
19      extends AbstractConfigurationFeature<V> {
20    /**
21     * If enabled, evaluate <a href=
22     * "https://www.w3.org/TR/xpath-31/#id-filter-expression">predicates</a>,
23     * otherwise skip evaluating them.
24     */
25    @NonNull
26    public static final MetapathEvaluationFeature<Boolean> METAPATH_EVALUATE_PREDICATES
27        = new MetapathEvaluationFeature<>("evaluate-predicates", Boolean.class, true);
28  
29    private MetapathEvaluationFeature(
30        @NonNull String name,
31        @NonNull Class<V> valueClass,
32        @NonNull V defaultValue) {
33      super(name, valueClass, defaultValue);
34    }
35  }
