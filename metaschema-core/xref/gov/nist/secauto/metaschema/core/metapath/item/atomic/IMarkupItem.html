6   package gov.nist.secauto.metaschema.core.metapath.item.atomic;
7   
8   import gov.nist.secauto.metaschema.core.datatype.markup.IMarkupString;
9   import gov.nist.secauto.metaschema.core.datatype.markup.MarkupDataTypeProvider;
10  import gov.nist.secauto.metaschema.core.datatype.markup.MarkupLine;
11  import gov.nist.secauto.metaschema.core.datatype.markup.MarkupMultiline;
12  import gov.nist.secauto.metaschema.core.metapath.function.InvalidValueForCastFunctionException;
13  
14  import edu.umd.cs.findbugs.annotations.NonNull;
15  
16  public interface IMarkupItem extends IUntypedAtomicItem {
17    /**
18     * Construct a new item using the provided {@code value}.
19     *
20     * @param value
21     *          a line of markup
22     * @return the new item
23     */
24    @NonNull
25    static IMarkupItem valueOf(@NonNull MarkupLine value) {
26      return new MarkupLineItemImpl(value);
27    }
28  
29    /**
30     * Construct a new item using the provided {@code value}.
31     *
32     * @param value
33     *          multiple lines of markup
34     * @return the new item
35     */
36    @NonNull
37    static IMarkupItem valueOf(@NonNull MarkupMultiline value) {
38      return new MarkupMultiLineItemImpl(value);
39    }
40  
41    /**
42     * Cast the provided type to this item type.
43     *
44     * @param item
45     *          the item to cast
46     * @return the original item if it is already this type, otherwise a new item
47     *         cast to this type
48     * @throws InvalidValueForCastFunctionException
49     *           if the provided {@code item} cannot be cast to this type
50     */
51    @NonNull
52    static IMarkupItem cast(@NonNull IAnyAtomicItem item) {
53      return MarkupDataTypeProvider.MARKUP_MULTILINE.cast(item);
54    }
55  
56    /**
57     * Get the "wrapped" markup value.
58     *
59     * @return the underlying markup value
60     */
61    @NonNull
62    IMarkupString<?> asMarkup();
63  
64    @Override
65    default IMarkupItem castAsType(IAnyAtomicItem item) {
66      return cast(item);
67    }
68  
69    @Override
70    default int compareTo(IAnyAtomicItem item) {
71      return compareTo(cast(item));
72    }
73  
74    /**
75     * Compares this value with the argument.
76     *
77     * @param item
78     *          the item to compare with this value
79     * @return a negative integer, zero, or a positive integer if this value is less
80     *         than, equal to, or greater than the {@code item}.
81     */
82    default int compareTo(@NonNull IMarkupItem item) {
83      return asString().compareTo(item.asString());
84    }
85  }
