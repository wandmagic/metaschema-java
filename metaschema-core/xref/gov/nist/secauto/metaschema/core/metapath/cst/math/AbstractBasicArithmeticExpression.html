6   package gov.nist.secauto.metaschema.core.metapath.cst.math;
7   
8   import gov.nist.secauto.metaschema.core.metapath.DynamicContext;
9   import gov.nist.secauto.metaschema.core.metapath.ISequence;
10  import gov.nist.secauto.metaschema.core.metapath.cst.IExpression;
11  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IAnyAtomicItem;
12  
13  import edu.umd.cs.findbugs.annotations.NonNull;
14  import edu.umd.cs.findbugs.annotations.Nullable;
15  
16  public abstract class AbstractBasicArithmeticExpression
17      extends AbstractArithmeticExpression<IAnyAtomicItem> {
18  
19    /**
20     * An expression that represents a basic arithmetic operation on two values.
21     *
22     * @param left
23     *          the first item
24     * @param right
25     *          the second item
26     */
27    public AbstractBasicArithmeticExpression(@NonNull IExpression left, @NonNull IExpression right) {
28      super(left, right, IAnyAtomicItem.class);
29    }
30  
31    @Override
32    public Class<IAnyAtomicItem> getBaseResultType() {
33      return IAnyAtomicItem.class;
34    }
35  
36    @Override
37    public ISequence<? extends IAnyAtomicItem> accept(DynamicContext dynamicContext, ISequence<?> focus) {
38      IAnyAtomicItem leftItem = getFirstDataItem(getLeft().accept(dynamicContext, focus), true);
39      IAnyAtomicItem rightItem = getFirstDataItem(getRight().accept(dynamicContext, focus), true);
40  
41      return resultOrEmpty(leftItem, rightItem);
42    }
43  
44    /**
45     * Setup the operation on two atomic items.
46     *
47     * @param leftItem
48     *          the first item
49     * @param rightItem
50     *          the second item
51     * @return the result of the operation or an empty {@link ISequence} if either
52     *         item is {@code null}
53     */
54    @NonNull
55    protected ISequence<? extends IAnyAtomicItem> resultOrEmpty(
56        @Nullable IAnyAtomicItem leftItem,
57        @Nullable IAnyAtomicItem rightItem) {
58      ISequence<? extends IAnyAtomicItem> retval;
59      if (leftItem == null || rightItem == null) {
60        retval = ISequence.empty();
61      } else {
62        IAnyAtomicItem result = operation(leftItem, rightItem);
63        retval = ISequence.of(result);
64      }
65      return retval;
66    }
67  
68    /**
69     * Performs the arithmetic operation using the two provided values.
70     *
71     * @param left
72     *          the first item
73     * @param right
74     *          the second item
75     * @return the result of the operation
76     */
77    @NonNull
78    protected abstract IAnyAtomicItem operation(@NonNull IAnyAtomicItem left, @NonNull IAnyAtomicItem right);
79  }
