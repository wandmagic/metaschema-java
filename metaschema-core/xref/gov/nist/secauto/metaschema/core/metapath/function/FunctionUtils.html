94    /**
95     * Gets the first item of the provided sequence as a {@link INumericItem} value.
96     * If the sequence is empty, then a {@code null} value is returned.
97     *
98     * @param sequence
99     *          a Metapath sequence containing the value to convert
100    * @param requireSingleton
101    *          if {@code true} then a {@link TypeMetapathException} is thrown if
102    *          the sequence contains more than one item
103    * @return the numeric item value, or {@code null} if the result is an empty
104    *         sequence
105    * @throws TypeMetapathException
106    *           if the sequence contains more than one item, or the item cannot be
107    *           cast to a numeric value
108    *
109    */
110   @Nullable
111   public static INumericItem toNumeric(@NonNull ISequence<?> sequence, boolean requireSingleton) {
112     IItem item = sequence.getFirstItem(requireSingleton);
113     return item == null ? null : toNumeric(item);
114   }

116   /**
117    * Gets the provided item value as a {@link INumericItem} value.
118    *
119    * @param item
120    *          the value to convert
121    * @return the numeric item value
122    * @throws TypeMetapathException
123    *           if the sequence contains more than one item, or the item cannot be
124    *           cast to a numeric value
125    */
126   @NonNull
127   public static INumericItem toNumeric(@NonNull IItem item) {
128     // atomize
129     IAnyAtomicItem atomicItem = ISequence.getFirstItem(FnData.atomize(item), true);
130     return toNumeric(atomicItem);
131   }

133   /**
134    * Gets the provided item value as a {@link INumericItem} value.
135    *
136    * @param item
137    *          the value to convert
138    * @return the numeric item value
139    * @throws TypeMetapathException
140    *           if the item cannot be cast to a numeric value
141    */
142   @NonNull
143   public static INumericItem toNumeric(@Nullable IAnyAtomicItem item) {
144     try {
145       return IDecimalItem.cast(item);
146     } catch (InvalidValueForCastFunctionException ex) {
147       throw new InvalidTypeMetapathException(item, ex.getLocalizedMessage(), ex);
148     }
149   }

219   /**
220    * Casts the provided {@code item} as the result type.
221    *
222    * @param <TYPE>
223    *          the Java type to cast to
224    * @param clazz
225    *          the Java class instance for the requested type
226    * @param item
227    *          the value to cast
228    * @return the item cast to the required type
229    * @throws InvalidTypeMetapathException
230    *           if the provided item is {@code null} or if the item's type is not
231    *           assignment compatible to the requested type
232    */
233   @NonNull
234   public static <TYPE extends IItem> TYPE requireType(Class<TYPE> clazz, IItem item) {
235     if (item == null) {
236       throw new InvalidTypeMetapathException(
237           null,
238           String.format("Expected non-null type '%s', but the node was null.",
239               clazz.getName()));
240     } else if (!clazz.isInstance(item)) {
241       throw new InvalidTypeMetapathException(
242           item,
243           String.format("Expected type '%s', but the node was type '%s'.",
244               clazz.getName(),
245               item.getClass().getName()));
246     }
247     return asType(item);
248   }
249 
250   /**
251    * Casts the provided {@code item} as the result type, if the item is not
252    * {@code null}.
253    *
254    * @param <TYPE>
255    *          the Java type to cast to
256    * @param clazz
257    *          the Java class instance for the requested type
258    * @param item
259    *          the value to cast
260    * @return the item cast to the required type or {@code null} if the item is
261    *         {@code null}
262    * @throws InvalidTypeMetapathException
263    *           if the provided item is {@code null} or if the item's type is not
264    *           assignment compatible to the requested type
265    */
266   @Nullable
267   public static <TYPE extends IItem> TYPE requireTypeOrNull(Class<TYPE> clazz, @Nullable IItem item) {
268     if (item == null || clazz.isInstance(item)) {
269       return asTypeOrNull(item);
270     }
271     throw new InvalidTypeMetapathException(
272         item,
273         String.format("Expected type '%s', but the node was type '%s'.",
274             clazz.getName(),
275             item.getClass().getName()));
276   }

278   /**
279    * Get a stream of item data types for the stream of items.
280    *
281    * @param items
282    *          the Metapath items to get the data types for
283    * @return a stream of data type classes

286   public static Stream<Class<?>> getTypes(@NonNull Stream<? extends IItem> items) {

288   }

290   /**
291    * Generate a list of Metapath item Java type classes from a list of Metapath
292    * items.
293    *
294    * @param <T>
295    *          the base Java types of the items
296    * @param items
297    *          the items to get Java type class for
298    * @return a list of corresponding Java type classes for the provided items
299    */
300   @SuppressWarnings("unchecked")
301   @NonNull
302   public static <T extends IItem> List<Class<? extends T>> getTypes(@NonNull List<T> items) {
303     return ObjectUtils.notNull(items.stream()
304         .map(item -> (Class<? extends T>) item.getClass())
305         .collect(Collectors.toList()));
306   }

308   /**
309    * Count the occurrences of the provided data type item {@code classes} used in
310    * the set of provided {@code items}.
311    *
312    * @param <T>
313    *          the class type
314    * @param classes
315    *          the Metapath item classes to count
316    * @param items
317    *          the Metapath items to analyze
318    * @return a mapping of Metapath item class to count
319    */
320   @NonNull
321   public static <T extends IItem> Map<Class<? extends T>, Integer> countTypes(
322       @NonNull Set<Class<? extends T>> classes,
323       @NonNull Collection<? extends T> items) {
324     Map<Class<? extends T>, Integer> retval = new HashMap<>(); // NOPMD
325     for (T item : items) {
326       Class<?> itemClass = item.getClass();
327       for (Class<? extends T> clazz : classes) {
328         if (clazz.isAssignableFrom(itemClass)) {
329           retval.compute(clazz, (cl, current) -> current == null ? 1 : current + 1);
330         }
331       }
332     }
333     return retval;
334   }
