21  /**
22   * An implementation of
23   * <a href="https://www.w3.org/TR/xpath-31/#id-unary-lookup">Unary Lookup
24   * Operators</a> supporting access to items in Metapath maps and arrays.
25   * <p>
26   * Provides support for various types of key- and index-based lookups related to
27   * {@link IMapItem} and {@link IArrayItem} objects.
28   */
29  public class UnaryLookup
30      extends AbstractLookup {
31    /**
32     * Construct a new unary lookup expression that uses the provided key specifier.
33     *
34     * @param keySpecifier
35     *          the key specifier used to determine matching entries
36     */
37    public UnaryLookup(@NonNull IKeySpecifier keySpecifier) {
38      super(keySpecifier);
39    }
40  
41    @SuppressWarnings("null")
42    @Override
43    public List<? extends IExpression> getChildren() {
44      return List.of();
45    }
46  
47    @Override
48    public ISequence<? extends IItem> accept(DynamicContext dynamicContext, ISequence<?> focus) {
49      IKeySpecifier specifier = getKeySpecifier();
50  
51      return ISequence.of(ObjectUtils.notNull(focus.stream()
52          .flatMap(item -> {
53            assert item != null;
54            return specifier.lookup(item, dynamicContext, focus);
55          })
56          .flatMap(ICollectionValue::normalizeAsItems)));
57    }
58  
59    @Override
60    public <RESULT, CONTEXT> RESULT accept(@NonNull IExpressionVisitor<RESULT, CONTEXT> visitor, CONTEXT context) {
61      return visitor.visitUnaryLookup(this, context);
62    }
63  }
