2   package gov.nist.secauto.metaschema.core.metapath.item.node;
3   
4   import gov.nist.secauto.metaschema.core.metapath.item.atomic.IAnyAtomicItem;
5   import gov.nist.secauto.metaschema.core.model.IFlagInstance;
6   import gov.nist.secauto.metaschema.core.util.ObjectUtils;
7   
8   import edu.umd.cs.findbugs.annotations.NonNull;
9   import nl.talsmasoftware.lazy4j.Lazy;
10  
11  /**
12   * A {@link INodeItem} supported by a {@link IFlagInstance}, that may have an
13   * associated value.
14   */
15  class FlagInstanceNodeItemImpl
16      extends AbstractFlagInstanceNodeItem
17      implements IFeatureAtomicValuedItem,
18      IFeatureChildNodeItem {
19  
20    @NonNull
21    private final Object value;
22  
23    /**
24     * Used to cache this object as an atomic item.
25     */
26    @NonNull
27    private final Lazy<IAnyAtomicItem> atomicItem;
28  
29    public FlagInstanceNodeItemImpl(
30        @NonNull IFlagInstance instance,
31        @NonNull IModelNodeItem<?, ?> parent,
32        @NonNull Object value) {
33      super(instance, parent);
34      this.value = value;
35      this.atomicItem = ObjectUtils.notNull(Lazy.lazy(this::newAtomicItem));
36    }
37  
38    @Override
39    @NonNull
40    public Object getValue() {
41      return value;
42    }
43  
44    @Override
45    public Object getAtomicValue() {
46      return getValue();
47    }
48  
49    @Override
50    public IAnyAtomicItem toAtomicItem() {
51      return atomicItem.get();
52    }
53  }
