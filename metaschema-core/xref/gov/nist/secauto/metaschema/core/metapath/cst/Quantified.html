30  public class Quantified
31      extends AbstractExpression {
32    public enum Quantifier {
33      SOME,
34      EVERY;
35    }
36  
37    @NonNull
38    private final Quantifier quantifier;
39    @NonNull
40    private final Map<QName, IExpression> inClauses;
41    @NonNull
42    private final IExpression satisfies;
43  
44    /**
45     * Construct a new quantified expression.
46     *
47     * @param quantifier
48     *          the quantifier operation
49     * @param inClauses
50     *          the set of expressions that define the variables to use for
51     *          determining the Cartesian product for evaluation
52     * @param satisfies
53     *          the expression used for evaluation using the Cartesian product of
54     *          the variables
55     */
56    public Quantified(
57        @NonNull Quantifier quantifier,
58        @NonNull Map<QName, IExpression> inClauses,
59        @NonNull IExpression satisfies) {
60      this.quantifier = quantifier;
61      this.inClauses = inClauses;
62      this.satisfies = satisfies;
63    }
64  
65    /**
66     * Get the quantifier operation.
67     *
68     * @return the quantifier operations
69     */
70    @NonNull
71    public Quantifier getQuantifier() {
72      return quantifier;
73    }
74  
75    /**
76     * Get the set of expressions that define the variables to use for determining
77     * the Cartesian product for evaluation.
78     *
79     * @return the variable names mapped to the associated Metapath expression
80     */
81    @NonNull
82    public Map<QName, IExpression> getInClauses() {
83      return inClauses;
84    }
85  
86    /**
87     * Get the expression used for evaluation using the Cartesian product of the
88     * variables.
89     *
90     * @return the evaluation expression
91     */
92    @NonNull
93    public IExpression getSatisfies() {
94      return satisfies;
95    }
96  
97    @Override
98    public List<? extends IExpression> getChildren() {
99      return ObjectUtils.notNull(Stream.concat(inClauses.values().stream(), Stream.of(satisfies))
100         .collect(Collectors.toList()));
101   }

103   @SuppressWarnings("PMD.SystemPrintln")
104   @Override
105   public ISequence<? extends IItem> accept(DynamicContext dynamicContext, ISequence<?> focus) {
106     Map<QName, ISequence<? extends IItem>> clauses = getInClauses().entrySet().stream()
107         .map(entry -> Map.entry(
108             entry.getKey(),
109             entry.getValue().accept(dynamicContext, focus)))
110         .collect(Collectors.toMap(Map.Entry::getKey, Map.Entry::getValue, (e1, e2) -> e1, LinkedHashMap::new));
111 
112     List<QName> clauseKeys = new ArrayList<>(clauses.keySet());
113     List<? extends Collection<? extends IItem>> clauseValues = new ArrayList<>(clauses.values());
114 
115     boolean retval = true;
116     for (List<IItem> product : new CartesianProduct<>(clauseValues)) {
117       DynamicContext subDynamicContext = dynamicContext.subContext();
118       for (int idx = 0; idx < product.size(); idx++) {
119         QName var = clauseKeys.get(idx);
120         IItem item = product.get(idx);
121 
122         assert var != null;
123 
124         subDynamicContext.bindVariableValue(var, ISequence.of(item));
125       }
126       boolean result = FnBoolean.fnBooleanAsPrimitive(getSatisfies().accept(subDynamicContext, focus));
127       if (Quantifier.EVERY.equals(quantifier) && !result) {
128         // fail on first false
129         retval = false;
130         break;
131       } else if (Quantifier.SOME.equals(quantifier)) {
132         if (result) {
133           // pass on first true
134           retval = true;
135           break;
136         }
137         // store (false) result
138         retval = false;
139       }
140     }
141 
142     return ISequence.of(IBooleanItem.valueOf(retval));
143   }

145   @Override
146   public <RESULT, CONTEXT> RESULT accept(IExpressionVisitor<RESULT, CONTEXT> visitor, CONTEXT context) {
147     return visitor.visitQuantified(this, context);
148   }

150   /**
151    * Get the Cartesian product of the provided lists of value axis.
152    *
153    * @param <T>
154    *          the Java type of value item
155    * @param axes
156    *          the values to compute the Cartesian product of
157    * @return an iterator of lists contain the Cartesian product of the axis values
158    */
159   public static <T extends IItem> Iterable<List<T>> cartesianProduct(
160       @NonNull List<? extends Collection<? extends T>> axes) {
161     return new CartesianProduct<>(axes);
162   }

164   // based on https://gist.github.com/jhorstmann/a7aba9947bc4926a75f6de8f69560c6e
165   private static class CartesianProductIterator<T extends IItem> implements Iterator<List<T>> {
166     private final Object[][] dimensions;
167     private final int length;
168     private final int[] indizes;
169     private boolean reachedMax;
170 
171     @SuppressWarnings({
172         "PMD.UseVarargs",
173         "PMD.ArrayIsStoredDirectly" // ok for internal use
174     })
175     CartesianProductIterator(final Object[][] dimensions) {
176       this.dimensions = dimensions;
177       this.length = dimensions.length;
178       this.indizes = new int[length];
179     }
180 
181     private void increment(final int index) {
182       if (index >= length) {
183         reachedMax = true;
184       } else {
185         indizes[index]++;
186         if (indizes[index] == dimensions[index].length) {
187           indizes[index] = 0;
188           increment(index + 1);
189         }
190       }
191     }
192 
193     private void increment() {
194       increment(0);
195     }
196 
197     @SuppressWarnings("unchecked")
198     @Override
199     public List<T> next() {
200       if (reachedMax) {
201         throw new NoSuchElementException();
202       }
203 
204       List<T> list = new ArrayList<>();
205       for (int i = 0; i < length; i++) {
206         list.add((T) dimensions[i][indizes[i]]);
207       }
208 
209       increment();
210 
211       return Collections.unmodifiableList(list);
212     }
213 
214     @Override
215     public boolean hasNext() {
216       return !reachedMax;
217     }
218 
219     @Override
220     public void remove() {
221       throw new UnsupportedOperationException("remove not supported");
222     }
223   }

225   // based on https://gist.github.com/jhorstmann/a7aba9947bc4926a75f6de8f69560c6e
226   private static final class CartesianProduct<T extends IItem> implements Iterable<List<T>> {
227     private final Object[][] dimensions;
228     private final long size;
229 
230     private CartesianProduct(final List<? extends Collection<? extends T>> axes) {
231       Object[][] dimensions = new Object[axes.size()][];
232       long size = dimensions.length == 0 ? 0 : 1;
233       for (int i = 0; i < axes.size(); i++) {
234         dimensions[i] = axes.get(i).toArray();
235         size *= dimensions[i].length;
236       }
237       this.dimensions = dimensions;
238       this.size = size;
239     }
240 
241     @SuppressWarnings("PMD.OnlyOneReturn") // readability
242     @Override
243     public Iterator<List<T>> iterator() {
244       if (size == 0) {
245         return Collections.emptyListIterator();
246       }
247       return new CartesianProductIterator<>(dimensions);
248     }
