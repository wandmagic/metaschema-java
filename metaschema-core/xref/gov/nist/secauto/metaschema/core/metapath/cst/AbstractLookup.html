6   package gov.nist.secauto.metaschema.core.metapath.cst;
7   
8   import gov.nist.secauto.metaschema.core.metapath.item.function.IArrayItem;
9   import gov.nist.secauto.metaschema.core.metapath.item.function.IKeySpecifier;
10  import gov.nist.secauto.metaschema.core.metapath.item.function.IMapItem;
11  
12  import edu.umd.cs.findbugs.annotations.NonNull;
13  
14  /**
15   * An implementation of
16   * <a href="https://www.w3.org/TR/xpath-31/#id-lookup">Lookup Operators</a>
17   * supporting access to items in Metapath maps and arrays.
18   * <p>
19   * Provides support for various types of key- and index-based lookups related to
20   * {@link IMapItem} and {@link IArrayItem} objects.
21   */
22  public abstract class AbstractLookup implements IExpression {
23    @NonNull
24    private final IKeySpecifier keySpecifier;
25  
26    /**
27     * Construct a new lookup expression that uses the provided key specifier.
28     *
29     * @param keySpecifier
30     *          the key specifier that identifies how to lookup entries
31     */
32    protected AbstractLookup(@NonNull IKeySpecifier keySpecifier) {
33      this.keySpecifier = keySpecifier;
34    }
35  
36    /**
37     * Get the key specifier implementation.
38     *
39     * @return the key specifier
40     */
41    @NonNull
42    public IKeySpecifier getKeySpecifier() {
43      return keySpecifier;
44    }
45  }
