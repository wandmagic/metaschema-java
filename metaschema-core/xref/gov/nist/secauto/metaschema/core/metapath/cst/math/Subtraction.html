6   package gov.nist.secauto.metaschema.core.metapath.cst.math;
7   
8   import gov.nist.secauto.metaschema.core.metapath.ISequence;
9   import gov.nist.secauto.metaschema.core.metapath.InvalidTypeMetapathException;
10  import gov.nist.secauto.metaschema.core.metapath.cst.IExpression;
11  import gov.nist.secauto.metaschema.core.metapath.cst.IExpressionVisitor;
12  import gov.nist.secauto.metaschema.core.metapath.function.FunctionUtils;
13  import gov.nist.secauto.metaschema.core.metapath.function.OperationFunctions;
14  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IAnyAtomicItem;
15  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IDateItem;
16  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IDateTimeItem;
17  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IDayTimeDurationItem;
18  import gov.nist.secauto.metaschema.core.metapath.item.atomic.INumericItem;
19  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IYearMonthDurationItem;
20  
21  import edu.umd.cs.findbugs.annotations.NonNull;
22  
23  public class Subtraction
24      extends AbstractBasicArithmeticExpression {
25  
26    /**
27     * An expression that gets the difference of two atomic data items.
28     *
29     * @param minuend
30     *          an expression whose result is the value being subtracted from
31     * @param subtrahend
32     *          an expression whose result is the value being subtracted
33     */
34    public Subtraction(@NonNull IExpression minuend, @NonNull IExpression subtrahend) {
35      super(minuend, subtrahend);
36    }
37  
38    @Override
39    public <RESULT, CONTEXT> RESULT accept(IExpressionVisitor<RESULT, CONTEXT> visitor, CONTEXT context) {
40      return visitor.visitSubtraction(this, context);
41    }
42  
43    /**
44     * Get the difference of two atomic items.
45     *
46     * @param minuend
47     *          the item being subtracted from
48     * @param subtrahend
49     *          the item being subtracted
50     * @return the difference of the items or an empty {@link ISequence} if either
51     *         item is {@code null}
52     */
53    @Override
54    @NonNull
55    protected IAnyAtomicItem operation(@NonNull IAnyAtomicItem minuend, @NonNull IAnyAtomicItem subtrahend) {
56      return subtract(minuend, subtrahend);
57    }
58  
59    /**
60     * Get the difference of two atomic items.
61     *
62     * @param minuend
63     *          the item being subtracted from
64     * @param subtrahend
65     *          the item being subtracted
66     * @return the difference of the items
67     */
68    @NonNull
69    public static IAnyAtomicItem subtract(@NonNull IAnyAtomicItem minuend, // NOPMD - intentional
70        @NonNull IAnyAtomicItem subtrahend) {
71  
72      IAnyAtomicItem retval = null;
73      if (minuend instanceof IDateItem) {
74        IDateItem left = (IDateItem) minuend;
75  
76        if (subtrahend instanceof IDateItem) {
77          retval = OperationFunctions.opSubtractDates(left, (IDateItem) subtrahend);
78        } else if (subtrahend instanceof IYearMonthDurationItem) {
79          retval = OperationFunctions.opSubtractYearMonthDurationFromDate(left, (IYearMonthDurationItem) subtrahend);
80        } else if (subtrahend instanceof IDayTimeDurationItem) {
81          retval = OperationFunctions.opSubtractDayTimeDurationFromDate(left, (IDayTimeDurationItem) subtrahend);
82        }
83      } else if (minuend instanceof IDateTimeItem) {
84        IDateTimeItem left = (IDateTimeItem) minuend;
85        if (subtrahend instanceof IDateTimeItem) {
86          retval = OperationFunctions.opSubtractDateTimes(left, (IDateTimeItem) subtrahend);
87        } else if (subtrahend instanceof IYearMonthDurationItem) {
88          retval = OperationFunctions.opSubtractYearMonthDurationFromDateTime(left, (IYearMonthDurationItem) subtrahend);
89        } else if (subtrahend instanceof IDayTimeDurationItem) {
90          retval = OperationFunctions.opSubtractDayTimeDurationFromDateTime(left, (IDayTimeDurationItem) subtrahend);
91        }
92      } else if (minuend instanceof IYearMonthDurationItem) {
93        IYearMonthDurationItem left = (IYearMonthDurationItem) minuend;
94        if (subtrahend instanceof IYearMonthDurationItem) {
95          retval = OperationFunctions.opSubtractYearMonthDurations(left, (IYearMonthDurationItem) subtrahend);
96        }
97      } else if (minuend instanceof IDayTimeDurationItem) {
98        IDayTimeDurationItem left = (IDayTimeDurationItem) minuend;
99        if (subtrahend instanceof IDayTimeDurationItem) {
100         retval = OperationFunctions.opSubtractDayTimeDurations(left, (IDayTimeDurationItem) subtrahend);
101       }
102     } else {
103       // handle as numeric
104       INumericItem left = FunctionUtils.toNumeric(minuend);
105       INumericItem right = FunctionUtils.toNumeric(subtrahend);
106       retval = OperationFunctions.opNumericSubtract(left, right);
107     }
108     if (retval == null) {
109       throw new InvalidTypeMetapathException(
110           null,
111           String.format("The expression '%s - %s' is not supported", minuend.getClass().getName(),
112               subtrahend.getClass().getName()));
113     }
114     return retval;
115   }
116 }
