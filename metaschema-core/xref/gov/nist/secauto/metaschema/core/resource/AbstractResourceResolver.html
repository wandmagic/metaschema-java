6   package gov.nist.secauto.metaschema.core.resource;
7   
8   import gov.nist.secauto.metaschema.core.model.IResourceResolver;
9   import gov.nist.secauto.metaschema.core.model.IUriResolver;
10  
11  import edu.umd.cs.findbugs.annotations.NonNull;
12  
13  public class AbstractResourceResolver implements IResourceResolver {
14    /**
15     * An {@link IUriResolver} is not provided by default.
16     */
17    private IUriResolver uriResolver;
18  
19    @Override
20    public IUriResolver getUriResolver() {
21      return uriResolver;
22    }
23  
24    /**
25     * Set the URI resolver for this resource resolver.
26     *
27     * @param uriResolver
28     *          the URI resolver
29     */
30    public void setUriResolver(@NonNull IUriResolver uriResolver) {
31      this.uriResolver = uriResolver;
32    }
33  }
