6   package gov.nist.secauto.metaschema.core.datatype.adapter;
7   
8   import java.time.format.DateTimeFormatter;
9   import java.time.format.DateTimeFormatterBuilder;
10  import java.time.temporal.ChronoField;
11  
12  /**
13   * Provides common date formats used by the data type implementations.
14   */
15  @SuppressWarnings("PMD.DataClass")
16  final class DateFormats {
17  
18    // public static final DateTimeFormatter dateWithOptionalTZ;
19    public static final DateTimeFormatter DATE_WITH_TZ;
20    public static final DateTimeFormatter DATE_WITHOUT_TZ;
21    // public static final DateTimeFormatter dateTimeWithOptionalTZ;
22    public static final DateTimeFormatter DATE_TIME_WITH_TZ;
23    public static final DateTimeFormatter DATE_TIME_WITHOUT_TZ;
24  
25    static {
26      DATE_WITH_TZ = new DateTimeFormatterBuilder()
27          .appendPattern("yyyy-MM-dd")
28          .appendPattern("XXX")
29          .toFormatter();
30      DATE_WITHOUT_TZ = new DateTimeFormatterBuilder()
31          .appendPattern("yyyy-MM-dd")
32          .toFormatter();
33      DATE_TIME_WITH_TZ = new DateTimeFormatterBuilder()
34          .appendPattern("yyyy-MM-dd'T'HH:mm:ss")
35          .optionalStart()
36          .appendFraction(ChronoField.NANO_OF_SECOND, 0, 9, true)
37          .optionalEnd()
38          .appendPattern("XXX")
39          .toFormatter();
40      DATE_TIME_WITHOUT_TZ = new DateTimeFormatterBuilder()
41          .appendPattern("yyyy-MM-dd'T'HH:mm:ss")
42          .optionalStart()
43          .appendFraction(ChronoField.NANO_OF_SECOND, 0, 9, true)
44          .optionalEnd()
45          .toFormatter();
46    }
47  
48    private DateFormats() {
49      // disable construction
50    }
51  }
