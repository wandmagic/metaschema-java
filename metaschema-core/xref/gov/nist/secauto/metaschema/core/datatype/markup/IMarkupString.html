7   
8   import com.vladsch.flexmark.formatter.Formatter;
9   import com.vladsch.flexmark.util.ast.Document;
10  import com.vladsch.flexmark.util.ast.Node;
11  
12  import gov.nist.secauto.metaschema.core.datatype.ICustomJavaDataType;
13  import gov.nist.secauto.metaschema.core.datatype.markup.flexmark.FlexmarkFactory;
14  import gov.nist.secauto.metaschema.core.datatype.markup.flexmark.InsertAnchorExtension.InsertAnchorNode;
15  
16  import org.codehaus.stax2.XMLStreamWriter2;
17  import org.codehaus.stax2.evt.XMLEventFactory2;
18  
19  import java.io.IOException;
20  import java.util.List;
21  import java.util.function.Predicate;
22  import java.util.stream.Stream;
23  
24  import javax.xml.stream.XMLEventWriter;
25  import javax.xml.stream.XMLStreamException;
26  
27  import edu.umd.cs.findbugs.annotations.NonNull;
28  
29  public interface IMarkupString<TYPE extends IMarkupString<TYPE>>
30      extends ICustomJavaDataType<TYPE> {
31    /**
32     * Get the underlying Flexmark factory supporting markup serialization.
33     *
34     * @return the factory
35     */
36    @NonNull
37    FlexmarkFactory getFlexmarkFactory();
38  
39    /**
40     * Get the top-level Flexmark document node for the markup.
41     *
42     * @return the node
43     */
44    @NonNull
45    Document getDocument();
46  
47    /**
48     * Determine if the markup has no contents.
49     *
50     * @return {@code true} if the markup has no contents or {@code false} otherwise
51     */
52    boolean isEmpty();
53  
54    // /**
55    // * Write HTML content to the provided {@code xmlStreamWriter} using the
56    // provided {@code
57    // namespace}.
58    // *
59    // * @param writer
60    // * the writer
61    // * @param namespace
62    // * the XML namespace for the HTML
63    // * @throws XMLStreamException
64    // * if an error occurred while writing
65    // */
66    // void writeHtml(@NonNull XMLStreamWriter2 writer, @NonNull String namespace)
67    // throws
68    // XMLStreamException;
69  
70    /**
71     * Get the HyperText Markup Language (HTML) representation of this markup
72     * string.
73     *
74     * @return the HTML
75     */
76    @NonNull
77    String toHtml();
78  
79    /**
80     * Get the Extensible HyperText Markup Language (XHTML) representation of this
81     * markup string.
82     *
83     * @param namespace
84     *          the XML namespace to use for XHTML elements
85     *
86     * @return the XHTML
87     * @throws XMLStreamException
88     *           if an error occurred while establishing or writing to the
89     *           underlying XML stream
90     * @throws IOException
91     *           if an error occurred while generating the XHTML data
92     */
93    @NonNull
94    String toXHtml(@NonNull String namespace) throws XMLStreamException, IOException;
95  
96    /**
97     * Get the Commonmark Markdown representation of this markup string.
98     *
99     * @return the Markdown
100    */
101   @NonNull
102   String toMarkdown();
103 
104   /**
105    * Get a Markdown representation of this markup string, which will be created by
106    * the provided formatter.
107    *
108    * @param formatter
109    *          the specific Markdown formatter to use in producing the Markdown
110    *
111    * @return the Markdown
112    */
113   @NonNull
114   String toMarkdown(@NonNull Formatter formatter);
115 
116   /**
117    * Retrieve all nodes contained within this markup text as a stream.
118    *
119    * @return a depth first stream
120    */
121   @NonNull
122   Stream<Node> getNodesAsStream();
123 
124   /**
125    * Get markup inserts used as place holders within the string.
126    *
127    * @return a list of insets or an empty list if no inserts are present
128    */
129   @NonNull
130   default List<InsertAnchorNode> getInserts() {
131     return getInserts(insert -> true);
132   }
133 
134   /**
135    * Retrieve all insert statements that are contained within this markup text
136    * that match the provided filter.
137    *
138    * @param filter
139    *          a filter used to identify matching insert statements
140    * @return the matching insert statements
141    */
142   @NonNull
143   List<InsertAnchorNode> getInserts(
144       @NonNull Predicate<InsertAnchorNode> filter);
145 
146   /**
147    * Determine if the Markup consists of block elements.
148    *
149    * @return {@code true} if the markup consists of block elements, or
150    *         {@code false} otherwise
151    */
152   boolean isBlock();
153 
154   /**
155    * Write the Extensible HyperText Markup Language (XHTML) representation of this
156    * markup string to the provided stream writer.
157    *
158    * @param namespace
159    *          the XML namespace to use for XHTML elements
160    * @param streamWriter
161    *          the XML stream to write to
162    * @throws XMLStreamException
163    *           if an error occurred while establishing or writing to the XML
164    *           stream
165    */
166   void writeXHtml(
167       @NonNull String namespace,
168       @NonNull XMLStreamWriter2 streamWriter) throws XMLStreamException;
169 
170   /**
171    * Write the Extensible HyperText Markup Language (XHTML) representation of this
172    * markup string to the provided stream writer using the provided XML event
173    * factory.
174    *
175    * @param namespace
176    *          the XML namespace to use for XHTML elements
177    * @param eventFactory
178    *          the XML event factory used to generate XML events to write
179    * @param eventWriter
180    *          the XML event stream to write to
181    * @throws XMLStreamException
182    *           if an error occurred while establishing or writing to the XML
183    *           stream
184    */
185   void writeXHtml(
186       @NonNull String namespace,
187       @NonNull XMLEventFactory2 eventFactory,
188       @NonNull XMLEventWriter eventWriter) throws XMLStreamException;
189 }
