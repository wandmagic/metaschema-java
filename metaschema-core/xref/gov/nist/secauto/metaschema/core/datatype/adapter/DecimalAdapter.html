6   package gov.nist.secauto.metaschema.core.datatype.adapter;
7   
8   import com.fasterxml.jackson.core.JsonGenerator;
9   import com.fasterxml.jackson.databind.jsonFormatVisitors.JsonFormatTypes;
10  
11  import gov.nist.secauto.metaschema.core.datatype.AbstractDataTypeAdapter;
12  import gov.nist.secauto.metaschema.core.metapath.MetapathConstants;
13  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IAnyAtomicItem;
14  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IBooleanItem;
15  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IDecimalItem;
16  import gov.nist.secauto.metaschema.core.metapath.item.atomic.INumericItem;
17  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
18  
19  import java.io.IOException;
20  import java.math.BigDecimal;
21  import java.math.MathContext;
22  import java.util.List;
23  
24  import javax.xml.namespace.QName;
25  
26  import edu.umd.cs.findbugs.annotations.NonNull;
27  
28  public class DecimalAdapter
29      extends AbstractDataTypeAdapter<BigDecimal, IDecimalItem> {
30    public static final MathContext MATH_CONTEXT = MathContext.DECIMAL64;
31    @NonNull
32    private static final BigDecimal DECIMAL_BOOLEAN_TRUE = new BigDecimal("1.0");
33    @NonNull
34    private static final BigDecimal DECIMAL_BOOLEAN_FALSE = new BigDecimal("0.0");
35    @NonNull
36    private static final List<QName> NAMES = ObjectUtils.notNull(
37        List.of(new QName(MetapathConstants.NS_METAPATH.toASCIIString(), "decimal")));
38  
39    DecimalAdapter() {
40      super(BigDecimal.class);
41    }
42  
43    @Override
44    public List<QName> getNames() {
45      return NAMES;
46    }
47  
48    @Override
49    public JsonFormatTypes getJsonRawType() {
50      return JsonFormatTypes.NUMBER;
51    }
52  
53    @Override
54    public BigDecimal parse(String value) {
55      return new BigDecimal(value, MATH_CONTEXT);
56    }
57  
58    @Override
59    public void writeJsonValue(Object value, JsonGenerator generator) throws IOException {
60      try {
61        generator.writeNumber((BigDecimal) value);
62      } catch (ClassCastException ex) {
63        throw new IOException(ex);
64      }
65    }
66  
67    @Override
68    public BigDecimal copy(Object obj) {
69      // a BigDecimal is immutable
70      return (BigDecimal) obj;
71    }
72  
73    @Override
74    public Class<IDecimalItem> getItemClass() {
75      return IDecimalItem.class;
76    }
77  
78    @Override
79    public IDecimalItem newItem(Object value) {
80      BigDecimal item = toValue(value);
81      return IDecimalItem.valueOf(item);
82    }
83  
84    @Override
85    protected IDecimalItem castInternal(@NonNull IAnyAtomicItem item) {
86      IDecimalItem retval;
87      if (item instanceof INumericItem) {
88        retval = newItem(((INumericItem) item).asDecimal());
89      } else if (item instanceof IBooleanItem) {
90        boolean value = ((IBooleanItem) item).toBoolean();
91        retval = newItem(value ? DECIMAL_BOOLEAN_TRUE : DECIMAL_BOOLEAN_FALSE);
92      } else {
93        retval = super.castInternal(item);
94      }
95      return retval;
96    }
97  }
