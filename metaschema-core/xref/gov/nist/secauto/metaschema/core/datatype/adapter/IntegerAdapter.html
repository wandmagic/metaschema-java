6   package gov.nist.secauto.metaschema.core.datatype.adapter;
7   
8   import gov.nist.secauto.metaschema.core.metapath.MetapathConstants;
9   import gov.nist.secauto.metaschema.core.metapath.item.atomic.IAnyAtomicItem;
10  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IBooleanItem;
11  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IIntegerItem;
12  import gov.nist.secauto.metaschema.core.metapath.item.atomic.INumericItem;
13  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
14  
15  import java.math.BigInteger;
16  import java.util.List;
17  
18  import javax.xml.namespace.QName;
19  
20  import edu.umd.cs.findbugs.annotations.NonNull;
21  
22  public class IntegerAdapter
23      extends AbstractIntegerAdapter<IIntegerItem> {
24    @NonNull
25    private static final List<QName> NAMES = ObjectUtils.notNull(
26        List.of(new QName(MetapathConstants.NS_METAPATH.toASCIIString(), "integer")));
27  
28    IntegerAdapter() {
29      // avoid general construction
30    }
31  
32    @Override
33    public List<QName> getNames() {
34      return NAMES;
35    }
36  
37    @Override
38    public @NonNull Class<IIntegerItem> getItemClass() {
39      return IIntegerItem.class;
40    }
41  
42    @Override
43    public IIntegerItem newItem(Object value) {
44      BigInteger item = toValue(value);
45      return IIntegerItem.valueOf(item);
46    }
47  
48    @Override
49    protected IIntegerItem castInternal(@NonNull IAnyAtomicItem item) {
50      IIntegerItem retval;
51      if (item instanceof INumericItem) {
52        retval = newItem(((INumericItem) item).asInteger());
53      } else if (item instanceof IBooleanItem) {
54        boolean value = ((IBooleanItem) item).toBoolean();
55        retval = newItem(ObjectUtils.notNull(value ? BigInteger.ONE : BigInteger.ZERO));
56      } else {
57        retval = super.castInternal(item);
58      }
59      return retval;
60    }
61  }
