6   package gov.nist.secauto.metaschema.core.datatype.adapter;
7   
8   import com.fasterxml.jackson.databind.jsonFormatVisitors.JsonFormatTypes;
9   
10  import gov.nist.secauto.metaschema.core.datatype.AbstractDataTypeAdapter;
11  import gov.nist.secauto.metaschema.core.metapath.MetapathConstants;
12  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IUuidItem;
13  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
14  
15  import java.util.List;
16  import java.util.UUID;
17  import java.util.regex.Pattern;
18  
19  import javax.xml.namespace.QName;
20  
21  import edu.umd.cs.findbugs.annotations.NonNull;
22  
23  public class UuidAdapter
24      extends AbstractDataTypeAdapter<UUID, IUuidItem> {
25    @NonNull
26    private static final List<QName> NAMES = ObjectUtils.notNull(
27        List.of(new QName(MetapathConstants.NS_METAPATH.toASCIIString(), "uuid")));
28    public static final Pattern UUID_PATTERN
29        = Pattern.compile("^[0-9A-Fa-f]{8}-[0-9A-Fa-f]{4}-[45][0-9A-Fa-f]{3}-[89ABab][0-9A-Fa-f]{3}-[0-9A-Fa-f]{12}$");
30  
31    UuidAdapter() {
32      super(UUID.class);
33    }
34  
35    @Override
36    public List<QName> getNames() {
37      return NAMES;
38    }
39  
40    @Override
41    public JsonFormatTypes getJsonRawType() {
42      return JsonFormatTypes.STRING;
43    }
44  
45    @SuppressWarnings("null")
46    @Override
47    public UUID parse(String value) {
48      return UUID.fromString(value);
49    }
50  
51    @Override
52    public UUID copy(Object obj) {
53      // a UUID is immutable
54      return (UUID) obj;
55    }
56  
57    @Override
58    public Class<IUuidItem> getItemClass() {
59      return IUuidItem.class;
60    }
61  
62    @Override
63    public IUuidItem newItem(Object value) {
64      UUID item = toValue(value);
65      return IUuidItem.valueOf(item);
66    }
67  }
