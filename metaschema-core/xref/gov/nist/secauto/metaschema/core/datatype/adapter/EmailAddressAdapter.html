6   package gov.nist.secauto.metaschema.core.datatype.adapter;
7   
8   import gov.nist.secauto.metaschema.core.metapath.MetapathConstants;
9   import gov.nist.secauto.metaschema.core.metapath.item.atomic.IEmailAddressItem;
10  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
11  
12  import java.util.List;
13  
14  import javax.xml.namespace.QName;
15  
16  import edu.umd.cs.findbugs.annotations.NonNull;
17  
18  public class EmailAddressAdapter
19      extends AbstractStringAdapter<IEmailAddressItem> {
20    @NonNull
21    private static final List<QName> NAMES = ObjectUtils.notNull(
22        List.of(
23            new QName(MetapathConstants.NS_METAPATH.toASCIIString(), "email-address"),
24            // for backwards compatibility with original type name
25            new QName(MetapathConstants.NS_METAPATH.toASCIIString(), "email")));
26  
27    EmailAddressAdapter() {
28      // avoid general construction
29    }
30  
31    @Override
32    public List<QName> getNames() {
33      return NAMES;
34    }
35  
36    @Override
37    public @NonNull Class<IEmailAddressItem> getItemClass() {
38      return IEmailAddressItem.class;
39    }
40  
41    @Override
42    public IEmailAddressItem newItem(Object value) {
43      String item = asString(value);
44      return IEmailAddressItem.valueOf(item);
45    }
46  }
