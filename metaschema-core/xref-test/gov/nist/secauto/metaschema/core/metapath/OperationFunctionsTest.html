6   package gov.nist.secauto.metaschema.core.metapath;
7   
8   import static gov.nist.secauto.metaschema.core.metapath.TestUtils.decimal;
9   import static gov.nist.secauto.metaschema.core.metapath.TestUtils.integer;
10  import static org.junit.jupiter.api.Assertions.assertEquals;
11  
12  import gov.nist.secauto.metaschema.core.metapath.function.OperationFunctions;
13  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IIntegerItem;
14  import gov.nist.secauto.metaschema.core.metapath.item.atomic.INumericItem;
15  
16  import org.junit.jupiter.params.ParameterizedTest;
17  import org.junit.jupiter.params.provider.Arguments;
18  import org.junit.jupiter.params.provider.MethodSource;
19  
20  import java.util.stream.Stream;
21  
22  import edu.umd.cs.findbugs.annotations.NonNull;
23  
24  class OperationFunctionsTest {
25    private static Stream<Arguments> provideValuesForIntegerDivide() {
26      return Stream.of(
27          Arguments.of(integer(10), integer(3), integer(3)),
28          Arguments.of(integer(3), integer(-2), integer(-1)),
29          Arguments.of(integer(-3), integer(2), integer(-1)),
30          Arguments.of(integer(-3), integer(-2), integer(1)),
31          Arguments.of(decimal("9.0"), integer(3), integer(3)),
32          Arguments.of(decimal("-3.5"), integer(3), integer(-1)),
33          Arguments.of(decimal("3.0"), integer(4), integer(0)),
34          Arguments.of(decimal("3.1E1"), integer(6), integer(5)),
35          Arguments.of(decimal("3.1E1"), integer(7), integer(4)));
36    }
37  
38    @ParameterizedTest
39    @MethodSource("provideValuesForIntegerDivide")
40    void testIntegerDivide(@NonNull INumericItem dividend, @NonNull INumericItem divisor,
41        @NonNull IIntegerItem expected) {
42      INumericItem result = OperationFunctions.opNumericIntegerDivide(dividend, divisor);
43      assertEquals(expected, result);
44    }
45  
46    private static Stream<Arguments> provideValuesForMod() {
47      return Stream.of(
48          Arguments.of(integer(5), integer(3), decimal("2")),
49          Arguments.of(integer(6), integer(-2), decimal("0")),
50          Arguments.of(decimal("4.5"), decimal("1.2"), decimal("0.9")),
51          Arguments.of(integer(123), integer(6), decimal("3")));
52    }
53  
54    @ParameterizedTest
55    @MethodSource("provideValuesForMod")
56    void test(@NonNull INumericItem dividend, @NonNull INumericItem divisor, @NonNull INumericItem expected) {
57      INumericItem result = OperationFunctions.opNumericMod(dividend, divisor);
58      assertEquals(expected, result);
59    }

61  }
