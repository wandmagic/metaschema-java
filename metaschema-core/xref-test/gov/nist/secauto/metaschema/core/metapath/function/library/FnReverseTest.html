23  class FnReverseTest
24      extends ExpressionTestBase {
25    private static Stream<Arguments> provideValues() { // NOPMD - false positive
26      return Stream.of(
27          Arguments.of(
28              ISequence.of(string("c"), string("b"), string("a")),
29              "reverse(('a', 'b', 'c'))"),
30          Arguments.of(
31              ISequence.of(string("hello")),
32              "reverse(('hello'))"),
33          Arguments.of(
34              ISequence.empty(),
35              "reverse(())"));
36      // TODO: Add tests when Metapath array syntax supported.
37      // Arguments.of(
38      // ISequence.of(array([1, 2, 3])),
39      // "reverse(([1,2,3]))"),
40      // Arguments.of(
41      // ISequence.of(array([1, 2, 3]), array([1, 2, 3])),
42      // "reverse(([1,2,3],[4,5,6]))");
43    }
44  
45    @ParameterizedTest
46    @MethodSource("provideValues")
47    void test(@NonNull ISequence<?> expected, @NonNull String metapath) {
48      assertEquals(expected, MetapathExpression.compile(metapath).evaluateAs(null, MetapathExpression.ResultType.SEQUENCE,
49          newDynamicContext()));
50    }
