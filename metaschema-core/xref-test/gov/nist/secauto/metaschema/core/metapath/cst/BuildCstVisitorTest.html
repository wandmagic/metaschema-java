107   private static IExpression parseExpression(@NonNull String path) {
108 
109     Metapath10Lexer lexer = new Metapath10Lexer(CharStreams.fromString(path));
110     CommonTokenStream tokens = new CommonTokenStream(lexer);
111     Metapath10 parser = new Metapath10(tokens);
112     parser.addErrorListener(new FailingErrorListener());
113 
114     ParseTree tree = parser.expr();
115     // ParseTreePrinter cstPrinter = new ParseTreePrinter(System.out);
116     // cstPrinter.print(tree, Arrays.asList(parser.getRuleNames()));
117 
118     return new BuildCSTVisitor(StaticContext.instance()).visit(tree);
119   }

122   void testAbbreviatedParentAxis() {

124     // compile expression
