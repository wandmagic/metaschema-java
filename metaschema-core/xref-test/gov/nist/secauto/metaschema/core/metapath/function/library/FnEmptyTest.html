22  /*
23   *
24   * The expression fn:empty((1,2,3)[10]) returns true().
25   * The expression fn:empty(fn:remove(("hello", "world"), 1)) returns false().
26   * The expression fn:empty([]) returns false().
27   * The expression fn:empty(map{}) returns false().
28   * The expression fn:empty("") returns false().
29   *
30   * Source: https://www.w3.org/TR/xpath-functions-31/#func-empty
31   *
32   */
33  
34  class FnEmptyTest
35      extends FunctionTestBase {
36    private static Stream<Arguments> provideValues() { // NOPMD - false positive
37      return Stream.of(
38          Arguments.of(
39              ISequence.of(bool(false)),
40              "empty((1,2,3))"),
41          Arguments.of(
42              ISequence.of(bool(false)),
43              "empty(('hello','world'))"),
44          Arguments.of(
45              ISequence.of(bool(false)),
46              "empty((''))"));
47      // TODO: Add when we support map constructor syntax.
48      // Arguments.of(
49      // ISequence.of(bool(true)),
50      // "empty(map{})"));
51    }
52  
53    @ParameterizedTest
54    @MethodSource("provideValues")
55    void test(@NonNull ISequence<?> expected, @NonNull String metapath) {
56      assertEquals(expected, MetapathExpression.compile(metapath).evaluateAs(null, MetapathExpression.ResultType.SEQUENCE,
57          newDynamicContext()));
58    }
59  }
