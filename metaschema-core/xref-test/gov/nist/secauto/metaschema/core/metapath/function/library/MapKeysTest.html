6   package gov.nist.secauto.metaschema.core.metapath.function.library;
7   
8   import static gov.nist.secauto.metaschema.core.metapath.TestUtils.integer;
9   import static org.junit.jupiter.api.Assertions.assertEquals;
10  
11  import gov.nist.secauto.metaschema.core.metapath.ExpressionTestBase;
12  import gov.nist.secauto.metaschema.core.metapath.ISequence;
13  import gov.nist.secauto.metaschema.core.metapath.MetapathExpression;
14  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IAnyAtomicItem;
15  
16  import org.junit.jupiter.api.Test;
17  
18  import java.util.HashSet;
19  import java.util.Set;
20  
21  class MapKeysTest
22      extends ExpressionTestBase {
23  
24    @Test
25    void test() {
26      ISequence<IAnyAtomicItem> result = MetapathExpression.compile("map:keys(map{1:\"yes\", 2:\"no\"})")
27          .evaluateAs(null, MetapathExpression.ResultType.SEQUENCE);
28      assert result != null;
29  
30      // use a set to allow any ordering of the keys, since we have no control over
31      // their order
32      Set<IAnyAtomicItem> keys = new HashSet<>(result);
33  
34      assertEquals(Set.of(integer(1), integer(2)), keys);
35    }
36  }
