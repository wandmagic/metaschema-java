33  class FnSumTest
34      extends FunctionTestBase {
35  
36    private static Stream<Arguments> provideValuesOneArg() {
37      IYearMonthDurationItem yearMonth1 = yearMonthDuration("P20Y");
38      IYearMonthDurationItem yearMonth2 = yearMonthDuration("P10M");
39      IDayTimeDurationItem dayTime1 = dayTimeDuration("P1DT12H");
40      IDayTimeDurationItem dayTime2 = dayTimeDuration("P2D");
41  
42      return Stream.of(
43          Arguments.of(integer(0), new IAnyAtomicItem[] {}),
44          Arguments.of(integer(12), new IAnyAtomicItem[] { integer(3), integer(4), integer(5) }),
45          Arguments.of(null, new IAnyAtomicItem[] { integer(3), integer(4), string("test") }),
46          Arguments.of(dayTimeDuration("P3DT12H"), new IAnyAtomicItem[] { dayTime1, dayTime2 }),
47          Arguments.of(null, new IAnyAtomicItem[] { dayTime1, dayTime2, integer(1) }),
48          Arguments.of(yearMonthDuration("P20Y10M"), new IAnyAtomicItem[] { yearMonth1, yearMonth2 }),
49          Arguments.of(null, new IAnyAtomicItem[] { yearMonth1, yearMonth2, integer(1) }));
50    }
51  
52    @ParameterizedTest
53    @MethodSource("provideValuesOneArg")
54    void testAvg(@Nullable IAnyAtomicItem expected, @NonNull IAnyAtomicItem... values) {
55      try {
56        assertFunctionResult(
57            FnSum.SIGNATURE_ONE_ARG,
58            ISequence.of(expected),
59            List.of(ISequence.of(values)));
60      } catch (MetapathException ex) {
61        if (expected == null) {
62          assertAll(
63              () -> assertInstanceOf(InvalidArgumentFunctionException.class, ex.getCause()));
64        } else {
65          throw ex;
66        }
67      }
68    }
69  
70    private static Stream<Arguments> provideValuesTwoArg() {
71      IYearMonthDurationItem yearMonth1 = yearMonthDuration("P20Y");
72      IYearMonthDurationItem yearMonth2 = yearMonthDuration("P10M");
73  
74      return Stream.of(
75  
76          Arguments.of(integer(0), integer(0), new IAnyAtomicItem[] {}),
77          Arguments.of(
78              yearMonthDuration("P20Y10M"),
79              string("ein Augenblick"),
80              new IAnyAtomicItem[] { yearMonth1, yearMonth2 }));
81    }
82  
83    @ParameterizedTest
84    @MethodSource("provideValuesTwoArg")
85    void testAvgWithZero(@Nullable IAnyAtomicItem expected, @Nullable IAnyAtomicItem zero,
86        @NonNull IAnyAtomicItem... values) {
87      try {
88        assertFunctionResult(
89            FnSum.SIGNATURE_TWO_ARG,
90            ISequence.of(expected),
91            List.of(ISequence.of(values), ISequence.of(zero)));
92      } catch (MetapathException ex) {
93        if (expected == null) {
94          assertAll(
95              () -> assertInstanceOf(InvalidArgumentFunctionException.class, ex.getCause()));
96        } else {
97          throw ex;
98        }
99      }
100   }
101 
102   @Test
103   void testAvgNoOpTwoArg() {
104     assertFunctionResult(
105         FnSum.SIGNATURE_TWO_ARG,
106         ISequence.empty(),
107         List.of(ISequence.empty(), ISequence.empty()));
108   }
