20  import java.util.stream.Stream;
21  
22  import edu.umd.cs.findbugs.annotations.NonNull;
23  
24  class ArrayGetTest
25      extends ExpressionTestBase {
26    private static Stream<Arguments> provideValues() { // NOPMD - false positive
27      return Stream.of(
28          Arguments.of(
29              string("b"),
30              "[\"a\", \"b\", \"c\"] => array:get(2)"),
31          Arguments.of(
32              array(string("b"), string("c")),
33              "[\"a\", [\"b\", \"c\"]] => array:get(2)"));
34    }
35  
36    @ParameterizedTest
37    @MethodSource("provideValues")
38    void testExpression(@NonNull IItem expected, @NonNull String metapath) {
39  
40      IItem result = MetapathExpression.compile(metapath)
41          .evaluateAs(null, MetapathExpression.ResultType.NODE, newDynamicContext());
42      assertEquals(expected, result);
43    }
44  }
