6   package gov.nist.secauto.metaschema.core.metapath.function.library;
7   
8   import static gov.nist.secauto.metaschema.core.metapath.TestUtils.decimal;
9   import static gov.nist.secauto.metaschema.core.metapath.TestUtils.integer;
10  
11  import gov.nist.secauto.metaschema.core.metapath.ISequence;
12  import gov.nist.secauto.metaschema.core.metapath.item.atomic.INumericItem;
13  
14  import org.junit.jupiter.api.Test;
15  import org.junit.jupiter.params.ParameterizedTest;
16  import org.junit.jupiter.params.provider.Arguments;
17  import org.junit.jupiter.params.provider.MethodSource;
18  
19  import java.util.List;
20  import java.util.stream.Stream;
21  
22  import edu.umd.cs.findbugs.annotations.NonNull;
23  
24  class FnAbsTest
25      extends FunctionTestBase {
26  
27    private static Stream<Arguments> provideValues() {
28      return Stream.of(
29          Arguments.of(decimal("10.5"), decimal("10.5")),
30          Arguments.of(decimal("10.5"), decimal("-10.5")),
31          Arguments.of(integer(5), integer(5)),
32          Arguments.of(integer(5), integer(-5)));
33    }
34  
35    @ParameterizedTest
36    @MethodSource("provideValues")
37    void testRound(@NonNull INumericItem expected, @NonNull INumericItem actual) {
38      assertFunctionResult(
39          FnAbs.SIGNATURE,
40          ISequence.of(expected),
41          List.of(ISequence.of(actual)));
42    }
43  
44    @Test
45    void testNoOp() {
46      assertFunctionResult(
47          FnAbs.SIGNATURE,
48          ISequence.empty(),
49          List.of(ISequence.empty()));
50    }
51  }
