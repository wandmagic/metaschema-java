25  class MapGetTest
26      extends ExpressionTestBase {
27  
28    private static Stream<Arguments> provideValues() { // NOPMD - false positive
29      return Stream.of(
30          Arguments.of(
31              sequence(string("Donnerstag")),
32              "let $week :=  " + LookupTest.WEEKDAYS_GERMAN + " return map:get($week, 4)"),
33          Arguments.of(
34              sequence(),
35              "let $week :=  " + LookupTest.WEEKDAYS_GERMAN + " return map:get($week, 9)"),
36          Arguments.of(
37              sequence(),
38              "map:get(map:entry(7,()), 7)"));
39    }
40  
41    @ParameterizedTest
42    @MethodSource("provideValues")
43    void testExpression(@NonNull ISequence<?> expected, @NonNull String metapath) {
44  
45      ISequence<?> result = MetapathExpression.compile(metapath)
46          .evaluateAs(null, MetapathExpression.ResultType.SEQUENCE, newDynamicContext());
47      assertEquals(expected, result);
48    }
49  }
