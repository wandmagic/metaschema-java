28  class MapRemoveTest
29      extends ExpressionTestBase {
30  
31    private static Stream<Arguments> provideValues() { // NOPMD - false positive
32      return Stream.of(
33          Arguments.of(
34              map(
35                  entry(integer(0), string("Sonntag")),
36                  entry(integer(1), string("Montag")),
37                  entry(integer(2), string("Dienstag")),
38                  entry(integer(3), string("Mittwoch")),
39                  entry(integer(5), string("Freitag")),
40                  entry(integer(6), string("Samstag"))),
41              "let $week :=  " + LookupTest.WEEKDAYS_GERMAN + " return map:remove($week, 4)"),
42          Arguments.of(
43              map(
44                  entry(integer(0), string("Sonntag")),
45                  entry(integer(1), string("Montag")),
46                  entry(integer(2), string("Dienstag")),
47                  entry(integer(3), string("Mittwoch")),
48                  entry(integer(4), string("Donnerstag")),
49                  entry(integer(5), string("Freitag")),
50                  entry(integer(6), string("Samstag"))),
51              "let $week :=  " + LookupTest.WEEKDAYS_GERMAN + " return map:remove($week, 23)"),
52          Arguments.of(
53              map(
54                  entry(integer(1), string("Montag")),
55                  entry(integer(2), string("Dienstag")),
56                  entry(integer(3), string("Mittwoch")),
57                  entry(integer(4), string("Donnerstag")),
58                  entry(integer(5), string("Freitag"))),
59              "let $week :=  " + LookupTest.WEEKDAYS_GERMAN + " return map:remove($week, (0, 6 to 7))"),
60          Arguments.of(
61              map(
62                  entry(integer(0), string("Sonntag")),
63                  entry(integer(1), string("Montag")),
64                  entry(integer(2), string("Dienstag")),
65                  entry(integer(3), string("Mittwoch")),
66                  entry(integer(4), string("Donnerstag")),
67                  entry(integer(5), string("Freitag")),
68                  entry(integer(6), string("Samstag"))),
69              "let $week :=  " + LookupTest.WEEKDAYS_GERMAN + " return map:remove($week, ())"));
70    }
71  
72    @ParameterizedTest
73    @MethodSource("provideValues")
74    void testExpression(@NonNull IMapItem<?> expected, @NonNull String metapath) {
75  
76      IItem result = MetapathExpression.compile(metapath)
77          .evaluateAs(null, MetapathExpression.ResultType.NODE, newDynamicContext());
78      assertEquals(expected, result);
79    }
80  }
