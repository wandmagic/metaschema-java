9   import static gov.nist.secauto.metaschema.core.metapath.TestUtils.integer;
10  import static gov.nist.secauto.metaschema.core.metapath.TestUtils.string;
11  import static gov.nist.secauto.metaschema.core.metapath.TestUtils.uri;
12  import static org.junit.jupiter.api.Assertions.assertAll;
13  import static org.junit.jupiter.api.Assertions.assertInstanceOf;
14  
15  import gov.nist.secauto.metaschema.core.metapath.ISequence;

17  import gov.nist.secauto.metaschema.core.metapath.function.InvalidArgumentFunctionException;
18  import gov.nist.secauto.metaschema.core.metapath.item.atomic.IAnyAtomicItem;
19  
20  import org.junit.jupiter.api.Test;
21  import org.junit.jupiter.params.ParameterizedTest;
22  import org.junit.jupiter.params.provider.Arguments;
23  import org.junit.jupiter.params.provider.MethodSource;
24  
25  import java.util.List;
26  import java.util.stream.Stream;
27  
28  import edu.umd.cs.findbugs.annotations.NonNull;
29  import edu.umd.cs.findbugs.annotations.Nullable;
30  
31  class FnMinMaxTest
32      extends FunctionTestBase {
33  
34    private static Stream<Arguments> provideValuesMin() {
35      return Stream.of(
36          Arguments.of(null, new IAnyAtomicItem[] { integer(3), string("text") }),
37          Arguments.of(string("same"), new IAnyAtomicItem[] { string("same"), uri("uri/") }),
38          Arguments.of(integer(3), new IAnyAtomicItem[] { integer(3) }),
39          Arguments.of(decimal("5"), new IAnyAtomicItem[] { decimal("5") }),
40          Arguments.of(string("same"), new IAnyAtomicItem[] { string("same") }),
41          Arguments.of(integer(3), new IAnyAtomicItem[] { integer(3), integer(4), integer(5) }),
42          Arguments.of(integer(5), new IAnyAtomicItem[] { integer(5), decimal("5"), decimal("10") }),
43          Arguments.of(decimal("5"), new IAnyAtomicItem[] { decimal("5"), integer(5), decimal("10") }));
44    }
45  
46    @ParameterizedTest
47    @MethodSource("provideValuesMin")
48    void testMin(@Nullable IAnyAtomicItem expected, @NonNull IAnyAtomicItem... values) {
49      try {
50        assertFunctionResult(
51            FnMinMax.SIGNATURE_MIN,
52            ISequence.of(expected),
53            List.of(ISequence.of(values)));
54      } catch (MetapathException ex) {
55        if (expected == null) {
56          assertAll(
57              () -> assertInstanceOf(InvalidArgumentFunctionException.class, ex.getCause()));
58        } else {
59          throw ex;
60        }
61      }
62    }
63  
64    @Test
65    void testMinNoOp() {
66      assertFunctionResult(
67          FnMinMax.SIGNATURE_MIN,
68          ISequence.empty(),
69          List.of(ISequence.empty()));
70    }
71  
72    private static Stream<Arguments> provideValuesMax() {
73      return Stream.of(
74          Arguments.of(null, new IAnyAtomicItem[] { integer(3), string("text") }),
75          Arguments.of(string("uri/"), new IAnyAtomicItem[] { string("same"), uri("uri/") }),
76          Arguments.of(integer(3), new IAnyAtomicItem[] { integer(3) }),
77          Arguments.of(decimal("5"), new IAnyAtomicItem[] { decimal("5") }),
78          Arguments.of(string("same"), new IAnyAtomicItem[] { string("same") }),
79          Arguments.of(integer(5), new IAnyAtomicItem[] { integer(3), integer(4), integer(5) }),
80          Arguments.of(integer(10), new IAnyAtomicItem[] { integer(5), decimal("5"), integer(10), decimal("10") }),
81          Arguments.of(decimal("10"), new IAnyAtomicItem[] { decimal("5"), integer(5), decimal("10"), integer(10) }));
82    }
83  
84    @ParameterizedTest
85    @MethodSource("provideValuesMax")
86    void testMax(@Nullable IAnyAtomicItem expected, @NonNull IAnyAtomicItem... values) {
87      try {
88        assertFunctionResult(
89            FnMinMax.SIGNATURE_MAX,
90            ISequence.of(expected),
91            List.of(ISequence.of(values)));
92      } catch (MetapathException ex) {
93        if (expected == null) {
94          assertAll(
95              () -> assertInstanceOf(InvalidArgumentFunctionException.class, ex.getCause()));
96        } else {
97          throw ex;
98        }
99      }
100   }
101 
102   @Test
103   void testMaxNoOp() {
104     assertFunctionResult(
105         FnMinMax.SIGNATURE_MAX,
106         ISequence.empty(),
107         List.of(ISequence.empty()));
108   }
109 }
