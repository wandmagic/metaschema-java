33  
34  class FnAvgTest
35      extends FunctionTestBase {
36  
37    private static Stream<Arguments> provideValuesForAvg() {
38      IYearMonthDurationItem yearMonth1 = yearMonthDuration("P20Y");
39      IYearMonthDurationItem yearMonth2 = yearMonthDuration("P10M");
40      IDayTimeDurationItem dayTime1 = dayTimeDuration("P1DT12H");
41      IDayTimeDurationItem dayTime2 = dayTimeDuration("P2D");
42  
43      return Stream.of(
44          Arguments.of(decimal("4"), new IAnyAtomicItem[] { integer(3), integer(4), integer(5) }),
45          Arguments.of(null, new IAnyAtomicItem[] { integer(3), integer(4), string("test") }),
46          Arguments.of(dayTimeDuration("P1DT18H"), new IAnyAtomicItem[] { dayTime1, dayTime2 }),
47          Arguments.of(null, new IAnyAtomicItem[] { dayTime1, dayTime2, integer(1) }),
48          Arguments.of(yearMonthDuration("P10Y5M"), new IAnyAtomicItem[] { yearMonth1, yearMonth2 }),
49          Arguments.of(null, new IAnyAtomicItem[] { yearMonth1, yearMonth2, integer(1) }));
50    }
51  
52    @ParameterizedTest
53    @MethodSource("provideValuesForAvg")
54    void testAvg(@Nullable IAnyAtomicItem expected, @NonNull IAnyAtomicItem... values) {
55      try {
56        assertFunctionResult(
57            FnAvg.SIGNATURE,
58            ISequence.of(expected),
59            List.of(ISequence.of(values)));
60      } catch (MetapathException ex) {
61        if (expected == null) {
62          assertAll(
63              () -> assertInstanceOf(InvalidArgumentFunctionException.class, ex.getCause()));
64        } else {
65          throw ex;
66        }
67      }
68    }
69  
70    @Test
71    void testAvgNoOp() {
72      assertFunctionResult(
73          FnAvg.SIGNATURE,
74          ISequence.empty(),
75          List.of(ISequence.empty()));
76    }
77  }
