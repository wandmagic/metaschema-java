6   package gov.nist.secauto.metaschema.core.datatype.adapter;
7   
8   import static org.junit.jupiter.api.Assertions.assertNotNull;
9   
10  import org.junit.jupiter.params.ParameterizedTest;
11  import org.junit.jupiter.params.provider.ValueSource;
12  
13  import java.time.ZonedDateTime;
14  
15  import edu.umd.cs.findbugs.annotations.NonNull;
16  
17  class DateTimeWithTZAdapterTest {
18    private static final DateTimeWithTZAdapter ADAPTER = MetaschemaDataTypeProvider.DATE_TIME_WITH_TZ;
19  
20    @ParameterizedTest
21    @ValueSource(strings = {
22        "2020-12-20T14:47:48.623-05:00",
23        "2019-09-28T23:20:50.52Z",
24        "2019-09-28T23:20:50.0Z",
25        "2019-12-02T16:39:57-08:00",
26        "2019-12-02T16:39:57.100-08:00",
27        "2019-12-31T23:59:59Z"
28    })
29    void testParse(@NonNull String value) {
30      ZonedDateTime obj = ADAPTER.parse(value);
31      assertNotNull(obj, "not null");
32    }
33  
34  }
