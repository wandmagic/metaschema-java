8   import static org.junit.jupiter.api.Assertions.assertEquals;

11  
12  import java.util.regex.Pattern;
13  
14  class ReplacementScannerTest {
15  
16    @Test
17    void test() {
18      String string = "Some { first} text \\{ to {second\\}} scan";
19  
20      Pattern pattern = Pattern.compile("(?<!\\\\)(\\{\\s*((?:(?:\\\\})|[^}])*)\\s*\\})");
21      assert pattern != null;
22      String result = ReplacementScanner.replaceTokens(string, pattern, match -> match.group(2)).toString();
23      assertEquals("Some first text \\{ to second\\} scan", result, "replacement doesn't match expectation");
24    }
