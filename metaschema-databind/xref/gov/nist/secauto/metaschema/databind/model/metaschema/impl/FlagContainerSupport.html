28  @SuppressWarnings("PMD.OnlyOneReturn")
29  public final class FlagContainerSupport {
30    @SuppressFBWarnings(value = "CT_CONSTRUCTOR_THROW", justification = "Use of final fields")
31    @NonNull
32    public static IContainerFlagSupport<IFlagInstance> newFlagContainer(
33        @Nullable List<Object> flags,
34        @NonNull IBoundInstanceModelGroupedAssembly bindingInstance,
35        @NonNull IBindingDefinitionModel parent,
36        @Nullable String jsonKeyName) {
37      if (flags == null || flags.isEmpty()) {
38        return IContainerFlagSupport.empty();
39      }
40  
41      // create temporary collections to store the child binding objects
42      IFlagContainerBuilder<IFlagInstance> builder = jsonKeyName == null
43          ? IContainerFlagSupport.builder()
44          : IContainerFlagSupport.builder(parent.getContainingModule().toFlagQName(jsonKeyName));
45  
46      // create counter to track child positions
47      int flagReferencePosition = 0;
48      int flagInlineDefinitionPosition = 0;
49  
50      IBoundInstanceModelChoiceGroup instance = ObjectUtils.requireNonNull(
51          bindingInstance.getDefinition().getChoiceGroupInstanceByName("flags"));
52      for (Object obj : flags) {
53        IBoundInstanceModelGroupedAssembly objInstance
54            = (IBoundInstanceModelGroupedAssembly) instance.getItemInstance(obj);
55  
56        IFlagInstance flag;
57        if (obj instanceof InlineDefineFlag) {
58          flag = new InstanceFlagInline(
59              (InlineDefineFlag) obj,
60              objInstance,
61              flagInlineDefinitionPosition++,
62              parent);
63        } else if (obj instanceof FlagReference) {
64          flag = newFlagInstance(
65              (FlagReference) obj,
66              objInstance,
67              flagReferencePosition++,
68              parent);
69        } else {
70          throw new UnsupportedOperationException(String.format("Unknown flag instance class: %s", obj.getClass()));
71        }
72  
73        builder.flag(flag);
74      }
75  
76      return builder.build();
77    }
78  
79    @NonNull
80    private static IFlagInstance newFlagInstance(
81        @NonNull FlagReference obj,
82        @NonNull IBoundInstanceModelGroupedAssembly objInstance,
83        int position,
84        @NonNull IBindingDefinitionModel parent) {
85      IModule module = parent.getContainingModule();
86  
87      QName qname = module.toFlagQName(ObjectUtils.requireNonNull(obj.getRef()));
88      IFlagDefinition definition = module.getScopedFlagDefinitionByName(qname);
89      if (definition == null) {
90        throw new IllegalStateException(
91            String.format("Unable to resolve flag reference '%s' in definition '%s' in module '%s'",
92                qname,
93                parent.getName(),
94                module.getShortName()));
95      }
96      return new InstanceFlagReference(obj, objInstance, position, definition, parent);
97    }
98  
99    private FlagContainerSupport() {
100     // disable construction
101   }

103 }
