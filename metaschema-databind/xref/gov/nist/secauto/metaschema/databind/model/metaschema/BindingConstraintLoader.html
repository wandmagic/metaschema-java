62  /**
63   * Provides methods to load a constraint set expressed in XML.
64   * <p>
65   * Loaded constraint instances are cached to avoid the need to load them for
66   * every use. Any constraint set imported is also loaded and cached
67   * automatically.
68   */
69  public class BindingConstraintLoader
70      extends AbstractLoader<List<IConstraintSet>>
71      implements IConstraintLoader {
72  
73    @NonNull
74    private final IBoundLoader loader;
75  
76    public BindingConstraintLoader(@NonNull IBindingContext bindingContext) {
77      // ensure the bindings are registered
78      bindingContext.registerBindingMatcher(MetaschemaMetaConstraints.class);
79      bindingContext.registerBindingMatcher(MetaschemaModuleConstraints.class);
80  
81      this.loader = bindingContext.newBoundLoader();
82      this.loader.enableFeature(DeserializationFeature.DESERIALIZE_VALIDATE_CONSTRAINTS);
83    }
84  
85    @Override
86    protected List<IConstraintSet> parseResource(@NonNull URI resource, @NonNull Deque<URI> visitedResources)
87        throws IOException {
88  
89      Object constraintsDocument = loader.load(resource);
90  
91      StaticContext.Builder builder = StaticContext.builder()
92          .baseUri(resource);
93  
94      builder.useWildcardWhenNamespaceNotDefaulted(true);
95  
96      List<IConstraintSet> retval;
97      if (constraintsDocument instanceof MetaschemaModuleConstraints) {
98        MetaschemaModuleConstraints obj = (MetaschemaModuleConstraints) constraintsDocument;
99  
100       // now check if this constraint set imports other constraint sets
101       List<MetaschemaModuleConstraints.Import> imports = CollectionUtil.listOrEmpty(obj.getImports());
102 
103       @NonNull Set<IConstraintSet> importedConstraints;
104       if (imports.isEmpty()) {
105         importedConstraints = CollectionUtil.emptySet();
106       } else {
107         try {
108           importedConstraints = new LinkedHashSet<>();
109           for (MetaschemaModuleConstraints.Import imported : imports) {
110             URI importedResource = imported.getHref();
111             importedResource = ObjectUtils.notNull(resource.resolve(importedResource));
112             importedConstraints.addAll(loadInternal(importedResource, visitedResources));
113           }
114         } catch (MetaschemaException ex) {
115           throw new IOException(ex);
116         }
117       }
118 
119       CollectionUtil.listOrEmpty(obj.getNamespaceBindings()).stream()
120           .forEach(binding -> builder.namespace(
121               ObjectUtils.notNull(binding.getPrefix()),
122               ObjectUtils.notNull(binding.getUri())));
123       ISource source = ISource.externalSource(builder.build());
124 
125       // now create this constraint set
126       retval = CollectionUtil.singletonList(new DefaultConstraintSet(
127           resource,
128           parseScopedConstraints(obj, source),
129           new LinkedHashSet<>(importedConstraints)));
130     } else if (constraintsDocument instanceof MetaschemaMetaConstraints) {
131       MetaschemaMetaConstraints obj = (MetaschemaMetaConstraints) constraintsDocument;
132 
133       // now check if this constraint set imports other constraint sets
134       List<MetaschemaMetaConstraints.Import> imports = CollectionUtil.listOrEmpty(obj.getImports());
135 
136       retval = new LinkedList<>();
137       if (!imports.isEmpty()) {
138         try {
139           for (MetaschemaMetaConstraints.Import imported : imports) {
140             URI importedResource = imported.getHref();
141             importedResource = ObjectUtils.notNull(resource.resolve(importedResource));
142             retval.addAll(loadInternal(importedResource, visitedResources));
143           }
144         } catch (MetaschemaException ex) {
145           throw new IOException(ex);
146         }
147       }
148 
149       CollectionUtil.listOrEmpty(obj.getNamespaceBindings()).stream()
150           .forEach(binding -> builder.namespace(
151               ObjectUtils.notNull(binding.getPrefix()),
152               ObjectUtils.notNull(binding.getUri())));
153 
154       ISource source = ISource.externalSource(builder.build());
155 
156       List<ITargetedConstraints> targetedConstraints = CollectionUtil.listOrEmpty(obj.getContexts()).stream()
157           .flatMap(context -> parseContext(ObjectUtils.notNull(context), null, source)
158               .getTargetedConstraints().stream())
159           .collect(Collectors.toList());
160       retval.add(new MetaConstraintSet(targetedConstraints));
161 
162       retval = CollectionUtil.unmodifiableList(retval);
163     } else {
164       throw new UnsupportedOperationException(String.format("Unsupported constraint content '%s'.", resource));
165     }
166     return retval;
167   }

169   /**
170    * Parse individual constraint definitions from the provided XMLBeans object.
171    *
172    * @param obj
173    *          the XMLBeans object
174    * @param source
175    *          the source of the constraint content
176    * @return the scoped constraint definitions
177    */
178   @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops") // intentional
179   @NonNull
180   protected List<IScopedContraints> parseScopedConstraints(
181       @NonNull MetaschemaModuleConstraints obj,
182       @NonNull ISource source) {
183     List<IScopedContraints> scopedConstraints = new LinkedList<>();
184 
185     for (MetaschemaModuleConstraints.Scope scope : CollectionUtil.listOrEmpty(obj.getScopes())) {
186       assert scope != null;
187 
188       List<ITargetedConstraints> targetedConstraints = new LinkedList<>();
189       try {
190         for (IValueConstraintsBase constraintsObj : CollectionUtil.listOrEmpty(scope.getConstraints())) {
191           if (constraintsObj instanceof MetaschemaModuleConstraints.Scope.Assembly) {
192             targetedConstraints.add(handleScopedAssembly(
193                 (MetaschemaModuleConstraints.Scope.Assembly) constraintsObj,
194                 source));
195           } else if (constraintsObj instanceof MetaschemaModuleConstraints.Scope.Field) {
196             targetedConstraints.add(handleScopedField(
197                 (MetaschemaModuleConstraints.Scope.Field) constraintsObj,
198                 source));
199           } else if (constraintsObj instanceof MetaschemaModuleConstraints.Scope.Flag) {
200             targetedConstraints.add(handleScopedFlag(
201                 (MetaschemaModuleConstraints.Scope.Flag) constraintsObj,
202                 source));
203           }
204         }
205       } catch (MetapathException | XmlValueNotSupportedException ex) {
206         if (ex.getCause() instanceof MetapathException) {
207           throw new MetapathException(
208               String.format("Unable to compile a Metapath in '%s'. %s",
209                   source.getSource(),
210                   ex.getLocalizedMessage()),
211               ex);
212         }
213         throw ex;
214       }
215 
216       URI namespace = ObjectUtils.requireNonNull(scope.getMetaschemaNamespace());
217       String shortName = ObjectUtils.requireNonNull(scope.getMetaschemaShortName());
218 
219       scopedConstraints.add(new DefaultScopedContraints(
220           namespace,
221           shortName,
222           CollectionUtil.unmodifiableList(targetedConstraints)));
223     }
224     return CollectionUtil.unmodifiableList(scopedConstraints);
225   }

227   private static AssemblyTargetedConstraints handleScopedAssembly(
228       @NonNull MetaschemaModuleConstraints.Scope.Assembly obj,
229       @NonNull ISource source) {
230     IModelConstrained constraints = new AssemblyConstraintSet();
231     ConstraintBindingSupport.parse(constraints, obj, source);
232     return new AssemblyTargetedConstraints(
233         ObjectUtils.requireNonNull(obj.getTarget()),
234         constraints);
235   }

238       @NonNull MetaschemaModuleConstraints.Scope.Field obj,
239       @NonNull ISource source) {
240     IValueConstrained constraints = new ValueConstraintSet();
241     ConstraintBindingSupport.parse(constraints, obj, source);
242 
243     return new FieldTargetedConstraints(
244         ObjectUtils.requireNonNull(obj.getTarget()),
245         constraints);
246   }

259   private Context parseContext(
260       @NonNull MetapathContext contextObj,
261       @Nullable Context parent,
262       @NonNull ISource source) {
263 
264     List<String> metapaths;
265     if (parent == null) {
266       metapaths = CollectionUtil.listOrEmpty(contextObj.getMetapaths()).stream()
267           .map(MetaschemaMetapath::getTarget)
268           .collect(Collectors.toList());
269     } else {
270       List<String> parentMetapaths = parent.getMetapaths().stream()
271           .collect(Collectors.toList());
272       metapaths = CollectionUtil.listOrEmpty(contextObj.getMetapaths()).stream()
273           .map(MetaschemaMetapath::getTarget)
274           .flatMap(childPath -> parentMetapaths.stream()
275               .map(parentPath -> parentPath + '/' + childPath))
276           .collect(Collectors.toList());
277     }
278 
279     AssemblyConstraints contextConstraints = contextObj.getConstraints();
280     IModelConstrained constraints = new AssemblyConstraintSet();
281     if (contextConstraints != null) {
282       ConstraintBindingSupport.parse(constraints, contextConstraints, source);
283     }
284     Context context = new Context(metapaths, constraints);
285 
286     List<Context> childContexts = CollectionUtil.listOrEmpty(contextObj.getContexts()).stream()
287         .map(childObj -> parseContext(ObjectUtils.notNull(childObj), context, source))
288         .collect(Collectors.toList());
289 
290     context.addAll(childContexts);
291 
292     return context;
293   }

295   private static class Context {
296     @NonNull
297     private final List<String> metapaths;
298     @NonNull
299     private final List<Context> childContexts = new LinkedList<>();
300     @NonNull
301     private final Lazy<List<ITargetedConstraints>> targetedConstraints;
302 
303     public Context(
304         @NonNull List<String> metapaths,
305         @NonNull IModelConstrained constraints) {
306       this.metapaths = metapaths;
307       this.targetedConstraints = ObjectUtils.notNull(Lazy.lazy(() -> {
308 
309         Stream<ITargetedConstraints> paths = getMetapaths().stream()
310             .map(metapath -> new MetaTargetedContraints(ObjectUtils.notNull(metapath), constraints));
311         Stream<ITargetedConstraints> childPaths = childContexts.stream()
312             .flatMap(child -> child.getTargetedConstraints().stream());
313 
314         return Stream.concat(paths, childPaths)
315             .collect(Collectors.toUnmodifiableList());
316       }));
317     }
318 
319     @NonNull
320     public List<ITargetedConstraints> getTargetedConstraints() {
321       return ObjectUtils.notNull(targetedConstraints.get());
322     }
323 
324     public void addAll(@NonNull Collection<Context> childContexts) {
325       childContexts.addAll(childContexts);
326     }
327 
328     @NonNull
329     public List<String> getMetapaths() {
330       return metapaths;
331     }
332   }

334   private static class MetaTargetedContraints
335       extends AbstractTargetedConstraints<IModelConstrained>
336       implements IFeatureModelConstrained {
337 
338     protected MetaTargetedContraints(
339         @NonNull String target,
340         @NonNull IModelConstrained constraints) {
341       super(target, constraints);
342     }
343 
344     /**
345      * Apply the constraints to the provided {@code definition}.
346      * <p>
347      * This will be called when a definition is found that matches the target
348      * expression.
349      *
350      * @param definition
351      *          the definition to apply the constraints to.
352      */
353     protected void applyTo(@NonNull IDefinition definition) {
354       getLetExpressions().values().forEach(definition::addLetExpression);
355       getAllowedValuesConstraints().forEach(definition::addConstraint);
356       getMatchesConstraints().forEach(definition::addConstraint);
357       getIndexHasKeyConstraints().forEach(definition::addConstraint);
358       getExpectConstraints().forEach(definition::addConstraint);
359     }
360 
361     protected void applyTo(@NonNull IAssemblyDefinition definition) {
362       applyTo((IDefinition) definition);
363       getIndexConstraints().forEach(definition::addConstraint);
364       getUniqueConstraints().forEach(definition::addConstraint);
365       getHasCardinalityConstraints().forEach(definition::addConstraint);
366     }
367 
368     @Override
369     public void target(IFlagDefinition definition) {
370       applyTo(definition);
371     }
372 
373     @Override
374     public void target(IFieldDefinition definition) {
375       applyTo(definition);
376     }
377 
378     @Override
379     public void target(IAssemblyDefinition definition) {
380       applyTo(definition);
381     }
382   }
