23  import java.util.Map;
24  import java.util.Set;

26  import edu.umd.cs.findbugs.annotations.NonNull;
27  import edu.umd.cs.findbugs.annotations.Nullable;
28  import nl.talsmasoftware.lazy4j.Lazy;
29  
30  public class InstanceFlagReference
31      extends AbstractFlagInstance<
32          IBindingDefinitionModel,
33          IFlagDefinition, IFlagInstance>
34      implements IFeatureValueless, IBindingInstance {
35    @NonNull
36    private final FlagReference binding;
37    @NonNull
38    private final IFlagDefinition definition;
39    @NonNull
40    private final Map<IAttributable.Key, Set<String>> properties;
41    @Nullable
42    private final Object defaultValue;
43    @NonNull
44    private final Lazy<IAssemblyNodeItem> boundNodeItem;
45  
46    public InstanceFlagReference(
47        @NonNull FlagReference binding,
48        @NonNull IBoundInstanceModelGroupedAssembly bindingInstance,
49        int position,
50        @NonNull IFlagDefinition definition,
51        @NonNull IBindingDefinitionModel parent) {
52      super(parent);
53      this.binding = binding;
54      this.definition = definition;
55      this.properties = ModelSupport.parseProperties(ObjectUtils.requireNonNull(binding.getProps()));
56      this.defaultValue = ModelSupport.defaultValue(binding.getDefault(), definition.getJavaTypeAdapter());
57      this.boundNodeItem = ObjectUtils.notNull(
58          Lazy.lazy(() -> (IAssemblyNodeItem) ObjectUtils.notNull(parent.getSourceNodeItem())
59              .getModelItemsByName(bindingInstance.getXmlQName())
60              .get(position)));
61    }
62  
63    @NonNull
64    protected FlagReference getBinding() {
65      return binding;
66    }
67  
68    @Override
69    public IBindingMetaschemaModule getContainingModule() {
70      return getContainingDefinition().getContainingModule();
71    }
72  
73    @Override
74    public IAssemblyNodeItem getSourceNodeItem() {
75      return boundNodeItem.get();
76    }
77  
78    @Override
79    public IFlagDefinition getDefinition() {
80      return definition;
81    }
82  
83    @Override
84    public Map<IAttributable.Key, Set<String>> getProperties() {
85      return properties;
86    }
87  
88    @Override
89    public Object getDefaultValue() {
90      return defaultValue;
91    }
92  
93    @Override
94    public String getName() {
95      return getDefinition().getName();
96    }
97  
98    @Override
99    public String getFormalName() {
100     return getBinding().getFormalName();
101   }
102 
103   @Override
104   public MarkupLine getDescription() {
105     return getBinding().getDescription();
106   }
107 
108   @Override
109   public Integer getIndex() {
110     return ModelSupport.index(getBinding().getIndex());
111   }
112 
113   @Override
114   public String getUseName() {
115     return ModelSupport.useName(getBinding().getUseName());
116   }
117 
118   @Override
119   public Integer getUseIndex() {
120     return ModelSupport.useIndex(getBinding().getUseName());
121   }
122 
123   @Override
124   public MarkupMultiline getRemarks() {
125     return ModelSupport.remarks(getBinding().getRemarks());
126   }
127 
128   @Override
129   public boolean isRequired() {
130     return ModelSupport.yesOrNo(getBinding().getRequired());
131   }
132 }
