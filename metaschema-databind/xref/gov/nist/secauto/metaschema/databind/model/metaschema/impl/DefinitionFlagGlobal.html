34  public class DefinitionFlagGlobal
35      extends AbstractGlobalFlagDefinition<IBindingMetaschemaModule, IFlagInstance> {
36    @NonNull
37    private final METASCHEMA.DefineFlag binding;
38    @NonNull
39    private final Map<IAttributable.Key, Set<String>> properties;
40    @NonNull
41    private final IDataTypeAdapter<?> javaTypeAdapter;
42    @Nullable
43    private final Object defaultValue;
44    @NonNull
45    private final Lazy<IValueConstrained> valueConstraints;
46    @NonNull
47    private final Lazy<IAssemblyNodeItem> boundNodeItem;
48  
49    /**
50     * Construct a new Metaschema module flag definition binding using an underlying
51     * bound class that describes the flag.
52     *
53     * @param binding
54     *          the underlying bound class
55     * @param bindingInstance
56     *          the assembly instance for the underlying bound class
57     * @param position
58     *          the zero-based position of this instance relative to its bound
59     *          siblings
60     * @param module
61     *          the Metaschema module containing this binding
62     */
63    public DefinitionFlagGlobal(
64        @NonNull METASCHEMA.DefineFlag binding,
65        @NonNull IBoundInstanceModelGroupedAssembly bindingInstance,
66        int position,
67        @NonNull IBindingMetaschemaModule module) {
68      super(module);
69      this.binding = binding;
70      this.properties = ModelSupport.parseProperties(ObjectUtils.requireNonNull(binding.getProps()));
71      this.javaTypeAdapter = ModelSupport.dataType(binding.getAsType());
72      this.defaultValue = ModelSupport.defaultValue(binding.getDefault(), this.javaTypeAdapter);
73      this.valueConstraints = ObjectUtils.notNull(Lazy.lazy(() -> {
74        IValueConstrained retval = new ValueConstraintSet();
75        FlagConstraints constraints = binding.getConstraint();
76        if (constraints != null) {
77          ConstraintBindingSupport.parse(retval, constraints, ISource.modelSource(module));
78        }
79        return retval;
80      }));
81      this.boundNodeItem = ObjectUtils.notNull(Lazy.lazy(() -> ObjectUtils.requireNonNull(ModelSupport.toNodeItem(
82          module,
83          bindingInstance.getXmlQName(),
84          position))));
85    }
86  
87    @NonNull
88    protected METASCHEMA.DefineFlag getBinding() {
89      return binding;
90    }
91  
92    @Override
93    public IValueConstrained getConstraintSupport() {
94      return ObjectUtils.notNull(valueConstraints.get());
95    }
96  
97    @Override
98    public Map<IAttributable.Key, Set<String>> getProperties() {
99      return properties;
100   }
101 
102   @Override
103   public IDataTypeAdapter<?> getJavaTypeAdapter() {
104     return javaTypeAdapter;
105   }
106 
107   @Override
108   public Object getDefaultValue() {
109     return defaultValue;
110   }
111 
112   @Override
113   public String getName() {
114     return ObjectUtils.notNull(getBinding().getName());
115   }
116 
117   @Override
118   public String getFormalName() {
119     return getBinding().getFormalName();
120   }
121 
122   @Override
123   public MarkupLine getDescription() {
124     return getBinding().getDescription();
125   }
126 
127   @Override
128   public ModuleScopeEnum getModuleScope() {
129     return ModelSupport.moduleScope(ObjectUtils.requireNonNull(getBinding().getScope()));
130   }
131 
132   @Override
133   public Integer getIndex() {
134     return ModelSupport.index(getBinding().getIndex());
135   }
136 
137   @Override
138   public String getUseName() {
139     return ModelSupport.useName(getBinding().getUseName());
140   }
141 
142   @Override
143   public Integer getUseIndex() {
144     return ModelSupport.useIndex(getBinding().getUseName());
145   }
146 
147   @Override
148   public MarkupMultiline getRemarks() {
149     return ModelSupport.remarks(getBinding().getRemarks());
150   }
151 
152   @NonNull
153   public INodeItem getNodeItem() {
154     return boundNodeItem.get();
155   }
156 }
