6   package gov.nist.secauto.metaschema.databind.model;
7   
8   import gov.nist.secauto.metaschema.core.model.IFieldDefinition;
9   
10  import edu.umd.cs.findbugs.annotations.Nullable;
11  
12  /**
13   * Represents a field definition bound to Java data.
14   * <p>
15   * Classes and interfaces implementing this interface can be:
16   * <ul>
17   * <li>"scalar", meaning they have only a scalar field value, or</li>
18   * <li>"complex", meaning they allow flags and are bound to a Java class.
19   * </ul>
20   *
21   * @param <ITEM>
22   *          the Java type for associated bound objects
23   */
24  public interface IBoundDefinitionModelField<ITEM>
25      extends IFieldDefinition, IBoundDefinitionModel<ITEM> {
26  
27    @Override
28    default IBoundInstanceModelField<ITEM> getInlineInstance() {
29      // never inline
30      return null;
31    }
32  
33    @Override
34    @Nullable
35    IBoundInstanceFlag getJsonValueKeyFlagInstance();
36  }
