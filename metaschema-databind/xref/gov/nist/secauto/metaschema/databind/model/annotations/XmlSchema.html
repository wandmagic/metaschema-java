6   package gov.nist.secauto.metaschema.databind.model.annotations;
7   
8   import static java.lang.annotation.ElementType.PACKAGE;
9   import static java.lang.annotation.RetentionPolicy.RUNTIME;
10  
11  import java.lang.annotation.Retention;
12  import java.lang.annotation.Target;
13  
14  /**
15   * This annotation provides package-level Module information.
16   */
17  @Retention(RUNTIME)
18  @Target(PACKAGE)
19  public @interface XmlSchema {
20    /**
21     * The default value of a schema location, which indicates that no schema will
22     * be associated.
23     * <p>
24     * The value "##none" was chosen because ## is not a valid sequence in
25     * xs:anyURI.
26     */
27    String NO_LOCATION = ModelUtil.NO_STRING_VALUE;
28  
29    /**
30     * Defines the XML namespace URI and prefix to use for this model. If a prefix
31     * is not provided, the XML prefix will be auto-generated.
32     *
33     * @return an array of namespace definitions
34     */
35    XmlNs[] xmlns() default {};
36  
37    /**
38     * Name of the XML namespace.
39     * <p>
40     * If the value is "##none", then there is no prefix defined.
41     *
42     * @return a namespace string in the form of a URI
43     */
44    String namespace() default ModelUtil.NO_STRING_VALUE;
45  
46    /**
47     * The location of the associated XML schema.
48     *
49     * @return a location string in the form of a URI
50     */
51    String xmlSchemaLocation() default NO_LOCATION;
52  
53    /**
54     * The location of the associated JSON schema.
55     *
56     * @return a location string in the form of a URI
57     */
58    String jsonSchemaLocation() default NO_LOCATION;
59  
60    /**
61     * Get the default XML element form.
62     *
63     * @return the XML element form
64     */
65    XmlNsForm xmlElementFormDefault() default XmlNsForm.UNSET;
66  
67    /**
68     * Get the default XML attribute form.
69     *
70     * @return the XML attribute form
71     */
72    XmlNsForm xmlAttributeFormDefault() default XmlNsForm.UNSET;
73  }
