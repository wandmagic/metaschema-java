6   package gov.nist.secauto.metaschema.databind.codegen.typeinfo;
7   
8   import com.squareup.javapoet.FieldSpec;
9   import com.squareup.javapoet.MethodSpec;
10  import com.squareup.javapoet.ParameterSpec;
11  import com.squareup.javapoet.TypeName;
12  import com.squareup.javapoet.TypeSpec;
13  
14  import gov.nist.secauto.metaschema.core.model.IModelDefinition;
15  import gov.nist.secauto.metaschema.core.util.CollectionUtil;
16  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
17  import gov.nist.secauto.metaschema.databind.codegen.typeinfo.def.IDefinitionTypeInfo;
18  
19  import java.util.Set;
20  
21  import javax.lang.model.element.Modifier;
22  
23  import edu.umd.cs.findbugs.annotations.NonNull;
24  
25  public abstract class AbstractPropertyTypeInfo<PARENT extends IDefinitionTypeInfo>
26      extends AbstractTypeInfo<PARENT>
27      implements IPropertyTypeInfo {
28  
29    /**
30     * Construct a new type information for a Java property.
31     *
32     * @param parentDefinition
33     *          the definition containing the data this property is based on
34     */
35    protected AbstractPropertyTypeInfo(@NonNull PARENT parentDefinition) {
36      super(parentDefinition);
37    }
38  
39    @Override
40    public Set<IModelDefinition> build(@NonNull TypeSpec.Builder builder) {
41  
42      TypeName javaFieldType = getJavaFieldType();
43      FieldSpec.Builder field = FieldSpec.builder(javaFieldType, getJavaFieldName())
44          .addModifiers(Modifier.PRIVATE);
45      assert field != null;
46  
47      final Set<IModelDefinition> retval = buildField(builder, field);
48  
49      FieldSpec valueField = ObjectUtils.notNull(field.build());
50      builder.addField(valueField);
51  
52      buildExtraMethods(builder, valueField);
53      return retval;
54    }
55  
56    protected void buildExtraMethods(
57        @NonNull TypeSpec.Builder typeBuilder,
58        @NonNull FieldSpec fieldBuilder) {
59  
60      TypeName javaFieldType = getJavaFieldType();
61      String propertyName = getPropertyName();
62      {
63        MethodSpec.Builder method = MethodSpec.methodBuilder("get" + propertyName)
64            .returns(javaFieldType)
65            .addModifiers(Modifier.PUBLIC);
66        assert method != null;
67        method.addStatement("return $N", fieldBuilder);
68        typeBuilder.addMethod(method.build());
69      }
70  
71      {
72        ParameterSpec valueParam = ParameterSpec.builder(javaFieldType, "value").build();
73        MethodSpec.Builder method = MethodSpec.methodBuilder("set" + propertyName)
74            .addModifiers(Modifier.PUBLIC)
75            .addParameter(valueParam);
76        assert method != null;
77        method.addStatement("$N = $N", fieldBuilder, valueParam);
78        typeBuilder.addMethod(method.build());
79      }
80    }
81  
82    /**
83     * Generate the Java field associated with this property.
84     *
85     * @param typeBuilder
86     *          the class builder the field is on
87     * @param fieldBuilder
88     *          the field builder
89     * @return the set of definitions used by this field
90     */
91    protected Set<IModelDefinition> buildField(
92        @NonNull TypeSpec.Builder typeBuilder,
93        @NonNull FieldSpec.Builder fieldBuilder) {
94      buildFieldJavadoc(fieldBuilder);
95      return CollectionUtil.emptySet();
96    }
97  }
