6   package gov.nist.secauto.metaschema.databind.codegen.typeinfo;
7   
8   import com.squareup.javapoet.AnnotationSpec;
9   import com.squareup.javapoet.FieldSpec;
10  import com.squareup.javapoet.TypeSpec;
11  
12  import gov.nist.secauto.metaschema.core.model.IModelDefinition;
13  import gov.nist.secauto.metaschema.core.model.IModelInstanceAbsolute;
14  import gov.nist.secauto.metaschema.databind.codegen.typeinfo.def.IAssemblyDefinitionTypeInfo;
15  
16  import java.util.Set;
17  
18  import edu.umd.cs.findbugs.annotations.NonNull;
19  
20  public interface IModelInstanceTypeInfo extends IInstanceTypeInfo {
21  
22    @Override
23    @NonNull
24    IAssemblyDefinitionTypeInfo getParentTypeInfo();
25  
26    @Override
27    IModelInstanceAbsolute getInstance();
28  
29    /**
30     * Generate the Metaschema binding annotation for this instance.
31     * <p>
32     * This method uses a builder so child classes can add to the builder as needed.
33     *
34     * @param typeBuilder
35     *          the class builder the field is on
36     * @param fieldBuilder
37     *          the field builder the annotation is on
38     * @param annotation
39     *          if not {@code null} will seed the annotation to build
40     *
41     * @return a builder for the annotation
42     */
43    @NonNull
44    Set<IModelDefinition> buildBindingAnnotation(
45        @NonNull TypeSpec.Builder typeBuilder,
46        @NonNull FieldSpec.Builder fieldBuilder,
47        @NonNull AnnotationSpec.Builder annotation);
48  }
