6   package gov.nist.secauto.metaschema.databind.codegen.typeinfo;
7   
8   import gov.nist.secauto.metaschema.core.util.ObjectUtils;
9   import gov.nist.secauto.metaschema.databind.codegen.ClassUtils;
10  import gov.nist.secauto.metaschema.databind.codegen.typeinfo.def.IDefinitionTypeInfo;
11  
12  import edu.umd.cs.findbugs.annotations.NonNull;
13  
14  abstract class AbstractTypeInfo<PARENT extends IDefinitionTypeInfo> implements ITypeInfo {
15    @NonNull
16    private final PARENT parentDefinition;
17    private String propertyName;
18    private String fieldName;
19  
20    protected AbstractTypeInfo(@NonNull PARENT parentDefinition) {
21      this.parentDefinition = parentDefinition;
22    }
23  
24    @Override
25    @NonNull
26    public PARENT getParentTypeInfo() {
27      return parentDefinition;
28    }
29  
30    /**
31     * The property name of the instance, which must be unique within the class.
32     *
33     * @return the name
34     */
35    @Override
36    @NonNull
37    public String getPropertyName() {
38      synchronized (this) {
39        if (this.propertyName == null) {
40          String name = ClassUtils.toPropertyName(getBaseName());
41          IDefinitionTypeInfo parent = getParentTypeInfo();
42  
43          // first check if a property already exists with the same name
44          this.propertyName = parent.getTypeResolver().getPropertyName(parent, name);
45        }
46        return ObjectUtils.notNull(this.propertyName);
47      }
48    }
49  
50    /**
51     * Gets the name of the Java field for this property item.
52     *
53     * @return the Java field name
54     */
55    @Override
56    @NonNull
57    public final String getJavaFieldName() {
58      synchronized (this) {
59        if (this.fieldName == null) {
60          this.fieldName = "_" + ClassUtils.toVariableName(getPropertyName());
61        }
62        return ObjectUtils.notNull(this.fieldName);
63      }
64    }
65  }
