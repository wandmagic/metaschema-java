30  import edu.umd.cs.findbugs.annotations.NonNull;
31  
32  public class ChoiceGroupTypeInfoImpl
33      extends AbstractModelInstanceTypeInfo<IChoiceGroupInstance>
34      implements IChoiceGroupTypeInfo {
35  
36    public ChoiceGroupTypeInfoImpl(
37        @NonNull IChoiceGroupInstance instance,
38        @NonNull IAssemblyDefinitionTypeInfo parent) {
39      super(instance, parent);
40    }
41  
42    @Override
43    public TypeName getJavaItemType() {
44      return getParentTypeInfo().getTypeResolver().getClassName(getInstance());
45    }
46  
47    @Override
48    protected AnnotationSpec.Builder newBindingAnnotation() {
49      return ObjectUtils.notNull(AnnotationSpec.builder(BoundChoiceGroup.class));
50    }
51  
52    @SuppressWarnings("PMD.UseConcurrentHashMap")
53    @Override
54    public Set<IModelDefinition> buildBindingAnnotation(
55        TypeSpec.Builder typeBuilder,
56        FieldSpec.Builder fieldBuilder,
57        AnnotationSpec.Builder annotation) {
58      IChoiceGroupInstance choiceGroup = getInstance();
59  
60      String discriminator = choiceGroup.getJsonDiscriminatorProperty();
61      if (!IChoiceGroupInstance.DEFAULT_JSON_DISCRIMINATOR_PROPERTY_NAME.equals(discriminator)) {
62        annotation.addMember("discriminator", "$S", discriminator);
63      }
64  
65      int minOccurs = choiceGroup.getMinOccurs();
66      if (minOccurs != IGroupable.DEFAULT_GROUP_AS_MIN_OCCURS) {
67        annotation.addMember("minOccurs", "$L", minOccurs);
68      }
69  
70      int maxOccurs = choiceGroup.getMaxOccurs();
71      if (maxOccurs != IGroupable.DEFAULT_GROUP_AS_MAX_OCCURS) {
72        annotation.addMember("maxOccurs", "$L", maxOccurs);
73      }
74  
75      String jsonKeyName = choiceGroup.getJsonKeyFlagInstanceName();
76      if (jsonKeyName != null) {
77        annotation.addMember("jsonKey", "$S", jsonKeyName);
78      }
79  
80      Set<IModelDefinition> retval = new LinkedHashSet<>();
81  
82      IAssemblyDefinitionTypeInfo parentTypeInfo = getParentTypeInfo();
83      ITypeResolver typeResolver = parentTypeInfo.getTypeResolver();
84  
85      Map<ClassName, List<INamedModelInstanceGrouped>> referencedDefinitions = new LinkedHashMap<>();
86      Collection<? extends INamedModelInstanceGrouped> modelInstances = getInstance().getNamedModelInstances();
87      for (INamedModelInstanceGrouped modelInstance : modelInstances) {
88        ClassName className = typeResolver.getClassName(modelInstance.getDefinition());
89        List<INamedModelInstanceGrouped> instances = referencedDefinitions.get(className);
90        if (instances == null) {
91          instances = new LinkedList<>(); // NOPMD needed
92          referencedDefinitions.put(className, instances);
93        }
94        instances.add(modelInstance);
95      }
96  
97      for (INamedModelInstanceGrouped modelInstance : modelInstances) {
98        assert modelInstance != null;
99        IGroupedNamedModelInstanceTypeInfo instanceTypeInfo = typeResolver.getTypeInfo(modelInstance, this);
100 
101       ClassName className = typeResolver.getClassName(modelInstance.getDefinition());
102       retval.addAll(instanceTypeInfo.generateMemberAnnotation(
103           annotation,
104           typeBuilder,
105           referencedDefinitions.get(className).size() > 1));
106     }
107 
108     if (maxOccurs == -1 || maxOccurs > 1) {
109       // requires a group-as
110       annotation.addMember("groupAs", "$L", generateGroupAsAnnotation().build());
111     }
112     return retval;
113   }
114 }
