6   package gov.nist.secauto.metaschema.databind.codegen;
7   
8   import gov.nist.secauto.metaschema.core.model.IModule;
9   import gov.nist.secauto.metaschema.core.util.CollectionUtil;
10  import gov.nist.secauto.metaschema.databind.codegen.config.IBindingConfiguration;
11  
12  import org.apache.logging.log4j.LogManager;
13  import org.apache.logging.log4j.Logger;
14  
15  import java.io.IOException;
16  import java.nio.file.Path;
17  import java.util.Collection;
18  import java.util.Objects;
19  
20  import edu.umd.cs.findbugs.annotations.NonNull;
21  
22  /**
23   * Provides methods for generating Java classes based on a single or a
24   * collection of Metaschemas.
25   */
26  public final class JavaGenerator {
27    private static final Logger LOGGER = LogManager.getLogger(JavaGenerator.class);
28  
29    private JavaGenerator() {
30      // disable construction
31    }
32  
33    /**
34     * Generate Java sources for the provided Metaschema module.
35     *
36     * @param module
37     *          the Metaschema module to generate Java sources for
38     * @param targetDir
39     *          the directory to generate sources in
40     * @param bindingConfiguration
41     *          the binding customizations to use when generating the Java classes
42     * @return information about all the produced classes
43     * @throws IOException
44     *           if an error occurred while generating the class
45     */
46    public static IProduction generate(
47        @NonNull IModule module,
48        @NonNull Path targetDir,
49        @NonNull IBindingConfiguration bindingConfiguration) throws IOException {
50      return generate(CollectionUtil.singletonList(module), targetDir, bindingConfiguration);
51    }
52  
53    /**
54     * Generates Java classes for Module fields and flags.
55     *
56     * @param modules
57     *          the Metaschema modules to build classes for
58     * @param targetDirectory
59     *          the directory to generate classes in
60     * @param bindingConfiguration
61     *          binding customizations that can be used to set namespaces, class
62     *          names, and other aspects of generated classes
63     * @return information about all the produced classes
64     * @throws IOException
65     *           if a build error occurred while generating the class
66     */
67    @NonNull
68    public static IProduction generate(
69        @NonNull Collection<? extends IModule> modules,
70        @NonNull Path targetDirectory,
71        @NonNull IBindingConfiguration bindingConfiguration) throws IOException {
72      Objects.requireNonNull(modules, "metaschemas");
73      Objects.requireNonNull(targetDirectory, "generationTargetDirectory");
74      Objects.requireNonNull(bindingConfiguration, "bindingConfiguration");
75      if (LOGGER.isInfoEnabled()) {
76        LOGGER.info("Generating Java classes in: {}", targetDirectory);
77      }
78  
79      return IProduction.of(modules, bindingConfiguration, targetDirectory);
80    }
81  }
