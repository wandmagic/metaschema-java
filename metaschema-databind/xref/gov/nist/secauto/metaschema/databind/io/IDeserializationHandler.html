6   package gov.nist.secauto.metaschema.databind.io;
7   
8   /**
9    * Provides methods used during deserialization to perform additional actions
10   * before and after data is loaded into a bound object.
11   * <p>
12   * Methods with these method signatures will be called if defined on a bound
13   * object regardless of if this interface is implemented by the object.
14   */
15  public interface IDeserializationHandler {
16  
17    /**
18     * A method called just before the object data is read and added to the object.
19     *
20     * @param parent
21     *          the Java object containing this object
22     */
23    void beforeDeserialize(Object parent);
24  
25    /**
26     * A method called just after the object's data is read and added to the object.
27     *
28     * @param parent
29     *          the Java object containing this object
30     */
31    void afterDeserialize(Object parent);
32  
33  }
