6   package gov.nist.secauto.metaschema.databind.io;
7   
8   import edu.umd.cs.findbugs.annotations.NonNull;
9   
10  /**
11   * Provides objects used for parsing data associated with a specific format.
12   *
13   * @param <READER>
14   *          the format specific data reader
15   * @param <PROBLEM_HANDLER>
16   *          the format specific problem handler
17   */
18  public interface IParsingContext<READER, PROBLEM_HANDLER extends IProblemHandler> {
19    /**
20     * The parser used for reading data associated with the supported format.
21     *
22     * @return the parser
23     */
24    @NonNull
25    READER getReader();
26  
27    /**
28     * A handler that provides callbacks used to resolve parsing issues.
29     *
30     * @return the configured handler
31     */
32    @NonNull
33    PROBLEM_HANDLER getProblemHandler();
34  }
