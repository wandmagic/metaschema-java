6   package gov.nist.secauto.metaschema.databind.io;
7   
8   import gov.nist.secauto.metaschema.core.configuration.AbstractConfigurationFeature;
9   import gov.nist.secauto.metaschema.core.model.IAssemblyDefinition;
10  
11  import edu.umd.cs.findbugs.annotations.NonNull;
12  
13  @SuppressWarnings("PMD.DataClass") // not a data class
14  public final class DeserializationFeature<V>
15      extends AbstractConfigurationFeature<V> {
16    public static final int YAML_CODEPOINT_LIMIT_DEFAULT = Integer.MAX_VALUE - 1; // 2 GB
17    public static final int FORMAT_DETECTION_LOOKAHEAD = 32_768; // 2 GB
18  
19    /**
20     * If enabled, perform constraint validation on the deserialized bound objects.
21     */
22    @NonNull
23    public static final DeserializationFeature<Boolean> DESERIALIZE_VALIDATE_CONSTRAINTS
24        = new DeserializationFeature<>("validate", Boolean.class, false);
25  
26    /**
27     * If enabled, perform constraint validation on the deserialized bound objects.
28     */
29    @NonNull
30    public static final DeserializationFeature<Boolean> DESERIALIZE_XML_ALLOW_ENTITY_RESOLUTION
31        = new DeserializationFeature<>("allow-entity-resolution", Boolean.class, false);
32  
33    /**
34     * If enabled, process the next JSON node as a field, whose name must match the
35     * {@link IAssemblyDefinition#getRootJsonName()}. If not enabled, the next JSON
36     * node is expected to be an object containing the data of the
37     * {@link IAssemblyDefinition}.
38     */
39    @NonNull
40    public static final DeserializationFeature<Boolean> DESERIALIZE_JSON_ROOT_PROPERTY
41        = new DeserializationFeature<>("deserialize-root-property", Boolean.class, true);
42  
43    /**
44     * Determines the max YAML codepoints that can be read.
45     */
46    @NonNull
47    public static final DeserializationFeature<Integer> YAML_CODEPOINT_LIMIT
48        = new DeserializationFeature<>("yaml-codepoint-limit", Integer.class, YAML_CODEPOINT_LIMIT_DEFAULT);
49  
50    /**
51     * Determines how many bytes can be looked at to identify the format of a
52     * document.
53     */
54    @NonNull
55    public static final DeserializationFeature<Integer> FORMAT_DETECTION_LOOKAHEAD_LIMIT
56        = new DeserializationFeature<>("format-detection-lookahead-limit", Integer.class, FORMAT_DETECTION_LOOKAHEAD);
57  
58    private DeserializationFeature(
59        @NonNull String name,
60        @NonNull Class<V> valueClass,
61        @NonNull V defaultValue) {
62      super(name, valueClass, defaultValue);
63    }
64  }
