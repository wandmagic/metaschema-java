90    private static Object reflectMethod(Object obj, String name) throws NoSuchMethodException {
91      return ReflectionUtils.invokeMethod(obj.getClass().getMethod(name), obj);
92    }

94    @Test
95    void testFieldsWithFlagMetaschema()
96        throws MetaschemaException, IOException, ClassNotFoundException, BindingException {
97      runTests(
98          "fields_with_flags",
99          "gov.nist.csrc.ns.metaschema.testing.fields.with.flags.TopLevel",
100         ObjectUtils.notNull(generationDir),
101         obj -> {
102           try {
103             Assertions.assertEquals("test", reflectMethod(obj, "getId"));
104             Object field1 = ReflectionUtils.invokeMethod(obj.getClass().getMethod("getComplexField1"), obj);
105             Assertions.assertNotNull(field1);
106             Assertions.assertEquals("complex-field1", reflectMethod(field1, "getId"));
107             Assertions.assertEquals("test-string", reflectMethod(field1, "getValue"));
108 
109             @SuppressWarnings("unchecked") List<Object> field2s
110                 = (List<Object>) ReflectionUtils.invokeMethod(obj.getClass().getMethod("getComplexFields2"),
111                     obj);
112             Assertions.assertNotNull(field2s);
113             Assertions.assertEquals(1, field2s.size());
114             Object field2 = field2s.get(0);
115             Assertions.assertEquals("complex-field2-1", reflectMethod(field2, "getId"));
116             Assertions.assertEquals("test-string2", reflectMethod(field2, "getValue"));
117 
118             @SuppressWarnings("unchecked") List<Object> field3s
119                 = (List<Object>) ReflectionUtils.invokeMethod(obj.getClass().getMethod("getComplexFields3"),
120                     obj);
121             Assertions.assertEquals(2, field3s.size());
122             Assertions.assertAll("ComplexFields4 item", () -> {
123               Object item = field3s.get(0);
124               assertEquals("complex-field3-1", reflectMethod(item, "getId2"));
125               assertEquals("test-string3", reflectMethod(item, "getValue"));
126             });
127             Assertions.assertAll("ComplexFields4 item", () -> {
128               Object item = field3s.get(1);
129               assertEquals("complex-field3-2", reflectMethod(item, "getId2"));
130               assertEquals("test-string4", reflectMethod(item, "getValue"));
131             });
132 
133             Assertions.assertAll("ComplexFields4", () -> {
134               @SuppressWarnings("unchecked") Map<String, Object> collection
135                   = (Map<String, Object>) ReflectionUtils.invokeMethod(obj.getClass().getMethod("getComplexFields4"),
136                       obj);
137               Assertions.assertNotNull(collection);
138               Assertions.assertEquals(2, collection.size());
139               Set<Map.Entry<String, Object>> entries = collection.entrySet();
140               Iterator<Map.Entry<String, Object>> iter = entries.iterator();
141 
142               Assertions.assertAll("ComplexFields4 item", () -> {
143                 Map.Entry<String, Object> entry = iter.next();
144                 assertEquals("complex-field4-1", entry.getKey());
145                 assertEquals("complex-field4-1", reflectMethod(entry.getValue(), "getId2"));
146                 assertEquals("test-string5", reflectMethod(entry.getValue(), "getValue"));
147               });
148 
149               Assertions.assertAll("ComplexFields4 item", () -> {
150                 Map.Entry<String, Object> entry = iter.next();
151                 assertEquals("complex-field4-2", entry.getKey());
152                 assertEquals("complex-field4-2", reflectMethod(entry.getValue(), "getId2"));
153                 assertEquals("test-string6", reflectMethod(entry.getValue(), "getValue"));
154               });
155             });
156           } catch (NoSuchMethodException | SecurityException e) {
157             Assertions.fail(e);
158           }
159         });
160   }

187   @Test
188   void testExistsWithVariable() throws IOException, URISyntaxException, MetaschemaException {
189 
190     // IDocumentNodeItem moduleItem =
191     // IBindingContext.instance().newBoundLoader().loadAsNodeItem(
192     // new
193     // URI("https://raw.githubusercontent.com/usnistgov/OSCAL/main/src/metaschema/oscal_complete_metaschema.xml"));
194     //
195     // assert moduleItem != null;
196     //
197     // StaticContext staticContext = moduleItem..getStaticContext();
198 
199     IBindingMetaschemaModule module = new BindingModuleLoader(new DefaultBindingContext()).load(
200         new URI("https://raw.githubusercontent.com/usnistgov/OSCAL/main/src/metaschema/oscal_complete_metaschema.xml"));
201 
202     IDocumentNodeItem moduleItem = ObjectUtils.requireNonNull(module.getSourceNodeItem());
203     METASCHEMA moduleData = (METASCHEMA) moduleItem.getValue();
204     assert moduleItem != null;
205 
206     StaticContext staticContext = moduleItem.getStaticContext();
207 
208     DynamicContext dynamicContext = new DynamicContext(staticContext);
209     dynamicContext.setDocumentLoader(IBindingContext.instance().newBoundLoader());
210 
211     MetapathExpression importsMetapath = MetapathExpression.compile(
212         "for $import in /METASCHEMA/import return doc(resolve-uri($import/@href))/METASCHEMA",
213         staticContext);
214 
215     ISequence<?> imports = importsMetapath.evaluate(moduleItem, dynamicContext);
216 
217     MetapathExpression allImportsExpression = MetapathExpression.compile(
218         "recurse-depth(/METASCHEMA,'for $import in ./import return doc(resolve-uri($import/@href))/METASCHEMA')",
219         staticContext);
220 
221     ISequence<?> allImports = allImportsExpression.evaluate(moduleItem, dynamicContext);
222     allImports.getValue();
223 
224     MetapathExpression path = MetapathExpression.compile("exists($all-imports/define-assembly/root-name)",
225         staticContext);
226 
227     boolean result = ObjectUtils.requireNonNull(path.evaluateAs(
228         moduleItem,
229         MetapathExpression.ResultType.BOOLEAN,
230         dynamicContext.subContext().bindVariableValue(new QName("all-imports"), allImports)));
231 
232     assertTrue(result, "no root");
233   }

235   @Test
236   void codegenTest() throws MetaschemaException, IOException {
237     List<IConstraintSet> constraints;
238     {
239       BindingConstraintLoader constraintLoader = new BindingConstraintLoader(new DefaultBindingContext());
240 
241       constraints = constraintLoader.load(
242           Paths.get("../core/metaschema/schema/metaschema/metaschema-module-constraints.xml"));
243     }
244 
245     IBindingContext bindingContext = new DefaultBindingContext(
246         CollectionUtil.singletonList(new ExternalConstraintsModulePostProcessor(constraints)));
247     BindingModuleLoader loader = new BindingModuleLoader(bindingContext);
248 
249     IBindingMetaschemaModule module
250         = loader.load(Paths.get("../core/metaschema/schema/metaschema/metaschema-module-metaschema.xml"));
251 
252     Path compilePath = Paths.get("target/compile");
253     Files.createDirectories(compilePath);
254 
255     ClassLoader classLoader = ModuleCompilerHelper.newClassLoader(
256         compilePath,
257         ObjectUtils.notNull(Thread.currentThread().getContextClassLoader()));
258 
259     IProduction production = ModuleCompilerHelper.compileMetaschema(module, compilePath);
260     production.getModuleProductions().stream()
261         .map(item -> {
262           try {
263             return (Class<? extends IBoundModule>) classLoader.loadClass(item.getClassName().reflectionName());
264           } catch (ClassNotFoundException ex) {
265             throw new IllegalStateException(ex);
266           }
267         })
268         .forEachOrdered(clazz -> {
269           IBoundModule boundModule = bindingContext.registerModule(ObjectUtils.notNull(clazz));
270           // force the binding matchers to load
271           boundModule.getRootAssemblyDefinitions();
272         });
273   }
