6   package gov.nist.secauto.metaschema.databind.codegen.config;
7   
8   import static org.junit.jupiter.api.Assertions.assertEquals;
9   import static org.junit.jupiter.api.Assertions.assertNotNull;
10  
11  import gov.nist.secauto.metaschema.core.model.IModelDefinition;
12  import gov.nist.secauto.metaschema.core.model.IModule;
13  import gov.nist.secauto.metaschema.core.model.ModelType;
14  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
15  
16  import org.jmock.Expectations;
17  import org.jmock.junit5.JUnit5Mockery;
18  import org.junit.jupiter.api.Test;
19  import org.junit.jupiter.api.extension.RegisterExtension;
20  
21  import java.io.File;
22  import java.io.IOException;
23  import java.net.MalformedURLException;
24  import java.net.URI;
25  
26  class DefaultBindingConfigurationTest {
27    private static final URI METASCHEMA_LOCATION
28        = new File("src/test/resources/metaschema/metaschema.xml").getAbsoluteFile().toURI();
29    private static final String DEFINITION_NAME = "grandchild";
30    private static final ModelType DEFINITION_MODEL_TYPE = ModelType.ASSEMBLY;
31    private static final String DEFINITION__CLASS_NAME = "TheChild";
32  
33    @RegisterExtension
34    JUnit5Mockery context = new JUnit5Mockery();
35    private final IModelDefinition definition = context.mock(IModelDefinition.class);
36    private final IModule module = context.mock(IModule.class);
37  
38    @Test
39    void testLoader() throws MalformedURLException, IOException {
40      File bindingConfigFile = new File("src/test/resources/metaschema/binding-config.xml");
41  
42      DefaultBindingConfiguration config = new DefaultBindingConfiguration();
43      config.load(bindingConfigFile);
44  
45      assertEquals("gov.nist.itl.metaschema.codegen.xml.example.assembly",
46          config.getPackageNameForNamespace("http://csrc.nist.gov/ns/metaschema/testing/assembly"));
47  
48      context.checking(new Expectations() {
49        { // NOPMD - intentional
50          oneOf(module).getLocation();
51          will(returnValue(METASCHEMA_LOCATION));
52          allowing(definition).getContainingModule();
53          will(returnValue(module));
54          allowing(definition).getModelType();
55          will(returnValue(DEFINITION_MODEL_TYPE));
56          allowing(definition).getName();
57          will(returnValue(DEFINITION_NAME));
58        }
59      });
60      IDefinitionBindingConfiguration defConfig = config.getBindingConfigurationForDefinition(
61          ObjectUtils.notNull(definition));
62      assertNotNull(defConfig);
63      assertEquals(DEFINITION__CLASS_NAME, defConfig.getClassName());
64    }
65  
66  }
