27    @Test
28    void testOscalBindingModuleLoader() throws MetaschemaException, IOException {
29      BindingModuleLoader loader = new BindingModuleLoader(new DefaultBindingContext());
30      loader.set(DeserializationFeature.DESERIALIZE_XML_ALLOW_ENTITY_RESOLUTION, true);
31      IBindingMetaschemaModule module = loader.load(ObjectUtils.notNull(URI.create(
32          "https://raw.githubusercontent.com/usnistgov/OSCAL/main/src/metaschema/oscal_complete_metaschema.xml")));
33      IBindingContext context = IBindingContext.instance().registerModule(module, Paths.get("target/oscal-classes"));
34      assertAll(
35          () -> assertNotNull(module),
36          () -> assertNotNull(context));
