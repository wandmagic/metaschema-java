6   package gov.nist.secauto.metaschema.schemagen.json.impl.builder;
7   
8   import com.fasterxml.jackson.databind.node.ArrayNode;
9   import com.fasterxml.jackson.databind.node.ObjectNode;
10  
11  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
12  import gov.nist.secauto.metaschema.schemagen.json.IJsonGenerationState;
13  
14  import java.util.List;
15  
16  public class KeyedObjectBuilder
17      extends AbstractCollectionBuilder<KeyedObjectBuilder> {
18  
19    @Override
20    protected KeyedObjectBuilder thisBuilder() {
21      return this;
22    }
23  
24    @Override
25    public void build(
26        ObjectNode object,
27        IJsonGenerationState state) {
28      object.put("type", "object");
29  
30      if (getMinOccurrence() > 0) {
31        object.put("minProperties", getMinOccurrence());
32      }
33  
34      if (getMaxOccurrence() != -1) {
35        object.put("maxProperties", getMaxOccurrence());
36      }
37  
38      List<IType> types = getTypes();
39  
40      if (!types.isEmpty()) {
41        ObjectNode propertyNames = ObjectUtils.notNull(object.putObject("propertyNames"));
42        if (types.size() == 1) {
43          types.iterator().next().build(propertyNames, state);
44        } else {
45          ArrayNode anyOf = propertyNames.putArray("anyOf");
46          for (IType type : types) {
47            type.build(ObjectUtils.notNull(anyOf.objectNode()), state);
48          }
49        }
50      }
51  
52      ObjectNode patternProperties = ObjectUtils.notNull(object.putObject("patternProperties"));
53      ObjectNode wildcard = patternProperties.putObject("^.*$");
54      if (types.size() == 1) {
55        types.iterator().next().build(wildcard, state);
56      } else {
57        ArrayNode oneOf = wildcard.putArray("anyOf");
58        for (IType type : types) {
59          type.build(oneOf, state);
60        }
61      }
62    }
63  
64  }
