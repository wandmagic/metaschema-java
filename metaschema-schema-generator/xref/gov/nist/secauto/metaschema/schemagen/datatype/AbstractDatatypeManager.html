6   package gov.nist.secauto.metaschema.schemagen.datatype;
7   
8   import gov.nist.secauto.metaschema.core.datatype.IDataTypeAdapter;
9   
10  import java.util.Collections;
11  import java.util.HashMap;
12  import java.util.HashSet;
13  import java.util.LinkedHashMap;
14  import java.util.Map;
15  import java.util.Set;
16  
17  import edu.umd.cs.findbugs.annotations.NonNull;
18  
19  public abstract class AbstractDatatypeManager implements IDatatypeManager {
20    @NonNull
21    private static final Map<String, String> DATATYPE_TRANSLATION_MAP // NOPMD - intentional
22        = new LinkedHashMap<>();
23  
24    static {
25      DATATYPE_TRANSLATION_MAP.put("base64", "Base64Datatype");
26      DATATYPE_TRANSLATION_MAP.put("boolean", "BooleanDatatype");
27      DATATYPE_TRANSLATION_MAP.put("date", "DateDatatype");
28      DATATYPE_TRANSLATION_MAP.put("date-with-timezone", "DateWithTimezoneDatatype");
29      DATATYPE_TRANSLATION_MAP.put("date-time", "DateTimeDatatype");
30      DATATYPE_TRANSLATION_MAP.put("date-time-with-timezone", "DateTimeWithTimezoneDatatype");
31      DATATYPE_TRANSLATION_MAP.put("day-time-duration", "DayTimeDurationDatatype");
32      DATATYPE_TRANSLATION_MAP.put("decimal", "DecimalDatatype");
33      DATATYPE_TRANSLATION_MAP.put("email-address", "EmailAddressDatatype");
34      DATATYPE_TRANSLATION_MAP.put("hostname", "HostnameDatatype");
35      DATATYPE_TRANSLATION_MAP.put("integer", "IntegerDatatype");
36      DATATYPE_TRANSLATION_MAP.put("ip-v4-address", "IPV4AddressDatatype");
37      DATATYPE_TRANSLATION_MAP.put("ip-v6-address", "IPV6AddressDatatype");
38      DATATYPE_TRANSLATION_MAP.put("markup-line", "MarkupLineDatatype");
39      DATATYPE_TRANSLATION_MAP.put("markup-multiline", "MarkupMultilineDatatype");
40      DATATYPE_TRANSLATION_MAP.put("non-negative-integer", "NonNegativeIntegerDatatype");
41      DATATYPE_TRANSLATION_MAP.put("positive-integer", "PositiveIntegerDatatype");
42      DATATYPE_TRANSLATION_MAP.put("string", "StringDatatype");
43      DATATYPE_TRANSLATION_MAP.put("token", "TokenDatatype");
44      DATATYPE_TRANSLATION_MAP.put("uri", "URIDatatype");
45      DATATYPE_TRANSLATION_MAP.put("uri-reference", "URIReferenceDatatype");
46      DATATYPE_TRANSLATION_MAP.put("uuid", "UUIDDatatype");
47      DATATYPE_TRANSLATION_MAP.put("year-month-duration", "YearMonthDurationDatatype");
48    }
49  
50    @NonNull
51    private final Map<IDataTypeAdapter<?>, String> datatypeToTypeMap = new HashMap<>(); // NOPMD - intentional
52  
53    @SuppressWarnings("null")
54    @NonNull
55    protected static Map<String, String> getDatatypeTranslationMap() {
56      return Collections.unmodifiableMap(DATATYPE_TRANSLATION_MAP);
57    }
58  
59    @Override
60    public Set<String> getUsedTypes() {
61      return new HashSet<>(datatypeToTypeMap.values());
62    }
63  
64    @SuppressWarnings("null")
65    @Override
66    @NonNull
67    public String getTypeNameForDatatype(@NonNull IDataTypeAdapter<?> datatype) {
68      synchronized (this) {
69        String name = datatypeToTypeMap.get(datatype);
70        if (name == null) {
71          name = getDatatypeTranslationMap().get(datatype.getPreferredName().getLocalPart());
72          datatypeToTypeMap.put(datatype, name);
73        }
74        return name;
75      }
76    }
77  }
