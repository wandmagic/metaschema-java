6   package gov.nist.secauto.metaschema.schemagen.xml.datatype;
7   
8   import gov.nist.secauto.metaschema.core.model.IModule;
9   import gov.nist.secauto.metaschema.core.util.CollectionUtil;
10  import gov.nist.secauto.metaschema.core.util.ObjectUtils;
11  import gov.nist.secauto.metaschema.schemagen.datatype.IDatatypeContent;
12  
13  import org.jdom2.Element;
14  
15  import java.io.InputStream;
16  import java.util.List;
17  import java.util.Map;
18  import java.util.stream.Collectors;
19  
20  import edu.umd.cs.findbugs.annotations.NonNull;
21  
22  public abstract class AbstractXmlMarkupDatatypeProvider
23      extends AbstractXmlDatatypeProvider {
24  
25    @Override
26    protected InputStream getSchemaResource() {
27      return IModule.class.getResourceAsStream(getSchemaResourcePath());
28    }
29  
30    /**
31     * Get the absolute classpath of the schema resource.
32     *
33     * @return the resource path
34     */
35    @NonNull
36    protected abstract String getSchemaResourcePath();
37  
38    @Override
39    protected List<Element> queryElements(JDom2XmlSchemaLoader loader) {
40      return loader.getContent(
41          "/xs:schema/*",
42          CollectionUtil.singletonMap("xs", JDom2XmlSchemaLoader.NS_XML_SCHEMA));
43    }
44  
45    @NonNull
46    protected abstract String getDataTypeName();
47  
48    @Override
49    protected Map<String, IDatatypeContent> handleResults(@NonNull List<Element> items) {
50      String dataTypeName = getDataTypeName();
51      return CollectionUtil.singletonMap(
52          dataTypeName,
53          new JDom2DatatypeContent(
54              dataTypeName,
55              ObjectUtils.notNull(items.stream()
56                  .filter(element -> !"include".equals(element.getName()))
57                  .collect(Collectors.toList())),
58              CollectionUtil.emptyList()));
59    }
60  
61  }
