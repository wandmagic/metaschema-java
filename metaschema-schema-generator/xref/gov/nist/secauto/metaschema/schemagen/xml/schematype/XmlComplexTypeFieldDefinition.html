6   package gov.nist.secauto.metaschema.schemagen.xml.schematype;
7   
8   import gov.nist.secauto.metaschema.core.datatype.IDataTypeAdapter;
9   import gov.nist.secauto.metaschema.core.model.IFieldDefinition;
10  import gov.nist.secauto.metaschema.core.model.IFlagInstance;
11  import gov.nist.secauto.metaschema.schemagen.xml.XmlSchemaGenerator;
12  import gov.nist.secauto.metaschema.schemagen.xml.impl.XmlGenerationState;
13  
14  import javax.xml.namespace.QName;
15  import javax.xml.stream.XMLStreamException;
16  
17  import edu.umd.cs.findbugs.annotations.NonNull;
18  
19  public class XmlComplexTypeFieldDefinition
20      extends AbstractXmlComplexType<IFieldDefinition> {
21    public XmlComplexTypeFieldDefinition(
22        @NonNull QName qname,
23        @NonNull IFieldDefinition definition) {
24      super(qname, definition);
25    }
26  
27    @Override
28    protected void generateTypeBody(XmlGenerationState state) throws XMLStreamException {
29      IFieldDefinition definition = getDefinition();
30      IXmlSimpleType valueType = state.getSimpleType(definition);
31      IDataTypeAdapter<?> datatype = valueType.getDataTypeAdapter();
32  
33      String xmlContentType;
34      if (datatype.isXmlMixed()) {
35        xmlContentType = "complexContent"; // with attributes
36      } else {
37        xmlContentType = "simpleContent"; // without attributes
38      }
39      state.writeStartElement(XmlSchemaGenerator.PREFIX_XML_SCHEMA, xmlContentType, XmlSchemaGenerator.NS_XML_SCHEMA);
40      state.writeStartElement(XmlSchemaGenerator.PREFIX_XML_SCHEMA, "extension", XmlSchemaGenerator.NS_XML_SCHEMA);
41      state.writeAttribute("base", valueType.getTypeReference());
42  
43      for (IFlagInstance flagInstance : definition.getFlagInstances()) {
44        assert flagInstance != null;
45        generateFlagInstance(flagInstance, state);
46      }
47      state.writeEndElement(); // xs:extension
48  
49      state.writeEndElement(); // xs:simpleContent or xs:complexContent
50    }
51  }
