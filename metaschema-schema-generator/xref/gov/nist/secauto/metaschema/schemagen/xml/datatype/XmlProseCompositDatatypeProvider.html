6   package gov.nist.secauto.metaschema.schemagen.xml.datatype;
7   
8   import gov.nist.secauto.metaschema.schemagen.datatype.CompositeDatatypeProvider;
9   import gov.nist.secauto.metaschema.schemagen.datatype.IDatatypeContent;
10  import gov.nist.secauto.metaschema.schemagen.datatype.IDatatypeProvider;
11  
12  import org.codehaus.stax2.XMLStreamWriter2;
13  
14  import java.util.Collection;
15  import java.util.List;
16  import java.util.Set;
17  import java.util.stream.Collectors;
18  
19  import javax.xml.stream.XMLStreamException;
20  
21  import edu.umd.cs.findbugs.annotations.NonNull;
22  
23  public class XmlProseCompositDatatypeProvider
24      extends CompositeDatatypeProvider {
25  
26    private final XmlProseBaseDatatypeProvider proseBaseProvider = new XmlProseBaseDatatypeProvider();
27  
28    public XmlProseCompositDatatypeProvider(@NonNull List<IDatatypeProvider> proxiedProviders) {
29      super(proxiedProviders);
30    }
31  
32    @Override
33    public @NonNull Set<String> generateDatatypes(Set<String> requiredTypes,
34        @NonNull XMLStreamWriter2 writer) throws XMLStreamException {
35      Set<String> result = super.generateDatatypes(requiredTypes, writer);
36  
37      if (!result.isEmpty()) {
38        // apply core markup types
39        Collection<IDatatypeContent> datatypes = proseBaseProvider.getDatatypes().values();
40        Set<String> proseBaseTypes = datatypes.stream().map(content -> content.getTypeName()).collect(Collectors.toSet());
41        proseBaseProvider.generateDatatypes(proseBaseTypes, writer);
42      }
43      return result;
44    }
45  
46  }
