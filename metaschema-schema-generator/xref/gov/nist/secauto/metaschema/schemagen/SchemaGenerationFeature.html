6   package gov.nist.secauto.metaschema.schemagen;
7   
8   import gov.nist.secauto.metaschema.core.configuration.AbstractConfigurationFeature;
9   
10  import edu.umd.cs.findbugs.annotations.NonNull;
11  
12  /**
13   * Configuration options for schema generation.
14   *
15   * @param <V>
16   *          the feature value type
17   */
18  public final class SchemaGenerationFeature<V>
19      extends AbstractConfigurationFeature<V> {
20  
21    /**
22     * If enabled, definitions that are defined inline will be generated as inline
23     * types. If disabled, definitions will always be generated as global types.
24     */
25    @NonNull
26    public static final SchemaGenerationFeature<Boolean> INLINE_DEFINITIONS
27        = new SchemaGenerationFeature<>("inline-definitions", Boolean.class, false);
28  
29    /**
30     * If enabled, child definitions of a choice that are defined inline will be
31     * generated as inline types. If disabled, child definitions of a choice will
32     * always be generated as global types. This option will only be used if
33     * {@link #INLINE_DEFINITIONS} is also enabled.
34     */
35    @NonNull
36    public static final SchemaGenerationFeature<Boolean> INLINE_CHOICE_DEFINITIONS
37        = new SchemaGenerationFeature<>("inline-choice-definitions", Boolean.class, false);
38  
39    private SchemaGenerationFeature(
40        @NonNull String name,
41        @NonNull Class<V> valueClass,
42        @NonNull V defaultValue) {
43      super(name, valueClass, defaultValue);
44    }
45  
46  }
